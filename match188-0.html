import random

def alpha_beta_pruning(depth, is_maximizing_player, alpha, beta, path):
    if depth == 0:

      utility = random.choice([-1, 1])
      path.append(utility)
      return utility

    if is_maximizing_player:
        max_eval = float('-inf')
        for k in range(2):
            eval = alpha_beta_pruning(depth - 1, False, alpha, beta, path)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for l in range(2):
            eval = alpha_beta_pruning(depth - 1, True, alpha, beta, path)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break
        return min_eval

def simulate_game(starting_player):
    rounds = 0
    results = []
    current_player = starting_player

    while rounds < 5:
      path = []
      winner = alpha_beta_pruning(5, current_player == 1, float('-inf'), float('inf'), path)


      if winner == -1:
        round_winner = "Scorpion"
      else:
        round_winner = "Sub-Zero"
      results.append(round_winner)
      rounds += 1
      current_player = 1 - current_player

      scorpion_wins = results.count("Scorpion")
      subzero_wins = results.count("Sub-Zero")
      game_winner = "Scorpion" if scorpion_wins > subzero_wins else "Sub-Zero"

    return game_winner, rounds, results


starting_player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))
game_winner, total_rounds, round_winners = simulate_game(starting_player)
print(f"Game Winner: {game_winner}")

def get_dark_magic_cost():
    return int(input("Enter the cost of using dark magic (dark_magic_cost): "))

def get_outcome_values():
    return [3, 6, 2, 3, 7, 1, 2, 0]

def minimax_leaf(depth, outcomes):
    return outcomes.pop(0)

def alpha_beta_minimax(depth, is_maximizing_player, alpha, beta, outcomes, use_dark_magic=False):
    if depth == 3:
        return minimax_leaf(depth, outcomes)



    if is_maximizing_player:
        best_value = float('-inf')
        for v in range(2):
            eval_score = alpha_beta_minimax(depth + 1, False, alpha, beta, outcomes, use_dark_magic)
            best_value = max(best_value, eval_score)
            alpha = max(alpha, eval_score)
            if alpha >= beta:
                break
        return best_value
    else:
        if use_dark_magic:
            return alpha_beta_minimax(depth + 1, True, alpha, beta, outcomes, use_dark_magic)
        worst_value = float('inf')
        for m in range(2):
            eval_score = alpha_beta_minimax(depth + 1, True, alpha, beta, outcomes, use_dark_magic)
            worst_value = min(worst_value, eval_score)
            beta = min(beta, eval_score)
            if alpha >= beta:
                break
        return worst_value

def evaluate_without_dark_magic(outcome_values):
    outcomes_without_dark_magic = outcome_values[:]
    return alpha_beta_minimax(0, True, float('-inf'), float('inf'), outcomes_without_dark_magic)

def evaluate_with_dark_magic(outcome_values, dark_magic_cost):
    outcomes_with_dark_magic = outcome_values[:]
    result = alpha_beta_minimax(0, True, float('-inf'), float('inf'), outcomes_with_dark_magic, use_dark_magic=True)
    if result is None:
        return None
    return result - dark_magic_cost

def pacman_adventure():
    dark_magic_cost = get_dark_magic_cost()
    outcome_values = get_outcome_values()


    result_without_dark_magic = evaluate_without_dark_magic(outcome_values)
    result_with_dark_magic = evaluate_with_dark_magic(outcome_values, dark_magic_cost)

    print(f"The minimax value without dark magic is {result_without_dark_magic}.")

    if result_with_dark_magic is None:
        print(f"Error: Dark magic evaluation returned no valid result.")
    elif result_with_dark_magic > result_without_dark_magic:
        print(f"The minimax value with dark magic is {result_with_dark_magic}. Pacman uses dark magic.")
    else:
        print(f"The minimax value with dark magic is {result_with_dark_magic}. Pacman does not use dark magic.")


pacman_adventure()
