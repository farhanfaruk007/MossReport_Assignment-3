import random

#Task 1:

chosen_player = int(input("Choose 1 for sub-zero or 0 for scorpion: "))

def fight(chosen_player):
  round_number = 0
  track_winner_of_round = []
  player_for_this_round = chosen_player

  while round_number < 3:
    winner_of_round = gameTree(5, player_for_this_round == 1, float('-inf'), float('inf'))


    if winner_of_round == -1:
      winner = "Scorpion"
    else:
      winner = "Sub-Zero"
    track_winner_of_round.append(winner)
    round_number += 1
    player_for_this_round = 1 - player_for_this_round

  Scorpion = track_winner_of_round.count("Scorpion")
  Sub_Zero = track_winner_of_round.count("Sub-Zero")
  champion = "Scorpion" if Scorpion > Sub_Zero else "Sub-Zero"

  return champion, round_number, track_winner_of_round


def gameTree(max_depth, maximizer, alpha, beta):
  if max_depth == 0:
    utility_score = random.choice([1, -1])
    return utility_score

  if maximizer:
    current_max_score = float('-inf')
    for i in range(2):
      new_score = gameTree(max_depth - 1, False, alpha, beta)
      current_max_score = max(current_max_score, new_score)
      alpha = max(alpha, new_score)
      if alpha >= beta:
        break
    return current_max_score

  else:
    current_min_score = float('inf')
    for i in range(2):
      new_score = gameTree(max_depth - 1, True, alpha, beta)
      current_min_score = min(current_min_score, new_score)
      beta = min(beta, new_score)
      if alpha >= beta:
        break
    return current_min_score


game_winner, total_rounds, winner_of_rounds = fight(chosen_player)

print("Game winner: ", game_winner)
print("Total rounds: ", total_rounds)
for i, j in enumerate(winner_of_rounds, 1):
  print(f"Winner of Round {i}: {j}")

  leaf_nodes = [3,6,2,3,7,1,2,0]

  def minimax(depth, maximizer, alpha, beta, leaf_nodes, darkMagic = False):
    if depth == 3:
      return leaf_nodes.pop(0)



    if maximizer:
      current_max_score = float('-inf')
      for i in range(2):
        new_score = minimax(depth + 1, False, alpha, beta, leaf_nodes, darkMagic)
        current_max_score = max(current_max_score, new_score)
        alpha = max(alpha, new_score)
        if alpha >= beta:
          break
      return current_max_score
    else:
      if darkMagic:
        return minimax(depth + 1, True, alpha, beta, leaf_nodes, darkMagic)
      current_min_score = float('inf')
      for i in range(2):
        new_score = minimax(depth + 1, True, alpha, beta, leaf_nodes, darkMagic)
        current_min_score = min(current_min_score, new_score)
        beta = min(beta, new_score)
        if alpha >= beta:
          break
      return current_min_score

  dark_leaf_nodes = leaf_nodes[:]
  dark_utility_score = minimax(0, True, float('-inf'), float('inf'), dark_leaf_nodes, darkMagic = True) - c

  normal_leaf_nodes = leaf_nodes[:]
  normal_utility_score = minimax(0, True, float('-inf'), float('inf'), normal_leaf_nodes)

  print("Minimax value without dark magic: ", normal_utility_score)
  if dark_utility_score > normal_utility_score:
    print(f"Minimax value with dark magic is {dark_utility_score}. Pacman should use dark magic.")
