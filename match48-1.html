def prog(fh,o_fh):
    start_play=int(fh.readline().strip())
    # if start_play==1:
    #     start="subzero"
    # else:
    #     start="scorpion"
    # scorpion=2
    # subzero=1

    leaf_nodes=[-1,1,1,-1,1,1,1,-1] * 4  # 32 nodes



    def alphabeta(node,depth,max_play,alpha,beta):
        if depth==0 or node['leaf']:
            return node['value']
        if max_play:
            maxnode=float('-inf')
            for child in node['child']:
                val=alphabeta(child,depth-1,False,alpha,beta)
                maxnode=max(maxnode,val)
                alpha=max(alpha,val)
                if beta<=alpha:
                    break
            return maxnode
        else:
            minnode=float('inf')
            for child in node['child']:
                val=alphabeta(child,depth-1,True,alpha,beta)
                minnode=min(minnode,val)
                beta=min(beta,val)
                if beta<=alpha:
                    break
            return minnode

    def gametree(depth,index=0):
        if depth==0:
            return{'leaf': True,'value': leaf_nodes[index % len(leaf_nodes)]}
        else:
            return {'leaf': False,'child': [gametree(depth-1, index * 2),gametree(depth-1, index * 2 + 1),]}

    roundoutcome =[]
    rounds=len(roundoutcome)
    start=start_play
    while rounds<3:
        rounds+=1
        mortalgame=gametree(5)
        result=alphabeta(mortalgame,5,start,float('-inf'),float('inf'))
        if result==-1:
            roundoutcome.append("scorpion")
        else:
            roundoutcome.append("subzero")
        if start==0:
            start=1
        else:
            start=0  #interchange


    scorpionwinner=roundoutcome.count("scorpion")
    subzerowinner=roundoutcome.count("subzero")
    winner=""
    if scorpionwinner>subzerowinner:
        winner="scorpion"
    else:
        winner="subzero"



    o_fh.write(f"Game Winner: {'Scorpion' if winner == 'scorpion' else 'Sub-Zero'}\n")

    o_fh.write(f"Total rounds played: {rounds}\n")
    index = 0
    for winner in roundoutcome:
        o_fh.write(f"Winner of round {index + 1}: {'Scorpion' if winner == 'scorpion' else 'Sub-Zero'}\n")
        index += 1


fh = open("/content/input.txt", "r")
o_fh = open("/content/output.txt", "w")

def pacman_game(c):
    leaf_nodes = [3,6,2,3,7,1,2,0]

    def alphabeta(node_index,depth,max_play,alpha,beta):
        if depth==0:
            return leaf_nodes[node_index]
        if max_play:  # pacmanturn
            maxnode=float('-inf')
            for i in range(2):  # two children
                val=alphabeta(node_index*2 + i,depth-1,False,alpha,beta)
                maxnode=max(maxnode,val)
                alpha=max(alpha,val)
                if beta<=alpha:
                    break


            return maxnode
        else:
            minnode=float('inf')
            for i in range(2):
                val=alphabeta(node_index*2+i,depth-1,True,alpha,beta)
                minnode=min(minnode,val)
                beta=min(beta,val)
                if beta<=alpha:
                    break
            return minnode


    root_minimax=alphabeta(0, 3,True,float('-inf'),float('inf'))
    left_subtree_best=max(leaf_nodes[0],leaf_nodes[1])
    right_subtree_best=max(leaf_nodes[2],leaf_nodes[3])
    left_magic_score=left_subtree_best-c
    right_magic_score=right_subtree_best-c

    if max(left_magic_score,right_magic_score)>root_minimax:
        if left_magic_score>right_magic_score:
            strategy = f"The new minimax value is {left_magic_score}. Pacman goes left and use black magic."
        else:
            strategy = f"The newminimax value is {right_magic_score}. Pacman goes right and use black magic."
    else:
        strategy = f"The minimax value is {root_minimax}. Pacman does not use dark magic."

    print(strategy)
