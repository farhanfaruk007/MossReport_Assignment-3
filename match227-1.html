def alpha_beta_minimax(level, position, is_max_turn, leaf_scores, max_score, min_score):

    if level == 3:
        return leaf_scores[position]

    if is_max_turn:
        best_score = float('-inf')
        for i in range(2):
            child_score = alpha_beta_minimax(level + 1, position * 2 + i, False, leaf_scores, max_score, min_score)
            best_score = max(best_score, child_score)
            max_score = max(max_score, child_score)
            if min_score <= max_score:
                break
        return best_score
    else:
        best_score = float('inf')
        for i in range(2):
            child_score = alpha_beta_minimax(level + 1, position * 2 + i, True, leaf_scores, max_score, min_score)


            best_score = min(best_score, child_score)
            min_score = min(min_score, child_score)
            if min_score <= max_score:
                break
        return best_score

import random


class Node:


    def __init__(self, fighter, depth):
        self.fighter = fighter
        self.depth = depth
        self.children = []
        self.score = None

    def is_leaf(self):

        return self.depth == 0

    def generate_score(self):

        if self.score is None:
            self.score = random.choice([-1, 1])
        return self.score

    def create_children(self):

        if not self.children:
            for _ in range(2):
                next_fighter = 1 - self.fighter
                self.children.append(Node(next_fighter, self.depth - 1))
        return self.children


def alpha_beta_pruning(node, depth, best_for_max, best_for_min, is_max_turn):

    if depth == 0 or node.is_leaf():
        return node.generate_score()

    if is_max_turn:
        for child in node.create_children():
            score = alpha_beta_pruning(child, depth - 1, best_for_max, best_for_min, False)
            best_score = max(best_score, score)
            best_for_max = max(best_for_max, score)
            if best_for_max >= best_for_min:
                break
        return best_score
    else:
        best_score = float('inf')
        for child in node.create_children():
            score = alpha_beta_pruning(child, depth - 1, best_for_max, best_for_min, True)
            best_score = min(best_score, score)
            best_for_min = min(best_for_min, score)
            if best_for_max >= best_for_min:
                break
        return best_score


def play_round(starting_fighter):

    root_node = Node(starting_fighter, 5)
    final_score = alpha_beta_pruning(root_node, 5, float('-inf'), float('inf'), starting_fighter == 0)
    winner = 'Scorpion' if final_score == -1 else 'Sub-Zero'
    return winner


def play_full_game(starting_fighter):
    """Play the entire game, consisting of multiple rounds, and determine the overall winner."""
    current_fighter = starting_fighter
    rounds_played = 0
    round_winners = {'Scorpion': 0, 'Sub-Zero': 0}
    round_results = []

    while round_winners['Scorpion'] < 3 and round_winners['Sub-Zero'] < 3:
        rounds_played += 1
        round_winner = play_round(current_fighter)
        round_results.append(f"Winner of Round {rounds_played}: {round_winner}")
        round_winners[round_winner] += 1
        current_fighter = 1 - current_fighter

    overall_winner = 'Scorpion' if round_winners['Scorpion'] == 3 else 'Sub-Zero'


    return overall_winner, rounds_played, round_results


def main():
    start_player = int(input("Enter which player will start first: ").strip())
    if start_player != 0 and start_player != 1:
        print("Invalid Input")
        return
    winner, total_round, round_winners = play_full_game(start_player)

    print(f"Game Winner: {winner}")

    for round_num, round_winner in enumerate(round_winners, start=1):
        print(f"Winner of Round {round_num}: {round_winner}")

if __name__ == '__main__':
