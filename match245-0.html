class mortal_kombat:
    def __init__(self, cntestnt):
        self.cntestnt = cntestnt
        self.bran = 2
        self.played_times = 0
        self.vict_Scorp = 0
        self.wins = 0
        self.rounds = []
        self.max_d = 5
        self.val_of_utility = [-1, 1]

    def game_play(self):

        current_cntestnt = self.cntestnt

        max_rnds = 3

        while self.played_times < max_rnds:

            self.played_times += 1

            champions = self.AL_BT_PRUN(0, current_cntestnt, float('-inf'), float('inf'), True)

            if champions == -1:


                self.rounds.append("Scorpion")

                self.vict_Scorp += 1
            else:
                self.rounds.append("Sub-Zero")
                self.wins += 1

            if self.vict_Scorp == 2:
                print("Scorpion wins the games_played by winning 2 rounds!")
                break
            elif self.wins == 2:
                print("Sub-Zero wins the games_played by winning 2 rounds!")
                break

            current_cntestnt = 1 - current_cntestnt

        if self.vict_Scorp > self.wins:
          victories_by = "Scorpion"
          return victories_by
        else:
          victories_by = "Sub-Zero"
          return victories_by

    def AL_BT_PRUN(self, dep, players_of_game, AL_value, BT_value, max_contstnt):
        if dep == self.max_d:
            return self.val_of_utility[players_of_game]

        if max_contstnt:
            max_evaluation = float('-inf')
            for s in range(self.bran):
                game_evaluation = self.AL_BT_PRUN(dep + 1, players_of_game, AL_value, BT_value, False)
                max_evaluation = max(max_evaluation, game_evaluation)
                AL_value = max(AL_value, game_evaluation)
                if BT_value <= AL_value:
                    break
            return max_evaluation

        else:
            min_evaluation = float('inf')
            for y in range(self.bran):
                game_evaluation = self.AL_BT_PRUN(dep + 1, 1 - players_of_game, AL_value, BT_value, True)


                min_evaluation = min(min_evaluation, game_evaluation)
                BT_value = min(BT_value, game_evaluation)
                if BT_value <= AL_value:
                    break
            return min_evaluation

cntestnt = int(input("who is the init players_of_game? (0 for Scorpion, 1 for Sub-Zero): "))
games_played = mortal_kombat(cntestnt)



game_winner = games_played.game_play()

print(f"Game champions: {game_winner}")
print(f"Total Rounds Played: {games_played.played_times}")
for count, champions in enumerate(games_played.rounds, 1):
    print(f"Winner of the Round {count}: {champions}")

def ghost_pocman(x):

    leaf_vals = [3, 6, 2, 3, 7, 1, 2, 0]

    def maximum_minimum(dep, index_of_values, players_max, values_that_given, AL, BT):
        if dep == 3:
            return values_that_given[index_of_values]
        if players_max:
            holders_at_max = float('-inf')

            for y in range(2):

                evaluation_ev = maximum_minimum(dep + 1, index_of_values * 2 + y, False, values_that_given, AL, BT)
                holders_at_max = max(holders_at_max, evaluation_ev)

                AL = max(AL, evaluation_ev)

                if BT <= AL:
                    break
            return holders_at_max
        else:
            holders_at_min = float('inf')
            for y in range(2):
                evaluation_ev = maximum_minimum(dep + 1, index_of_values * 2 + y, True, values_that_given, AL, BT)
                holders_at_min = min(holders_at_min, evaluation_ev)
                BT = min(BT, evaluation_ev)
                if BT <= AL:
                    break
            return holders_at_min

    maximum_minimum_values = maximum_minimum(0, 0, True, leaf_vals, float('-inf'), float('inf'))

    L_max_sub = max(leaf_vals[0:4])
    R_max_sub = max(leaf_vals[4:8])
    L_magic_utilized = L_max_sub - x
    R_magic_utilized = R_max_sub - x

    if L_magic_utilized > R_magic_utilized and L_magic_utilized > maximum_minimum_values:
        print(f"The new value of minimax is {L_magic_utilized}. Pacman goes left and uses dark magic")
    elif R_magic_utilized > L_magic_utilized and R_magic_utilized > maximum_minimum_values:
        print(f"The new value of minimax is {R_magic_utilized}. Pacman goes right and uses dark magic")
    else:
        print(f"The minimax value is {maximum_minimum_values}. Pacman does not use dark magic")
