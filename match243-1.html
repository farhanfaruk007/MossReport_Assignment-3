import random
def a_b(val, d, alp, bta, curr, idx, branch_fact=2):
    if d == 5:
        return val[idx]

    if curr == 0:
        temp = float('-inf')
        for i in range(branch_fact):
            temp = max(temp, a_b(val, d + 1, alp, bta, 1, idx * branch_fact + i, branch_fact)) 
            alp = max(alp, temp)
            if alp >= bta:
                break
        return temp
    else: 
        temp = float('inf')
        for i in range(branch_fact):
            temp = min(temp, a_b(val, d + 1, alp, bta, 0, idx * branch_fact + i, branch_fact)) 
            bta = min(bta, temp)
            if bta <= alp:
                break
        return temp

def mor_komb():
    start = int(input("Who starts the game: "))
    val = [random.choice([-1, 1]) for k in range(32)]
    result = []
    curr = start

    for round_no in range(1, 4):
        win_val = a_b(val, 0, -float('inf'), float('inf'), curr, 0)
        if win_val == -1:
            win_name = "Scorpion"
        else:
            win_name = "Sub-Zero"
        result.append((round_no, win_name))
        curr = 1 - curr
    last_r = result[-1]
    ult_winner = last_r[1]
    print(f"Game Winner: {ult_winner}")
    print("Total Rounds Played: 3")
    for round_no, win_name in result:
        print(f"Winner of Round {round_no}: {win_name}")
