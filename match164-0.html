class Mortal_Kombat:
    def __init__(self, starting_player):
        self.starting_player= starting_player
        self.rounds= 0
        self.winner= None
        self.round_winners= []

    def prune(self, depth, node_idx, is_maximizing, alpha, beta):
        if depth == 5:
            return self.leaf_node(node_idx)
        if is_maximizing:
            best_val = float('-infinity')
            for branch in range(2):
                value= self.prune(depth+1, node_idx*2+branch, False, alpha, beta)
                best_val= max(best_val, value)
                alpha= max(alpha, value)
                if beta <= alpha:
                    break
            return best_val
        else:
            best_val= float('infinity')
            for branch in range(2):
                value= self.prune(depth+1, node_idx*2+branch, True, alpha, beta)
                best_val= min(best_val, value)
                beta_val= min(beta, value)
                if beta <= alpha:
                    break
            return best_val

    @staticmethod
    def leaf_node(index):
        leaf_node= [1, -1] * 16
        return leaf_node[index]

    def round(self):
        current_player= self.starting_player
        while len(self.round_winners) < 3:
            self.rounds+= 1
            winner= self.prune(0, 0, current_player==1, float('-infinity'), float('infinity'))


            if winner== -1:
                self.round_winners.append("Scorpion")
                current_player= 1
            else:
                self.round_winners.append("Sub-Zero")
                current_player= 0
        if self.round_winners.count("Scorpion") > self.round_winners.count("Sub-Zero"):
            self.winner= "Scorpion"
        else:


            self.winner= "Sub-Zero"

    def result(self):
        return {
            "Game Winner": self.winner, "Total Rounds Played": self.rounds, "Round Winners": self.round_winners
        }
