import random

def create_leaf_values():
    return [random.choice([-1, 1]) for _ in range(child_count ** tree_depth)]

def alpha_beta_pruning(values, depth, pos, is_max_turn, a, b):


    if depth == tree_depth:
        return values[pos]
    if is_max_turn:
        max_val = float('-inf')
        for i in range(child_count):
            val = alpha_beta_pruning(values, depth + 1, pos * child_count + i, False, a, b)
            max_val = max(max_val, val)
            a = max(a, val)
            if b <= a:
                break
        return max_val
    else:
        min_val = float('inf')
        for i in range(child_count):
            val = alpha_beta_pruning(values, depth + 1, pos * child_count + i, True, a, b)
            min_val = min(min_val, val)
            b = min(b, val)
            if b <= a:
                break
        return min_val

starter = int(input("Enter starting player (1 for Scorpion, 0 for Sub-Zero): "))
child_count = 2
tree_depth = 5
round_count = 3
game_outcomes = []

is_scorpion_turn = True if starter == 1 else False

for current_round in range(round_count):
    leaf_nodes = create_leaf_values()
    outcome = alpha_beta_pruning(leaf_nodes, 0, 0, is_scorpion_turn, float('-inf'), float('inf'))
    game_outcomes.append(outcome)
    is_scorpion_turn = not is_scorpion_turn



sub_zero_wins = game_outcomes.count(-1)
scorpion_wins = game_outcomes.count(1)

if sub_zero_wins > scorpion_wins:
    print("Game Winner: Sub-Zero")
elif scorpion_wins > sub_zero_wins:
    print("Game Winner: Scorpion")
else:
    print("The game ended in a Tie!")

print(f"Total Rounds Played: {round_count}")
for idx, result in enumerate(game_outcomes, 1):
    winner_name = "Sub-Zero" if result == -1 else "Scorpion"
    print(f"Winner of Round {idx}: {winner_name}")

def shadow_force_pruning(power, terminal_values):
    def inner_pruning(depth, index, is_max_turn, lower_bound, upper_bound, terminal_values, max_depth):
        if depth == max_depth:
            return terminal_values[index]

        if is_max_turn:
            best_score = float('-inf')
            for j in range(2):
                current_score = inner_pruning(depth + 1, index * 2 + j, False, lower_bound, upper_bound, terminal_values, max_depth)
                best_score = max(best_score, current_score)
                lower_bound = max(lower_bound, current_score)
                if upper_bound <= lower_bound:
                    break
            return best_score
        else:
            best_score = float('inf')
            for j in range(2):
                current_score = inner_pruning(depth + 1, index * 2 + j, True, lower_bound, upper_bound, terminal_values, max_depth)
                best_score = min(best_score, current_score)
                upper_bound = min(upper_bound, current_score)
                if upper_bound <= lower_bound:
                    break
            return best_score

    max_depth = 3
    base_minimax = inner_pruning(0, 0, True, float('-inf'), float('inf'), terminal_values, max_depth)

    left_force_value = max(terminal_values[0:4])
    right_force_value = max(terminal_values[4:8])

    altered_left = left_force_value - power
    altered_right = right_force_value - power

    enhanced_value = max(altered_left, altered_right)
    chosen_path = "Pacman goes left and uses dark magic" if enhanced_value == altered_left else "Pacman goes right and uses dark magic"

    if enhanced_value > base_minimax:
        return f"The new minimax value is {enhanced_value}. {chosen_path}"
    else:
        return f"The new minimax value is {base_minimax}. Pacman does not use dark magic"

final_values = [3, 6, 2, 3, 7, 1, 2, 0]

outcome = shadow_force_pruning(force_power, final_values)
