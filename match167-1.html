def alpha_beta_pruning(depth, max_p, a, b, current_player):
  if depth==0:
    return random.choice([-1, 1])

  if max_p==True:


    max_node=float('-inf')
    for i in range(2):  # for branching factr is 2 given
      value=alpha_beta_pruning(depth-1,False, a,b,1-current_player)
      max_node=max(max_node,value)
      a=max(a, value)
      if b<=a:  # The branch is pruned if this condition mile jai
        break
    return max_node

  else:
    min_node=float('inf')
    for i in range(2):  #Branching factor 2 given
      value=alpha_beta_pruning(depth-1,True,a,b,1-current_player)
      min_node=min(min_node, value)
      b=min(b, value)
      if b<=a:  # The branch is pruned if this condition mile jai
        break
    return min_node

def mortal_kombat(satrt_player):
  curr_player=satrt_player
  r=[]
  for i in range(3):
    winner=alpha_beta_pruning(5, curr_player, float('-inf'), float('inf'), curr_player)
    if winner==-1:
      r.append("Scorpion")
    else:
      r.append("Sub-Zero")
    curr_player=1-curr_player  # To alternate turns after each move is played

  scorpion_win=r.count("Scorpion")
  subzero_win=r.count("Sub-Zero")
  if scorpion_win>subzero_win:
    winner="Scorpion"
  else:
    winner="Sub-Zero"

  #final results
  print(f"game Winner:{winner}")
  print(f"total rounds played:{len(r)}")

  for i in range(len(r)):
    print(f"winner of round {i+1}:{r[i]}")

mortal_kombat(strt_player)

 if max_p:

   a=max(a,max_val)

 else:

def with_magic(cost):
 left=max(minimax(2,0,True,-float('inf'),float('inf')),minimax(2,1,True,-float('inf'),float('inf')))
 left_magic=left-cost
 right=max(minimax(2,2,True,-float('inf'),float('inf')),minimax(2,3,True,-float('inf'),float('inf')))


 right_magic=right-cost
 return max(left_magic,right_magic)

def pacman_game(cost):
