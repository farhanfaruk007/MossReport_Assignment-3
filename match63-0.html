class Node:
    def __init__(self, value):

        self.children_list = []

    def check_leaf(self):
        if self.value is not None:


            return True
        else:
            return False

def create_tree(max_dep, curr_dep):
    node = Node(None)
    if curr_dep == max_dep:
        value = choice([-1, 1])
        node.value = value
        return node
    else:
        node.children_list.append(create_tree(max_dep, curr_dep+1))
        node.children_list.append(create_tree(max_dep, curr_dep+1))
        return node

def alpha_Beta_Pruning(node, alpha, beta, curr_dep, max_pl):
    if curr_dep == 0 and node.check_leaf() == True:
        return node.value

    if max_pl == True:
        max_value = -1e7


        for i in node.children_list:
            value = alpha_Beta_Pruning(i, alpha, beta, curr_dep - 1, False)
            max_value = max(max_value, value)
            alpha = max(alpha, value)
            if alpha >= beta:
                break
        return max_value
    else:
        min_value = 1e7
        for i in node.children_list:
            value = alpha_Beta_Pruning(i, alpha, beta, curr_dep - 1, True)
            min_value = min(min_value, value)
            beta = min(beta, value)
            if beta <= alpha:
                break
        return min_value

def pacman_game(c):
    root = tree_constructor()


    root_value_without_magic = alpha_Beta_Pruning(root, -1e7, 1e7, 3, True)

    left_tree_root = root.children_list[0]
    right_tree_root = root.children_list[1]
    left_tree_maxval = max(alpha_Beta_Pruning(left_tree_root.children_list[0], -1e7, 1e7, 1, True),
                           alpha_Beta_Pruning(left_tree_root.children_list[1], -1e7, 1e7, 1, True))
    right_tree_maxval = max(alpha_Beta_Pruning(right_tree_root.children_list[0], -1e7, 1e7, 1, True),
                           alpha_Beta_Pruning(right_tree_root.children_list[1], -1e7, 1e7, 1, True))

    rightval_darkmagic = right_tree_maxval - c
    leftval_darkmagic = left_tree_maxval - c

    max_value_darkmagic = None
    subtree = None
    if  rightval_darkmagic > leftval_darkmagic:
        max_value_darkmagic = rightval_darkmagic
        subtree = "right"
    else:
        max_value_darkmagic = leftval_darkmagic
        subtree = "left"

    if root_value_without_magic < max_value_darkmagic:
        if subtree == "right":

        else:
            print(f"The new minimax value is {max_value_darkmagic}. Pacman goes left and uses dark magic.")
    else:
        print(f"The minimax value is {root_value_without_magic}. Pacman does not use dark magic.")
