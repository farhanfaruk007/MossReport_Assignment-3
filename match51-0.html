def pacman_game(c):

    leaf_nodes=[3,6,2,3,7,1,2,0]

    def minimax_without_dark_magic():
      left_min1=min(leaf_nodes[0:2])
      left_min2=min(leaf_nodes[2:4])
      left_max=max(left_min1,left_min2)


      right_min1=min(leaf_nodes[4:6])
      right_min2=min(leaf_nodes[6:8])
      right_max=max(right_min1,right_min2)

      return max(left_max,right_max)

    def alpha_beta_pruning():
      alpha=float('-inf')
      beta=float('inf')

      left_min1=min(leaf_nodes[0:2])
      left_min2=min(leaf_nodes[2:4])
      left_max=max(left_min1,left_min2)
      alpha=max(alpha,left_max)

      right_min1=min(leaf_nodes[4:6])
      if right_min1<=alpha:
          return alpha
      right_min2=min(leaf_nodes[6:8])
      right_max=max(right_min1,right_min2)
      alpha=max(alpha,right_max)

      return alpha

    left_magic=max(leaf_nodes[0:4])-c
    right_magic=max(leaf_nodes[4:8])-c

    without_magic=minimax_without_dark_magic()
    if left_magic > without_magic and left_magic>right_magic:
      return f"The new minimax value is {left_magic}. Pacman goes left and uses dark magic."

      return f"The new minimax value is {right_magic}.Pacman goes right and uses dark magic."

    else:
      return f"The miniamax value is {without_magic}.pacman does not use dark magic."
