#AlphaBetaPruning

import random
import math

def alpha_beta_pruning(tree, node, depth, alpha_val, beta_val, is_maximizing_player):
    if depth == 0:
        return tree[node]
    if is_maximizing_player:
        max_val = float('-inf')
        for i in range(2):
            recursive_call = alpha_beta_pruning(tree, node*2+i, depth-1, alpha_val, beta_val, False)
            max_val = max(max_val, recursive_call)
            alpha_val = max(max_val, alpha_val)
            if alpha_val >= beta_val:
                break
        return max_val
    else:
        min_val = float('inf')
        for i in range(2):
            recursive_call = alpha_beta_pruning(tree, node*2+i, depth-1, alpha_val, beta_val, True)
            min_val = min(min_val, recursive_call)
            beta_val = min(min_val, beta_val)
            if alpha_val >= beta_val:
                break
        return min_val
