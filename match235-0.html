#Task-01
import random
scorpion = 0
sub_zero = 1

def alphabeta_pruning(node, depth, alpha, beta, maxi):
    if depth == 0 or leaf(node):
        return value_check(node)

    if maxi:
        max_v = float('-inf')
        for each_child in create_child(node):
            value = alphabeta_pruning(each_child, depth - 1, alpha, beta, False)
            max_v = max(max_v, value)
            alpha = max(alpha, value)
            if beta <= alpha:
                break
        return max_v
    else:
        min_v = float('inf')
        for each_child in create_child(node):
            value = alphabeta_pruning(each_child, depth - 1, alpha, beta, True)
            min_v = min(min_v, value)
            beta = min(beta, value)
            if beta <= alpha:
                break
        return min_v

def leaf(node):
    return node['depth'] == 0

def value_check(node):
    return random.choice([-1, 1])

def create_child(node):
    return [{'depth': node['depth'] - 1}, {'depth': node['depth'] - 1}]

def MortalKombat_game(starting_player):
    rounds = 0
    cur_player = starting_player
    winners = []
    while rounds < 3:
        winner = alphabeta_pruning({'depth': 5}, 5, float('-inf'), float('inf'), cur_player)
        if winner == -1:
            winners.append("Scorpion")
        else:
            winners.append("Sub-Zero")

        #changing player
        if cur_player == sub_zero:
            cur_player = scorpion
        else:
            cur_player = sub_zero


        rounds += 1

    game_winner = max(set(winners), key=winners.count)

    print(f"Game Winner: {game_winner}.")
    print(f"Total Rounds Played: {rounds}")
    for round, winner in enumerate(winners):
        print(f"Winner of Round {round+ 1}: {winner}")


start_player = int(input("Enter the starting player (0 for Scorpion, 1 for Sub-Zero): "))

MortalKombat_game(start_player)
