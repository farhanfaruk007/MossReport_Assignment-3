import random



def mini_max(level, index, max_player, points, alpha, beta, branch, rounds):

    if level == branch: #Maximum depth
        return points[index], rounds

    if max_player:  # Sub-Zero's turn
        maximum_man = float('-inf')
        gg = []

        for x in range(2):  # B.F. = 2
            val, gg_round = mini_max(level + 1, index * 2 + x, False, points, alpha, beta, branch, rounds)
            if val > maximum_man:
                maximum_man = val
                gg = gg_round

            alpha = max(alpha, maximum_man)

            if beta <= alpha:
                break

        if level < 3:
            gg.append("Sub-Zero")

        return maximum_man, gg

    else:  # Scorpion's turn
        minimum_man = float('inf')
        gg = []

        for x in range(2):  # B.F = 2
            val, gg_round = mini_max(level + 1, index * 2 + x, True, points, alpha, beta, branch, rounds)
            if val < minimum_man:
                minimum_man = val
                gg = gg_round

            beta = min(beta, minimum_man)

            if beta <= alpha:
                break

        if level < 3:
            gg.append("Scorpion")

        return minimum_man, gg

  if level == 3:  #Depth = 3
    return opx[idx]

  if max_player:
    maximum_maan = float('-inf')

    for x in range(2):

      val = mini_max(level + 1, idx * 2 + x, False, alpha, beta)

      maximum_maan = max(maximum_maan, val)

      alpha = max(alpha, maximum_maan)

      if beta <= alpha:
        break

    return maximum_maan



  else:
    minimum_man = float('inf')

    for i in range(2):
      val = mini_max(level + 1, idx * 2 + i, True, alpha, beta)



      minimum_man = min(minimum_man,val)

      beta = min(beta, minimum_man)

      if beta <= alpha:
        break


    return minimum_man

def minimax_dark_magic(c):


  left = max(mini_max(2, 0, True, -float('inf'), float('inf')),
             mini_max(2, 1, True, -float('inf'), float('inf')))
  left_magic = left - c


  right = max(mini_max(2, 2, True, -float('inf'), float('inf')),
              mini_max(2, 3, True, -float('inf'), float('inf')))
  right_magic = right - c

  return max(left_magic,right_magic)



def pacman_game(c):

  minimax_value = mini_max(0, 0, True, -float('inf'), float('inf'))



  darkmagic_value = minimax_dark_magic(c)



  if darkmagic_value > minimax_value:
      print(f"The new minimax value  is {darkmagic_value}"
            f". Pacman goes right and uses dark magic")

  else:
      print(f"The new minimax value  is {minimax_value}"
            f". Pacman does not use dark magic.")




input_c = int(input("Enter the cost of using dark magic (c): "))
