def alpha_beta_pruning(maximizing_player, beta_value, alpha_value, current_depth, current_node):
  leaf_nodes = [random.choice([scorpion_score, subzero_score]) for i in range(2 ** 5)]
  print(leaf_nodes)

  if current_depth == max_depth:
    return leaf_nodes[current_node]

  if maximizing_player:
    max_value = float("-inf")

    for child_index in child_indices[current_node]:
      new_value = alpha_beta_pruning(False, beta_value, alpha_value, current_depth + 1, child_index)
      max_value = max(max_value, new_value)
      alpha_value = max(alpha_value, new_value)

      if alpha_value >= beta_value:
        break

    return max_value

  else:
    min_value = float("inf")

    for child_index in child_indices[current_node]:
      new_value = alpha_beta_pruning(True, beta_value, alpha_value, current_depth + 1, child_index)
      min_value = min(min_value, new_value)
      beta_value = min(beta_value, new_value)

      if alpha_value >= beta_value:
        break

    return min_value
