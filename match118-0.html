def leaf_values():
  return [-1, 1, 1, -1, 1, -1, -1, 1]



def alpha_beta_pruning(depth, index, player, alpha, beta, value):
  if depth == 3:
    return value[index]
  if player:
    maxval = float('-inf')
    for i in range(0, 2):
        n_depth = depth + 1
        n_node = index * 2 + i
        eval = alpha_beta_pruning(n_depth, n_node, False, alpha, beta, value)
        maxval = max(maxval, eval)
        alpha = max(alpha, eval)
        if alpha >= beta:
          break
    return maxval
  else:
    minval = float('inf')
    for i in range(0, 2):
        n_depth = depth + 1
        n_node = index * 2 + i
        eval = alpha_beta_pruning(n_depth, n_node, True, alpha, beta, value)
        minval = min(minval, eval)
        beta = min(beta, eval)
        if alpha >= beta:
          break
    return minval

def simulate_game(start):


  values = leaf_values()
  rounds = 3
  round_winner = []
  current_player = start
  for round_num in range(rounds):
    player = current_player == 1
    winner = alpha_beta_pruning(0, 0, player, float('-inf'), float('inf'), values)
    if winner == -1:
      round_winner.append('Scorpion')
      current_player = 1
    else:
      round_winner.append('Sub-Zero')
      current_player = 0

  scorpion = round_winner.count('Scorpion')
  sub_zero = round_winner.count('Sub-Zero')
  if scorpion > sub_zero:
    game_winner = 'Scorpion'
  elif scorpion < sub_zero:
    game_winner = 'Sub-Zero'
  else:
    game_winner = 'Draw'

  print(f'Game Winner: {game_winner}')
  print(f'Total Rounds Played: {rounds}')

  counter = 1
  for winner in round_winner:
      print(f'Winner of Round {counter}: {winner}')
      counter += 1

start = int(input())
simulate_game(start)

def pacman_game(c):
  left_left=[3,6]
  left_right=[2,3]
  right_left=[7,1]
  right_right=[2,0]
  leaf_nodes=[left_left,left_right,right_left,right_right]
  left_first_choice=leaf_nodes[0][0]
  left_second_choice=leaf_nodes[1][0]
  right_first_choice=leaf_nodes[2][0]
  right_second_choice=leaf_nodes[3][0]
  left_ghost_choice=min(left_first_choice,left_second_choice)
  right_ghost_choice=min(right_first_choice,right_second_choice)
  dark_magic=max(left_ghost_choice,right_ghost_choice)
  left_dark_magic=max(leaf_nodes[0][1],leaf_nodes[1][1])-c
  right_dark_magic=max(leaf_nodes[2][1],leaf_nodes[3][1])-c
  fit_dark_magic=max(left_dark_magic,right_dark_magic)
  if fit_dark_magic>dark_magic:
    if fit_dark_magic!=left_dark_magic:
      print(f"The new minimax value is {fit_dark_magic}. Pacman goes right and uses dark magic")
    else:
      print(f"The new minimax value is {fit_dark_magic}. Pacman goes left and uses dark magic")
  else:
    print(f"The minimax value is {dark_magic}. Pacman does not use dark magic")

c=int(input("Input(c): "))
