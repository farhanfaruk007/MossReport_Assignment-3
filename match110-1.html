import random

# Part 1

def AlphaBeta(x, y, p, q, mx):

    if x.is_term() or y == 0:
        return x.eva_luate()



    if not mx:

        v = float("inf")

        for ch in x.g_child():
            v = min(v, AlphaBeta(ch, y-1, p, q, True))
            q = min(q, v)
            if q <= p:
                break
        return v





    else:
        v = float("-inf")

        for ch in x.g_child():
            v = max(v, AlphaBeta(ch, y-1, p, q, False))
            p = max(p, v)
            if q <= p :
                break
        return v


class P_N:

    def __init__(self, n):
        self.name = n
        self.ch = []
    def is_term(self):
        return not self.ch
    def eva_luate(self):
        return random.choice([-1, 1])
    def g_child(self):
        return []




def Game(plr):

    rt = P_N('Root')
    play = 0
    player = plr
    r = []
    c = 0

    while play < 3:
        a = AlphaBeta(rt, 5, float("-inf"), float("inf"), player == 1)


        if a != 1:
            w = 'Scorpion'
        else:
            #w = 'Scorpion'
            w = 'Sub-Zero'
            c = c + 1

        r.append(f'Winner of Round {play + 1}: {w}')
        player = 1 if player == 0 else 0
        play += 1

    if(c<1):
        gw = "Scorpion"
    else:
        gw = "Sub-zero"

    print(f'Game Winner: {gw}')
    print(f'Total Rounds Played: {play}')

    for b in r:
        print(b)

def pacman_game(c):

    lf = [3, 6, 2, 3, 7, 1, 2, 0]
    a = [min(lf[0:2]), min(lf[2:4])]
    b = [min(lf[4:6]), min(lf[6:8])]
    ma = max(a)
    mb = max(b)
    n = max(mb, ma)

    al = max(lf[0:4]) - c
    br = max(lf[4:8]) - c

    m = max(br, al)

    if m < n:
        return f'The minimax value is {n}. Pacman does not use dark magic.'

    else:
        return f'The new minimax value is {m}. Pacman goes right and uses dark magic.'
