#PART 1
import random
def alpha_beta_pruning(depth, max_player, alpha, beta):
    if depth == 0:
      if max_player:
        return 1
      else:
        return -1
    if max_player:
      value = float('-inf')
    else:
      value = float('inf')
    for i in range(2):
        inner_depth = depth - 1
        inner_max_player = not max_player
        inner_alpha = alpha
        inner_beta = beta
        while inner_depth > 0:
            result = random.choice([-1, 1])
            inner_depth -= 1
            if inner_max_player:
                value = max(value,result)
                inner_alpha = max(inner_alpha, value)
                if inner_beta <= inner_alpha:
                    break
            else:
                value = min(value, result)
                inner_beta = min(inner_beta, value)
                if inner_beta <= inner_alpha:
                    break
        if max_player:
            value = max(value, result)
            alpha = max(alpha, value)
        else:
            value = min(value, result)
            beta = min(beta, value)
        if beta <= alpha:
            break
    return value

def simulate_rounds(starting_player, rounds):
    round_results = []
    current_player = starting_player


    for i in range(rounds):
        winner = alpha_beta_pruning(5, current_player == 1, float('-inf'), float('inf'))
        if winner == 1:
            round_results.append("Sub-Zero")
        else:
            round_results.append("Scorpion")
        current_player = 1 - current_player
    return round_results

def determine_game_winner(round_results):
    scorpion_wins = round_results.count("Scorpion")
    sub_zero_wins = round_results.count("Sub-Zero")
    if scorpion_wins > sub_zero_wins:
        return "Scorpion"
    else:
        return "Sub-Zero"

def simulate_mortal_kombat(starting_player, rounds):
    round_results = simulate_rounds(starting_player, rounds)
    game_winner = determine_game_winner(round_results)
    return game_winner, rounds, round_results

def pacman_game(dark_magic_cost):
    scores = [3, 6, 2, 3, 7, 1, 2, 0]
    def alpha_beta_pruning(depth, is_maximizing, alpha, beta, start_idx, end_idx):
        if depth == 0:
            return max(scores[start_idx:end_idx])
        if is_maximizing:
            max_score = float('-inf')
            for i in range(2):
                branch_size = len(scores) // 4
                branch_start_idx = start_idx + i * branch_size
                branch_end_idx = start_idx + (i + 1) * branch_size
                score = alpha_beta_pruning(depth - 1, False, alpha, beta, branch_start_idx, branch_end_idx)
                max_score = max(max_score, score)
                alpha = max(alpha, score)
                if alpha >= beta:
                    break
            return max_score
        else:
            min_score = float('inf')
            for i in range(2):
                branch_size = len(scores) // 4
                next_start_idx = start_idx + i * branch_size
                next_end_idx = start_idx + (i + 1) * branch_size
                score = alpha_beta_pruning(depth - 1, True, alpha, beta, next_start_idx, next_end_idx)


                min_score = min(min_score, score)
                beta = min(beta, score)
                if alpha >= beta:
                    break
            return min_score

    root = alpha_beta_pruning(2, True, float('-inf'), float('inf'), 0, len(scores))
    left = max(scores[:4]) - dark_magic_cost
    right = max(scores[4:]) - dark_magic_cost
    best_with_dark_magic = max(left, right)
    dark_magic = best_with_dark_magic > root

    if dark_magic:
        if left > right:
            print(f"The new minimax value is {left}. Pacman goes left and uses dark magic.")
        else:
            print(f"The new minimax value is {right}. Pacman goes right and uses dark magic.")
    else:
        print(f"The minimax value is {root}. Pacman does not use dark magic.")

pacman_game(2)
