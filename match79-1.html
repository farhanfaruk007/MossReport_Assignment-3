import random



class CombatGame:
    def __init__(self, starting_player):
        self.starting_player = starting_player
        self.contestants = ["Scorpion", "Sub-Zero"]
        self.round_count = 3
        self.round_results = []

    def determine_result(self, level, is_maximizing):
        if level == 0:
            return self.generate_opinion()



        results = []
        for _ in range(2):
            result = self.determine_result(level - 1, not is_maximizing)
            results.append(result)

        if is_maximizing:
            return max(results)
        else:
            return min(results)

    def generate_opinion(self):
        return random.choice([-1, 1])



    def start_game(self):
        for _ in range(self.round_count):
            result = self.determine_result(5, self.starting_player == 0)
            self.round_results.append("Scorpion" if result == -1 else "Sub-Zero")

        final_winner = "Scorpion" if self.round_results.count("Scorpion") > self.round_results.count("Sub-Zero") else "Sub-Zero"
        return final_winner

def pacman_game_logic(cost):
    values = [3, 6, 2, 3, 7, 1, 2, 0]
    initial_minimax = minimax_algorithm(0, 0, True, values, float('-inf'), float('inf'))

    left_magic_cost = values[1] - cost
    right_magic_cost = values[4] - cost
    best_magic_cost = max(left_magic_cost, right_magic_cost)

    overall_magic_cost = max(best_magic_cost, max(values[:4]) - cost, max(values[4:]) - cost)
    result = f"The new minimax value is {overall_magic_cost if overall_magic_cost > initial_minimax else initial_minimax}. Pacman {'goes right and uses' if overall_magic_cost > initial_minimax else 'does not use'} dark magic"
    return result

def minimax_algorithm(depth, index, is_maximizing, values, alpha, beta):
    if depth == 3:
        return values[index]

    best_value = float('-inf') if is_maximizing else float('inf')
    for i in range(2):
        current_value = minimax_algorithm(depth + 1, index * 2 + i, not is_maximizing, values, alpha, beta)
        best_value = max(best_value, current_value) if is_maximizing else min(best_value, current_value)
        alpha, beta = max(alpha, best_value), min(beta, best_value)
        if beta <= alpha:
            break
    return best_value
