def alpha_beta_pruning(turn, depth, alpha, beta, maximizing_player, player_starts):
    if depth == 0:

        return -1 if maximizing_player else 1



    if maximizing_player:
        max_eval = float('-inf')
        for _ in range(2):
            eval = alpha_beta_pruning(turn, depth - 1, alpha, beta, False, player_starts)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for _ in range(2):
            eval = alpha_beta_pruning(turn, depth - 1, alpha, beta, True, player_starts)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break
        return min_eval


def mortal_kombat_simulation(start_player):
    rounds = []
    current_player = start_player
    total_rounds = 3
    scorpion_wins = 0
    sub_zero_wins = 0

    for i in range(total_rounds):

        result = alpha_beta_pruning(current_player, 5, float('-inf'), float('inf'), current_player == 0, start_player)
        if result == -1:
            rounds.append(f"Round {i + 1}: Scorpion wins")
            scorpion_wins += 1
        else:
            rounds.append(f"Round {i + 1}: Sub-Zero wins")
            sub_zero_wins += 1


        current_player = 1 - current_player


    game_winner = "Scorpion" if scorpion_wins > sub_zero_wins else "Sub-Zero"
    print(f"Game Winner: {game_winner}")
    print(f"Total Rounds Played: {total_rounds}")
    for r in rounds:
        print(r)



player_start = int(input("Enter the starting player (0 for Scorpion, 1 for Sub-Zero): "))
