def Pacman(c):
    final_node=[3,6,2,3,7,1,2,0]
    minimaxa=minimaxi(final_node[:],0,True,-99999999,99999999)
    
    magicFinal=0
    bestMove=''
    maxL=max(final_node[:4])
    magicL=maxL-c
