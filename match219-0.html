#Part-1:

import random

branching_factor = 2
max_depth = 4
no_leaf_node = branching_factor**max_depth

leaf_nodes = []
for _ in range(no_leaf_node) :
    leaf_value = random.choice([-1,1])
    leaf_nodes.append(leaf_value)

def alpha_beta_pruning(node_values, depth, index, maximizingP, alpha, beta):
    if depth == max_depth:
        return node_values[index]

    if maximizingP:
        max_eval = float('-inf')
        for i in range(branching_factor):
            eval = alpha_beta_pruning(node_values, depth + 1, index * 2 + i, False, alpha, beta)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if  alpha >= beta:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for i in range(branching_factor):
            eval = alpha_beta_pruning(node_values, depth + 1, index * 2 + i, True, alpha, beta)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if  alpha >= beta:
                break
        return min_eval
