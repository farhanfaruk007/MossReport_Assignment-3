import random


from math import inf

def GameTree(Depth):
    count, numberOfNodes = 0, 0
    while count < Depth:
        numberOfNodes += 2**count
        count += 1
    array = [0] * (1 + numberOfNodes)
    firstLeafNode = 1

    for i in range(1, Depth - 1):
        firstLeafNode = 2*firstLeafNode + 1
    firstLeafNode += 1

    for i in range(firstLeafNode, len(array)):
        array[i] = random.choice([-1, 1])

    depth = 0


    count = 1
    starting = 1
    while depth < Depth - 1:

        for i in range(starting, 2**depth + starting):
            count = i
            array[i] = [array[i], depth, -inf, inf]
        starting = count + 1
        depth += 1

    return array
