#task1
import random
def combat_game(player):
  current = player
  round = 0
  win1 = None
  total_win = []
  total = 5


  for i in range(0, total):
    round += 1
    num = random.choice([-1, 1])
    if num == 1:
      result = "Sub-Zero"
    elif num == -1:
      result = "Scorpion"
    else:
      result = "Draw"
    total_win.append(result)
    if total_win.count("Scorpion") > 2:
      win1 = "Scorpion"
      break
    elif total_win.count("Sub-Zero") > 2:
      win1 = "Sub-Zero"
      break
  if not win1:
      if total_win.count("Scorpion") > total_win.count("Sub-Zero"):
        win1 = "Scorpion"
      elif total_win.count("Sub-Zero") > total_win.count("Scorpion"):
          win1 = "Sub-Zero"
      else:
        win1 = "Draw"
  return win1, round, total_win
player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))
win1, round, total_win = combat_game(player)
print("Game Winner:", win1)
print("Total Rounds Played:", round)
for i in range(0, len(total_win)):
    print(f"Round {i + 1}: {total_win[i]}")
