import random

def Game(player):
    players = ["Scorpion", "Sub-Zero"]
    current_player = player
    round = 0
    wins = []

    while True:
        round += 1

        leaf = [random.choice([-1, 1]) for _ in range(32)]
        a = random.choice(leaf)


        if a == -1:
            round_winner = 0  # Scorpion win
        else:
            round_winner = 1  #Sub-Zero win



        wins.append(players[round_winner])


        scorpion_wins = wins.count("Scorpion")

        subzero_wins = wins.count("Sub-Zero")


        if scorpion_wins == 3:
            game_winner = "Scorpion"
            break
        elif subzero_wins == 3:

            game_winner = "Sub-Zero"
            break


        current_player = 1 - current_player

    return game_winner, round, wins

def pacman(darkmagic_cost):
    scores = [3, 6, 2, 3, 7, 1, 2, 0]

    def minimax_with_pruning(index, depth, alpha, beta, is_pacman):
        if depth == 3:
            return scores[index]

        if is_pacman:
            best_score = float('-inf')
            for branch in range(2):
                value = minimax_with_pruning(index * 2 + branch, depth + 1, alpha, beta, False)
                best_score = max(best_score, value)
                alpha = max(alpha, value)
                if beta <= alpha:
                    break
            return best_score
        else:
            worst_score = float('inf')
            for branch in range(2):
                value = minimax_with_pruning(index * 2 + branch, depth + 1, alpha, beta, True)
                worst_score = min(worst_score, value)


                beta = min(beta, value)
                if beta <= alpha:
                    break
            return worst_score


    base_score = minimax_with_pruning(0, 0, float('-inf'), float('inf'), True)

    left_max = max(scores[0:4])
    right_max = max(scores[4:8])
    left_magic_score = left_max - darkmagic_cost
    right_magic_score = right_max - darkmagic_cost


    if max(left_magic_score, right_magic_score) > base_score:
        direction = "right" if right_magic_score > left_magic_score else "left"
        new_score = max(left_magic_score, right_magic_score)
        print(f"The new minimax value is {new_score}. Pacman goes{direction} and uses dark magic.")


    else:
        print(f"The minimax value is {base_score}. Pacman does not use dark magic.")
    print("-----------------------------------------")

player = int(input("enter the starting player (0 for Scorpion, 1 for Sub-Zero): "))
winner, rounds, wins = Game(player)
