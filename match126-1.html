import random
#TASK - 1 
class MortalKombat:
    def __init__(self, first_player):
        self.first_player = first_player  # 0 for Scorpion, 1 for Sub-Zero
        self.round = []
        self.total_rounds = 0

    def alpha_beta(self, depth, alpha, beta, maxP):
        if depth == 5: 
            return random.choice([-1, 1]) 
        if maxP:
            BVal = float("-inf")
            for i in range(2):
                val = self.alpha_beta(depth + 1, alpha, beta, False)
                BVal = max(BVal, val)
                alpha = max(alpha, val)
                if beta <= alpha:
                    break  # Prune condition
            return BVal
        else:
            BVal = float("inf")
            for i in range(2):
                val = self.alpha_beta(depth + 1, alpha, beta, True)


                BVal = min(BVal, val)
                beta = min(beta, val)
                if beta <= alpha:
                    break  # Prune condition
            return BVal

    def letsplay(self):
        cur_player = self.first_player
        for i in range(1, 4): 
            self.total_rounds += 1
            winner = self.alpha_beta(0, float("-inf"), float("inf"), cur_player)
            if winner == -1:
                round_winner = "Scorpion"
            else:
                round_winner = "Sub-Zero"
            self.round.append(round_winner)
            cur_player = 1 - cur_player

    def game_result(self):
        self.letsplay()
        if self.round.count("Scorpion") > self.round.count("Sub-Zero"):
            round_winner = "Scorpion"
        else:
            round_winner = "Sub-Zero"
        return round_winner, self.total_rounds, self.round


first_player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))
game = MortalKombat(first_player)
winner, total_rounds, rounds_winners = game.game_result()


print(f"Game Winner: {winner}\nTotal Rounds Played: {total_rounds}")
for i in range(total_rounds):
    print(f"Winner of Round {i + 1}: {rounds_winners[i]}")
