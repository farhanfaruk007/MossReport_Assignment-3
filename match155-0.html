    if level == max_depth:
        return values[index]
    if is_maximizing:
        best_value = float('-inf')
        for child in range(branch_factor):
            score = alpha_beta_prune(values, level + 1, index * branch_factor + child, False, alpha, beta)
            best_value = max(best_value, score)
            alpha = max(alpha, score)
            if beta <= alpha:
                break
        return best_value
    else:
        best_value = float('inf')
        for child in range(branch_factor):
            score = alpha_beta_prune(values, level + 1, index * branch_factor + child, True, alpha, beta)
            best_value = min(best_value, score)
            beta = min(beta, score)
            if beta <= alpha:
                break
        return best_value
