import random

def alpha_beta_pruning(node, depth, alpha, beta, max_player, brnch_factor):

    if depth == 0 or node["is_terminal"]:
        return node["value"]

    if max_player:
        max_eval = float('-inf')
        for child in node["children"][:brnch_factor]:
            eval_value = alpha_beta_pruning(child, depth - 1, alpha, beta, False, brnch_factor)
            max_eval = max(max_eval, eval_value)
            alpha = max(alpha, eval_value)
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for child in node["children"][:brnch_factor]:
            eval_value = alpha_beta_pruning(child, depth - 1, alpha, beta, True, brnch_factor)
            min_eval = min(min_eval, eval_value)
            beta = min(beta, eval_value)
            if beta <= alpha:
                break
        return min_eval


def generate_game_tree(depth, brnchng_factor, current_player, leaf_values):

    if depth == 0:
        return {"is_terminal": True, "value": leaf_values.pop(0), "children": []}
    node = {"is_terminal": False, "value": None, "children": []}
    for _ in range(brnchng_factor):
        child = generate_game_tree(depth - 1, brnchng_factor, 1 - current_player, leaf_values)
        node["children"].append(child)
    return node


def simulate_game(starting_player):

    max_depth = 5
    branch_factor = 2

    leaf_values = [-1, 1] * (branch_factor ** max_depth // 2)
    game_tree = generate_game_tree(max_depth, branch_factor, starting_player, leaf_values.copy())


    rounds_played = 0
    round_winners = []
    current_player = starting_player

    while rounds_played < 3:
        winner = alpha_beta_pruning(game_tree, max_depth, float('-inf'), float('inf'), current_player == 1, branch_factor)
        if winner == -1:
            round_winners.append("Scorpion")
        else:
            round_winners.append("Sub-Zero")
        rounds_played += 1
        current_player = 1 - current_player


    scorpion_wins = round_winners.count("Scorpion")
    sub_zero_wins = round_winners.count("Sub-Zero")

    game_winner = "Scorpion" if scorpion_wins > sub_zero_wins else "Sub-Zero"
    return game_winner, rounds_played, round_winners

def pacman_game(c):

    leaf_scores = [3, 6, 2, 3, 7, 1, 2, 0]
    lmv = [min(leaf_scores[0:2]), min(leaf_scores[2:4])]
    rmv = [min(leaf_scores[4:6]), min(leaf_scores[6:8])]
    mlv = max(lmv)
    mrv = max(rmv)

    minimax_vlu_wthout_drk_mgc = max(mlv, mrv)


    drk_mgc_lft_vlu = max(leaf_scores[0:4]) - c
    drk_mgc_rht_vlu = max(leaf_scores[4:8]) - c
    minimax_vlu_wth_drk_mgc = max(drk_mgc_lft_vlu, drk_mgc_rht_vlu)

    if minimax_vlu_wth_drk_mgc > minimax_vlu_wthout_drk_mgc:
