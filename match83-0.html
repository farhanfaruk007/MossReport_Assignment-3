def mortal_kombat_game(start_player):

    scores = [-1, 1, -1, 1, -1, 1, -1, 1]

    def alpha_beta_pruning(depth, node_index, is_maximizing, alpha, beta, scores):

        if depth == 0:
            return scores[node_index]

        if is_maximizing:
            max_eval = float("-inf")
            for i in range(2):
                eval = alpha_beta_pruning(depth - 1, node_index * 2 + i, False, alpha, beta, scores)
                max_eval = max(max_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float("inf")
            for i in range(2):


                eval = alpha_beta_pruning(depth - 1, node_index * 2 + i, True, alpha, beta, scores)
                min_eval = min(min_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return min_eval


    current_player = start_player
    results = []
    total_rounds = 0

    for _ in range(3):
        winner = alpha_beta_pruning(3, 0, current_player == 0, float("-inf"), float("inf"), scores)
        if winner == -1:
            results.append("Scorpion")
        else:
            results.append("Sub-Zero")
        total_rounds += 1
        current_player = 1 - current_player


    scorpion_wins = results.count("Scorpion")
    sub_zero_wins = results.count("Sub-Zero")

    overall_winner = "Scorpion" if scorpion_wins > sub_zero_wins else "Sub-Zero"

    return {
        "Game Winner": overall_winner,
        "Total Rounds Played": total_rounds,
        "Round Winners": results
    }

def pacman_game(c):

    scores = [3, 6, 2, 3, 7, 1, 2, 0]


    def alpha_beta(depth, node_index, is_maximizing, alpha, beta):

        if depth == 0:


            return scores[node_index]

        if is_maximizing:
            max_eval = float("-inf")
            for i in range(2):
                eval = alpha_beta(depth - 1, node_index * 2 + i, False, alpha, beta)
                max_eval = max(max_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float("inf")
            for i in range(2):
                eval = alpha_beta(depth - 1, node_index * 2 + i, True, alpha, beta)
                min_eval = min(min_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return min_eval


    root_value_without_magic = alpha_beta(3, 0, True, float("-inf"), float("inf"))


    left_subtree_max = max(scores[0:4])


    right_subtree_max = max(scores[4:8])

    left_with_magic = left_subtree_max - c
    right_with_magic = right_subtree_max - c


    if left_with_magic >= right_with_magic:
        best_strategy = "Pacman goes left and uses dark magic"
        best_value = left_with_magic
    else:
        best_strategy = "Pacman goes right and uses dark magic"
        best_value = right_with_magic

    return {
        "Minimax value without dark magic": root_value_without_magic,
        "Best value with dark magic": best_value,
        "Best strategy with dark magic": best_strategy,
    }
