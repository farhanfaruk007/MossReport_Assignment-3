#part-1

def selector(current_playr):
    """Determines which player wins the round """
    if current_playr == 1:
        return 1
    else:
        return -1

def start_game(starter_plyr):
    """manager of the game, round controller which is 3, winner tracker and announxes the overall winner """
    round_tracker = 0
    current_playr = starter_plyr
    winner_list = []

    #Total 3 ta round
    for rounds in range(1, 4):
        round_tracker += 1
        outcome = selector(current_playr)

        # round winner find kora
        if outcome == 1:
            winner_list.append(f"Winner of Round {rounds}: Sub-Zero")
        else:
            winner_list.append(f"Winner of Round {rounds}: Scorpion")

    # Altering the player for next round


    scorp_winner = sum(1 for winner in winner_list if "Scorpion" in winner)
    sub_winner = sum(1 for winner in winner_list if "Sub-Zero" in winner)

    # checking who won
    if scorp_winner > sub_winner:
        winP = "Scorpion"
    else:
        winP = "Sub-Zero"

    return winP, round_tracker, winner_list

def minimaxAlgo(depth, pacmanTurn, values, alpha, beta):
    # Base case:
    if depth == 3:
        return values.pop(0)

    if pacmanTurn:
        best_score = -99999999  #Pac-Man maximum
        for i in range(2):
            result = minimaxAlgo(depth + 1, False, values, alpha, beta)
            best_score = max(best_score, result)
            alpha = max(alpha, result)
            if beta <= alpha:  #pruning condition
                break
        return best_score
    else:
        worst_score = 9999999  #Ghost er turn
        for j in range(2):
            result = minimaxAlgo(depth + 1, True, values, alpha, beta)


            worst_score = min(worst_score, result)
            beta = min(beta, result)
            if beta <= alpha:  ##conditions for pruning basically
                break
        return worst_score

# dark magic:
def minimax_dark(values, modifier):
    left_side = values[1] - modifier  # dark magic on left subtree
    right_side = values[4] - modifier  # dark magic on right subtree
    return left_side, right_side


def pacman_game(modifier):

    values = [3, 6, 2, 3, 7, 1, 2, 0] #Initial values
    #dark magic chara minimax:
    minimax_score = values[:]
    minimax_result = minimaxAlgo(0, True, minimax_score, -float('inf'), float('inf'))

    left_magic, right_magic = minimax_dark(values, modifier)

    print(f"The final value of the root node without using dark magic: {minimax_result}")

    # dark magic er sathe compare:
