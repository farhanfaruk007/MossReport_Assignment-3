def alpha_beta(position,depth,alpha,beta,maximizingplayer,adj_list,utility_dict):
    if depth == 0:
        return utility_dict[position]


    if maximizingplayer:
        maxEval = -9999999
        for i in adj_list[position]:
            eval = alpha_beta(i,depth - 1,alpha,beta,False,adj_list,utility_dict)


            maxEval = max(maxEval,eval)
            alpha = max(alpha,eval)
            if alpha >= beta:
                break
        return maxEval
    else:
        minEval = 9999999
        for i in adj_list[position]:
            eval = alpha_beta(i,depth - 1,alpha,beta,True,adj_list,utility_dict)
            minEval = min(minEval,eval)
            beta = min(beta,eval)
            if alpha >= beta:
                break
        return minEval
