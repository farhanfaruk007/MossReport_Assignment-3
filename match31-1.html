def MortalKomb(Initiator):
    if Initiator== 0:
      conq= "Scorpion"
    else:
      conq= "Sub-Zero"
    turn= 3
    champ= []

    i= 0
    while i<turn:
        if i%2== 0:
            if Initiator== 0:
                champ.append("Scorpion")
            else:
                champ.append("Sub-Zero")
        else:
            if Initiator== 0:
                champ.append("Sub-Zero")
            else:
                champ.append("Scorpion")
        i+= 1


    scorpion, sub_zero= 0, 0


    for i in champ:
        if i== "Scorpion":
            scorpion+= 1
        elif i== "Sub-Zero":
            sub_zero+= 1

    if scorpion<sub_zero:
        conq= "Sub-Zero"
    else:
        conq= "Scorpion"
    print("Game winner:",conq)
    print("Total Rounds Played:",turn)


    i= 0
    while i<turn:
        print("champ of Round:", champ[i])
        i+= 1
