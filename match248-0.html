def minmax_simulation(starting_player, max_depth=5, alpha=-float('inf'), beta=float('inf')):
        def minmax(depth, alpha, beta, maximizing):
           if depth == max_depth:
            if maximizing:
              return -1
            else:
              return 1

           if maximizing:
            return maximize(depth, alpha, beta)
           else:
            return minimize(depth, alpha, beta)

        def maximize(depth, alpha, beta):
          maxval = -float('inf')
          eval1 = minmax(depth + 1, alpha, beta, False)
          maxval = max(maxval, eval1)
          alpha = max(alpha, eval1)
          if beta > alpha:
            eval2 = minmax(depth + 1, alpha, beta, False)
            maxval = max(maxval, eval2)
            alpha = max(alpha, eval2)
          return maxval

        def minimize(depth, alpha, beta):
          minval = float('inf')
          eval1 = minmax(depth + 1, alpha, beta, True)
          minval = min(minval, eval1)
          beta = min(beta, eval1)
          if beta > alpha:
            eval2 = minmax(depth + 1, alpha, beta, True)
            minval = min(minval, eval2)
            beta = min(beta, eval2)
          return minval

def alpha_beta_pruning(depth, index, maximizing, values, alpha, beta):
    if depth == 3:
        return values[index]

    if maximizing:
        return maximizedvalue(depth, index, values, alpha, beta)
    else:


        return minimizedvalue(depth, index, values, alpha, beta)


def maximizedvalue(depth, index, values, alpha, beta):
    maxieval = float('-inf')
    for i in range(2):
        evaluation = alpha_beta_pruning(depth + 1, index * 2 + i, False, values, alpha, beta)
        maxieval = max(maxieval, evaluation)
        alpha = max(alpha, evaluation)
        if beta <= alpha:
            break
    return maxieval


def minimizedvalue(depth, index, values, alpha, beta):
    minieval = float('inf')
    for i in range(2):

        minieval = min(minieval, evaluation)
        beta = min(beta, evaluation)
        if beta <= alpha:
            break
    return minieval
