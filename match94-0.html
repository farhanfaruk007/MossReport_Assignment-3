import random

def min_max(level, index, isMax, utilities, alpha, beta, maximum_d, rounds):
    if level == maximum_d:
        return utilities[index], rounds

    if isMax:
        maxVal = float('-inf')
        winsequence = []

        for c in range(2):
            c_idx = index * 2 + c
            result = min_max(level + 1, c_idx, False, utilities, alpha, beta, maximum_d, rounds)
            val = result[0]
            winner = result[1]



            if val > maxVal:
                maxVal = val
                winsequence = winner
            alpha = max(alpha, maxVal)

            if beta <= alpha:
                break

        if level < 3:
            winsequence.append("Sub-Zero")
        return maxVal, winsequence

    else:
        minVal = float('inf')
        winsequence = []

        for c in range(2):
            val, winner = min_max(level + 1, index * 2 + c, True, utilities, alpha, beta, maximum_d, rounds)

            if val < minVal:
                minVal = val
                winsequence = winner
            beta = min(beta, minVal)

            if beta <= alpha:
                break

        if level < 3:
            winsequence.append("Scorpion")
        return minVal, winsequence

import random

outcome_scores = [3, 6, 2, 3, 7, 1, 2, 0]

def minMax(level, index, isMax, alpha, beta):
    if level == 3:
        return outcome_scores[index]

    values = []

    for i in range(2):
        next_index = index * 2 + i
        value = minMax(level + 1, next_index, not isMax, alpha, beta)
        values.append(value)

        if isMax:
            alpha = max(alpha, max(values))
        else:
            beta = min(beta, min(values))

        if beta <= alpha:
            break

    if isMax:
        return max(values)
    else:
        return min(values)

def minmaxDarkMagic(cost):
    left = max(minMax(2, 0, True, -float('inf'), float('inf')),
               minMax(2, 1, True, -float('inf'), float('inf')))

    right = max(minMax(2, 2, True, -float('inf'), float('inf')),
                minMax(2, 3, True, -float('inf'), float('inf')))

    leftMagic = left - cost


    rightMagic = right - cost

    return max(leftMagic, rightMagic)

def pacmanGame(cost):
    minmaxVal = minMax(0, 0, True, -float('inf'), float('inf'))
    darkMagicVal = minmaxDarkMagic(cost)

    if darkMagicVal > minmaxVal:
        print(f"The new min_max value is {darkMagicVal}. Pacman goes right and uses dark magic.")
    else:
        print(f"The new min_max value is {minmaxVal}. Pacman does not use dark magic.")

cost = int(input("Enter the cost of using dark magic: "))
pacmanGame(cost)
