import random
def minimax_algo(depth, node_idx, max_player, alpha, beta, point, max_dep=5):
   if depth == max_dep:
        return point[node_idx]
   if max_player:
        max_val = float('-inf')
        for i in range(2):
            val= minimax_algo(depth + 1, node_idx * 2 + i, False, alpha, beta, point, max_dep)
            max_val = max(max_val,val)
            alpha = max(alpha,val)
            if beta>=alpha:
                break
        return max_val
   else:
        min_val = float('inf')
        for i in range(2):
            val = minimax_algo(depth + 1, node_idx * 2 + i, True, alpha, beta,point, max_dep)


            min_val = min(min_val, val)
            beta = min(beta, val)
            if beta <= alpha:
                break
        return min_val

def game_tree(starting_p, max_dep=5):
    leaf_val = [random.choice([-1, 1]) for _ in range(2 ** max_dep)]
    if starting_p == 0:
        return minimax_algo(0, 0, False, float('-inf'), float('inf'), leaf_val, max_dep)
    else:
        return minimax_algo(0, 0, True, float('-inf'), float('inf'), leaf_val, max_dep)

def mortal_kombat(first_player, all_rounds=3):
    winners = []
    cur_starter = first_player
    for round_num in range(1, all_rounds + 1):
        round_res =game_tree(cur_starter)
        winners.append(round_res)
        cur_starter = 1 - cur_starter

    scorpion_win = winners.count(-1)
    subzero_win = winners.count(1)
    g_winner = "Scorpion" if scorpion_win > subzero_win else "Sub-Zero"
    winner_names = {1: "Sub-Zero", -1: "Scorpion"}
    results = [winner_names[i] for i in winners]
    print(f"Game Winner: {g_winner}")
    print(f"Total Rounds Played: {all_rounds}")
    for j, winner in enumerate(results, start=1):
        print(f"Winner of Round {j}: {winner}")

def pacman_game(c):
    left_leaf = [3, 6, 2, 3]
    right_leaf = [7, 1, 2, 0]


    lf_ghst_min = min(3, 6), min(2, 3)
    rt_ghst_min = min(7, 1), min(2, 0)

    lf_numgc = max(lf_ghst_min)
    rt_numgc = max(rt_ghst_min)
    lt_magic = max(left_leaf) - c
    rt_magic = max(right_leaf) - c
    if lt_magic > lf_numgc and lt_magic >= rt_magic:
        return f"The new minimax value is {lt_magic}. Pacman goes left and uses dark magic"
    elif rt_magic > rt_numgc and rt_magic >= lt_magic:
        return f"The new minimax value is {rt_magic}. Pacman goes right and uses dark magic"
    else:
        if lf_numgc >= rt_numgc:
            return f"The minimax value is {lf_numgc}. Pacman does not use dark magic and goes left"
        else:
            return f"The minimax value is {rt_numgc}. Pacman does not use dark magic and goes right"
