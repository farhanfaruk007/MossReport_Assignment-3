import random
class mortalKombat:
    def __init__(self, first_start, Number_of_round):
        self.first_start = first_start
        self.total_rounds_played = Number_of_round
        self.max_depth = 5              #max depth of the game tree = 5
        self.branching_factor = 2       #branching factor = 2
        self.winner = []
        self.scorpion = 0 #Counting scorpion win
        self.subZero = 0 #Counting sub Zero win

    def gameStart(self):
        current_player = self.first_start

        for i in range(0, self.total_rounds_played):
            round_winner = self.prunning(current_player, 0, -float('inf'), float('inf'))
            if round_winner == -1:
                self.winner.append('Scorpion')
                self.scorpion += 1
            else:
                self.winner.append('Sub-Zero')
                self.subZero += 1

            if self.scorpion == 2 or self.subZero == 2: #who will win atleast 2 match, will be the winner
                break

            current_player = 1 - current_player #The player who didn't take the first turn in the previous round will start in subsequent rounds.


        if self.scorpion > self.subZero:
            game_winner = 'Scorpion'
        else:
            game_winner ='Sub-Zero'

        each_round_winner=""
        for i in range (len(self.winner)):
          each_round_winner += f"\nWinner of Round {i+1}: {self.winner[i]}"

        return game_winner, len(self.winner),each_round_winner


    def prunning(self, player, depth, alpha, beta):


        if depth == self.max_depth:
            return random.choice([-1, 1])  # -1 means Scorpion wins, 1 means Sub-Zero wins

        if player == 0:  # Scorpion's turn-->minimizing
            node = float('inf')
            for i in range(self.branching_factor):


                node = min(node, self.prunning(1 - player, depth + 1, alpha, beta))
                beta = min(beta, node)
                if alpha >= beta:
                    break  # Deleting Branch
            return node

        elif player == 1:  # Sub-Zero's turn-->maximizing
            node = -float('inf')
            for i in range(self.branching_factor):
                node = max(node, self.prunning(1 - player, depth + 1, alpha, beta))


                alpha = max(alpha, node)
                if alpha >= beta:
                    break  # Deleting Branch
            return node

def alpha_beta_prunning(depth, scored, player, alpha, beta, leaf_values):
    if depth == 3:
        return leaf_values[scored]

    if player == 0:  # Maximizing player
        node = float('-inf')

            alpha = max(alpha, node)
            if beta <= alpha:  # Pruning unnecessary branches
                break
        return node

    elif player == 1:  # Minimizing player
        node = float('inf')


        for i in range(2):
            node = min(node, alpha_beta_prunning(depth + 1, scored * 2 + i, 0, alpha, beta, leaf_values))

            if beta <= alpha:  # Pruning unnecessary branches
                break
        return node
