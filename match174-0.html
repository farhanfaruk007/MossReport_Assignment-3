#result_tree = [1 if i % 2 == 0 else -1 for i in range(32)]
result_tree = []
p_tracker = 0
while p_tracker < 32:
    result_tree.append(-1 if p_tracker % 2 != 0 else 1)
    p_tracker += 1

def decide_best_move(tree_position, depth, alpha, beta, is_max_turn):
    if depth == 0:
        return result_tree[tree_position]

    if is_max_turn:
        max_eval = float('-inf')
        for i in range(2):
            eval_result = decide_best_move(tree_position * 2 + i, depth - 1, alpha, beta, False)
            max_eval = max(max_eval, eval_result)
            alpha = max(alpha, max_eval)
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for i in range(2):
            eval_result = decide_best_move(tree_position * 2 + i, depth - 1, alpha, beta, True)
            min_eval = min(min_eval, eval_result)
            beta = min(beta, min_eval)
            if beta <= alpha:
                break
        return min_eval

def battle_simulation(starting_turn):
    results = []
    scorpion_score, subzero_score = 0, 0
    current_player = starting_turn

    for _ in range(3):
        outcome = decide_best_move(0, 5, float('-inf'), float('inf'), current_player == 1)
        winner = 'Sub-Zero' if outcome == 1 else 'Scorpion'
        results.append(winner)
        subzero_score += (outcome == 1)
        scorpion_score += (outcome == -1)
        current_player = 1 - current_player

    overall_winner = 'Sub-Zero' if subzero_score > scorpion_score else 'Scorpion'
    return results, overall_winner, len(results)

first_turn = int(input("Select which player starts first: "))
round_history, match_winner, total_rounds_played = battle_simulation(first_turn)

output_message = f"Match Winner: {match_winner}\nTotal Rounds Fought: {total_rounds_played}\n"
output_message += "\n".join([f"Winner of Round {i + 1}: {victor}" for i, victor in enumerate(round_history)])
print(output_message)



class PacmanGame:
    def __init__(self, magic_price):
        self.leaf_values = [3, 6, 2, 3, 7, 1, 2, 0]
        self.magic_price = magic_price

    def minimax(self, depth, pos, is_max_turn, alpha, beta, magic_used):
        if depth == 3:
            return self.leaf_values[pos]



        if is_max_turn:
            max_eval = float('-inf')
            for i in range(2):
                eval_outcome = self.minimax(depth + 1, pos * 2 + i, False, alpha, beta, magic_used)
                max_eval = max(max_eval, eval_outcome)
                alpha = max(alpha, max_eval)
                if beta <= alpha:
                    break

            return max_eval - self.magic_price if magic_used else max_eval
        else:


            min_eval = float('inf')
            for i in range(2):
                eval_outcome = self.minimax(depth + 1, pos * 2 + i, True, alpha, beta, magic_used)
                min_eval = min(min_eval, eval_outcome)
                beta = min(beta, min_eval)
                if beta <= alpha:
                    break
            return min_eval

    def run_game(self):
        without_magic = self.minimax(0, 0, True, float('-inf'), float('inf'), False)

        with_magic = self.minimax(0, 0, True, float('-inf'), float('inf'), True)

        print(f"The new minimax value is {with_magic}. Pacman ", end="")
        print("goes right and uses dark magic." if with_magic > without_magic else "does not use dark magic.")
