# Task 1

import random

def alpha_beta_pruning(node, depth, alpha, beta, player):
    if depth == 0 or node.is_terminal():
        return node.evaluate()

    if player:
        value = float('-inf')
        for child in node.get_children():
            value = max(value, alpha_beta_pruning(child, depth - 1, alpha, beta, False))
            alpha = max(alpha, value)
            if beta <= alpha:
                break
        return value


    else:
        value = float('inf')
        for child in node.get_children():
            value = min(value, alpha_beta_pruning(child, depth - 1, alpha, beta, True))
            beta = min(beta, value)
            if beta <= alpha:
                break
        return value

class Node:
    def __init__(self, name):
        self.name = name
        self.children = []

    def is_terminal(self):
        return not self.children

    def evaluate(self):
        return random.choice([-1, 1])

    def get_children(self):
        return self.children

def simulate_game(starting_player):
    root = Node("Root")
    rounds_played = 0
    current_player = starting_player
    round_results = []
    sub_zero_wins = 0

    while rounds_played < 3:
        result = alpha_beta_pruning(root, 5, float('-inf'), float('inf'), current_player == 1)

        if result == 1:
            winner = "Sub-Zero"
            sub_zero_wins += 1
        else:
            winner = "Scorpion"

        round_results.append(f"Winner of Round {rounds_played + 1}: {winner}")
        current_player = 1 if current_player == 0 else 0
        rounds_played += 1

    game_winner = "Sub-Zero" if sub_zero_wins > 1 else "Scorpion"

    print(f"Game Winner: {game_winner}")
    print(f"Total Rounds Played: {rounds_played}")
    for i in round_results:
        print(i)

players_input = list(map(int, input().split()))
for j in players_input:
    simulate_game(j)

def pacman_game(cost):
    leaf_scores = [3, 6, 2, 3, 7, 1, 2, 0]

    left_min_values = [min(leaf_scores[0:2]), min(leaf_scores[2:4])]
    right_min_values = [min(leaf_scores[4:6]), min(leaf_scores[6:8])]
    left_max_values = max(left_min_values)
    right_max_values = max(right_min_values)

    minimax_value = max(left_max_values, right_max_values)

    dark_magic_left = max(leaf_scores[0:4]) - cost
    dark_magic_right = max(leaf_scores[4:8]) - cost
    dark_magic_value = max(dark_magic_left, dark_magic_right)

    if dark_magic_value > minimax_value:
        return f"The new minimax value is {dark_magic_value}. Pacman goes right and uses dark magic."
    else:
        return f"The minimax value is {minimax_value}. Pacman does not use dark magic."

input_cost = list(map(int, input().split()))

for i in input_cost:
    print(pacman_game(i))
