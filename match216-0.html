#task1
class Mortal_Kombat:
    def __init__(self, first_player):
        self.rounds = 0
        self.branchingFactor = 2
        self.max_depth=5
        self.winner = None
        self.first_player = first_player
        self.scorpion_win = 0
        self.subzero_win = 0
        self.winner_per_round = {}
        self.leafnode_val = [-1, 1]
        self.game()

    def game(self):
        currentplayer = self.first_player
        roundstotal=3
        for i in range(roundstotal):
            self.rounds += 1
            winner = self.alpha_beta_pruning(0, currentplayer, float('-inf'), float('inf'), True)
            if winner == -1:
                self.winner_per_round[self.rounds] = "Scorpion"
                self.scorpion_win += 1
                currentplayer = 1
            else:
                self.winner_per_round[self.rounds] = "Sub-Zero"
                self.subzero_win += 1
                currentplayer = 0

        if self.scorpion_win > self.subzero_win:
            self.winner = "Scorpion"
        else:
            self.winner = "Sub-Zero"



    def alpha_beta_pruning(self, depth, player, alpha, beta, maximizing_player):
        if depth == self.max_depth:
            return self.leafnode_val[player]

        if maximizing_player:
            maxim_eval = float('-inf')
            for x in range(self.branchingFactor):
                eval = self.alpha_beta_pruning(depth + 1, player, alpha, beta, False)
                maxim_eval = max(maxim_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break
            return maxim_eval

        else:
            minim_eval = float('inf')
            for y in range(self.branchingFactor):
                eval = self.alpha_beta_pruning(depth + 1, 1 - player, alpha, beta, True)
                minim_eval = min(minim_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return minim_eval

        if max_level:
            max_placeholder = float('-inf')
            for i in range(2):
                eval = alpha_beta_pruning(depth + 1, node_index * 2 + i, False, alpha, beta)
                max_placeholder = max(max_placeholder, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break
            return max_placeholder
        else:
            minim_placeholder = float('inf')
            for i in range(2):
                eval = alpha_beta_pruning(depth + 1, node_index * 2 + i, True, alpha, beta)
                minim_placeholder = min(minim_placeholder, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return minim_placeholder


    minimax_val = alpha_beta_pruning(0, 0, True, float('-inf'), float('inf'))

    left_subtree = max(leaf_value[0:4])

    when_leftmagic = left_subtree - c

        new_minimax_value = when_rightmagic
