#task 1

import random

def evaluate_outcome(depth, player, max_depth):
    if depth == max_depth:
        return random.choice([-1, 1])
    outcomes = []
    for _ in range(3):
        outcome = evaluate_outcome(depth + 1, 1 - player, max_depth)
        outcomes.append(outcome)
    if player == 1:
        return max(outcomes)
    else:


        return min(outcomes)


def simulate_battle(start_player):
    rounds = []
    current_player = start_player
    for _ in range(4):
        result = evaluate_outcome(0, current_player, 3)
        winner = "Sub-Zero" if result == 1 else "Scorpion"
        rounds.append(winner)
        current_player = 1 - current_player
        if rounds.count("Scorpion") == 2 or rounds.count("Sub-Zero") == 2:
            break

    game_winner = "Scorpion" if rounds.count("Scorpion") > rounds.count("Sub-Zero") else "Sub-Zero"

    print(f"Game Winner: {game_winner}")
    print(f"Total Rounds Played: {len(rounds)}")

    for i, winner in enumerate(rounds, 1):
        print(f"Winner of Round {i}: {winner}")

def main():
    start_player = int(input().strip())
    simulate_battle(start_player)
if __name__ == "__main__":
    main()

def pacman_game(magic_cost):
    outcome_scores = [3, 6, 2, 3, 7, 1, 2, 0]

    def strategic_search(tree_depth, current_node, is_max_player, lower_limit, upper_limit, magic_used):
        if tree_depth == 3:
            return outcome_scores[current_node]

        if is_max_player:
            peak_score = float('-inf')
            optimal_path = None
            magic_applied = False

            for branch in range(2):
                base_score = strategic_search(tree_depth + 1, current_node * 2 + branch, False, lower_limit, upper_limit, magic_used)
                if base_score > peak_score:
                    peak_score = base_score
                    optimal_path = branch
                lower_limit = max(lower_limit, base_score)

                if not magic_used and magic_cost is not None:
                    enhanced_score = max(0, base_score - magic_cost)
                    if enhanced_score > peak_score:
                        peak_score = enhanced_score
                        optimal_path = branch
                        magic_applied = True

                if upper_limit <= lower_limit:
                    break

            if tree_depth == 0:
                if magic_applied:
                    print(f"The new minimax value is {peak_score}. Pacman goes {'right' if optimal_path else 'left'} and uses dark magic")
                else:
                    print(f"The minimax value is {peak_score}. Pacman does not use dark magic")
            best_score = peak_score
            return best_score


        else:
            min_score = float('inf')
            for branch in range(2):
                enemy_score = strategic_search(tree_depth + 1, current_node * 2 + branch, True, lower_limit, upper_limit, magic_used)
                min_score = min(min_score, enemy_score)
                upper_limit = min(upper_limit, enemy_score)

                if upper_limit <= lower_limit:
                    break
            return min_score

    result = strategic_search(0, 0, True, float('-inf'), float('inf'), False)
    return result

magic_cost = int(input())
pacman_game(magic_cost)
