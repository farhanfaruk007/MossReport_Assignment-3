#part - 1
def game_simulate(hero_select):
    if hero_select== 0:
      hero_store = "Scorpion"
    else:
      hero_store = "Sub-Zero"
    game_turn= 3
    game_champion= []

    a= 0
    while a < game_turn:
        if a%2== 0:
            if hero_select== 0:
                game_champion.append("Scorpion")
            else:
                game_champion.append("Sub-Zero")

        else:
            if hero_select== 0:
                game_champion.append("Sub-Zero")
            else:
                game_champion.append("Scorpion")
        a += 1


    hero1_score = 0
    hero2_score = 0


    for x in game_champion:
        if x == "Scorpion":
            hero1_score += 1
        elif x == "Sub-Zero":
            hero2_score += 1

    if hero1_score < hero2_score:
        hero_store = "Sub-Zero"
    else:
        hero_store = "Scorpion"

    print("Game winner:",hero_store)
    print("Total Rounds Played:",game_turn)


    b = 0
    while b < game_turn:
        print(f"game_champion of Round:{b+1}", game_champion[b])
        b += 1

game_simulate(int(input("0->Scorpion, 1->Sub-Zero: ")))

def PacmaN(c):
    nodes = {"root": {"ls": {"l_l": {"f_l": 3, "f_r": 6},"r_l": {"l_r_l": 2, "r_r_l": 3}}, "r_b": {"l_r": {"l_l_r": 7, "r_l_r": 1},"r_r": {"l_r_r": 2, "r_r_r": 0}}}}

    def calculate_A_B(x, game_champ, slt_a, slt_b, y):
        if type(x) is not dict:


            return x
        else:
            store_s = list(x.values())

            a = 0
            if not game_champ:
                z = float('inf')
                while a < len(store_s):
                    b = store_s[a]
                    z = min(z, calculate_A_B(b, True, slt_a, slt_b, y-1))
                    slt_b = min(slt_b, z)
                    if slt_a >= slt_b:
                        break
                    a += 1
            else:
                z = -float('inf')
                while a < len(store_s):
                    b = store_s[a]
                    z = max(z, calculate_A_B(b, False, slt_a, slt_b, y-1))
                    slt_a = max(slt_a, z)
                    if slt_a >= slt_b:
                        break
                    a += 1
            return z

    z = calculate_A_B(nodes["root"], True, -float('inf'), float('inf'), 2)

    l_nodes = max(max(nodes["root"]["ls"]["r_l"]["l_r_l"], nodes["root"]["ls"]["r_l"]["r_r_l"])-c,
                  max(nodes["root"]["ls"]["l_l"]["f_l"], nodes["root"]["ls"]["l_l"]["f_r"])-c)

    r_nodes = max(max(nodes["root"]["r_b"]["r_r"]["l_r_r"], nodes["root"]["r_b"]["r_r"]["r_r_r"])-c,
                  max(nodes["root"]["r_b"]["l_r"]["l_l_r"], nodes["root"]["r_b"]["l_r"]["r_l_r"])-c)

    temp_store = max(l_nodes, r_nodes)
    if temp_store > z:
        print("The new minimax value is", temp_store, ". Pacman goes", end=" ")
        if r_nodes < l_nodes:
            print("left and uses dark magic.")
        else:
            print("right and uses dark magic.")
    else:
        print("The minimax value is", z, ". Pacman does not use dark magic.")
