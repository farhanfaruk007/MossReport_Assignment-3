def mk(a, b, mxpl, a_val, b_val):
    abs_a=abs(a)
    if b==0 or abs_a==1:
        return a
    if mxpl:
        mx_rs=-float('inf')
        for opt in [-1, 1]:
            result=mk(opt, b - 1, False, a_val, b_val)
            mx_rs=max(mx_rs, result)
            a_val=max(a_val, result)
            if b_val<=a_val:
                break
        return mx_rs
    else:
        min_res=float('inf')
        for opt in [-1, 1]:
            result=mk(opt, b - 1, True, a_val, b_val)
            min_res=min(min_res, result)
            b_val=min(b_val, result)
            if b_val<=a_val:
                break
        return min_res
def sg(start_var):
    winners=[]
    total_rnds=0
    for rnd in range(1, 4):
        is_p1=start_var if rnd % 2==1 else 1-start_var
        res = mk(0, 5, is_p1==1, -float('inf'), float('inf'))
        total_rnds+=1
        if res==1:
            winners.append("Sub-Zero")
        else:
            winners.append("Scorpion")
    sz_wins = winners.count("Sub-Zero")
    sc_wins = winners.count("Scorpion")
    overall_winner = "Sub-Zero" if sz_wins > sc_wins else "Scorpion"
    return overall_winner, total_rnds, winners

def pcm(c):
    vals = [3, 6, 2, 3, 7, 1, 2, 0]
    def mmx(n, d, mx, a, b):
        if d==3:
            return vals[n]
        if mx:
            best = float('-inf')
            for ch in range(2):
                v = mmx(n * 2 + ch, d + 1, False, a, b)
                best = max(best, v)
                a = max(a, best)
                if a >= b:
                    break
            return best
        else:
            best = float('inf')
            for ch in range(2):
                v = mmx(n * 2 + ch, d + 1, True, a, b)
                best = min(best, v)
                b = min(b, best)
                if a >= b:
                    break
            return best
    res = mmx(0, 0, True, float('-inf'), float('inf'))
    lmax = max(vals[:4])
    rmax = max(vals[4:])
    lmag = lmax - c
    rmag = rmax - c
    if lmag > rmag and lmag > res:
        out = f"The new minimax value is {lmag}. Pacman goes left and uses dark magic."
    elif rmag > res:
        out = f"The new minimax value is {rmag}. Pacman goes right and uses dark magic."
    else:
        out = f"The minimax value is {res}. Pacman does not use dark magic."
    print(out)
