import random
def game(initial_player):
    players = ["Scorpion", "Sub-Zero"]
    current_player = initial_player
    played_rounds = 0
    round_winners = []
    while True:
        played_rounds += 1
        leaf_nodes = [random.choice([-1, 1]) for i in range(2 ** 5)]
        round_utility = random.choice(leaf_nodes)
        round_winner = 0 if round_utility == -1 else 1


        round_winners.append(players[round_winner])
        scorpion_wins = round_winners.count("Scorpion")
        subzero_wins = round_winners.count("Sub-Zero")
        if scorpion_wins == 3:
            game_winner = "Scorpion"
            break
        elif subzero_wins == 3:
            game_winner = "Sub-Zero"
            break
        current_player = 1 - current_player
    return game_winner, played_rounds, round_winners
initial_player = int(input("Enter a single-digit number (0 for Scorpion, 1 for Sub-Zero): " ))
winner, rounds, round_winners = game(initial_player)
print(f"Game Winner: {winner}")
print(f"Total Rounds Played: {rounds}")
for i, rw in enumerate(round_winners, 1):
    print(f"Winner of Round {i}: {rw}")

def pacman_game(cost):
    scores = [3, 6, 2, 3, 7, 1, 2, 0]
    def minimax_with_pruning(index, depth, alpha, beta, is_pacman):
        if depth == 3:
            return scores[index]
        if is_pacman:
            best_score = float('-inf')
            for branch in range(2):
                value = minimax_with_pruning(index * 2 + branch, depth + 1, alpha, beta, False)
                best_score = max(best_score, value)
                alpha = max(alpha, value)
                if beta <= alpha:
                    break
            return best_score
        else:
            worst_score = float('inf')
            for branch in range(2):
                value = minimax_with_pruning(index * 2 + branch, depth + 1, alpha, beta, True)
                worst_score = min(worst_score, value)


                beta = min(beta, value)
                if beta <= alpha:
                    break
            return worst_score
    base_score = minimax_with_pruning(0, 0, float('-inf'), float('inf'), True)
    left_max = max(scores[0:4])
    right_max = max(scores[4:8])
    left_magic_score = left_max - cost
    right_magic_score = right_max - cost
    if max(left_magic_score, right_magic_score) > base_score:
        direction = "right" if right_magic_score > left_magic_score else "left"
        new_score = max(left_magic_score, right_magic_score)
        print(
            f"The new minimax value is {new_score}. Pacman goes {direction} and uses dark magic."
        )
    else:
        print(f"The minimax value is {base_score}. Pacman does not use dark magic.")
magic_cost = int(input("Enter the cost of using dark magic (c): "))
pacman_game(magic_cost)
