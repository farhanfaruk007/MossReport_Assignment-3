import random
class Game:
    def __init__(self, s_p):
        self.p = ['Scorpion', 'Sub-Zero']
        self.s_p = s_p
        self.rounds = []
        self.l_n = []
        count = 0
        while count < 32:
            self.l_n.append(random.choice([-1, 1]))
            count += 1
    def minimax(self, index, depth, a, b, max_p):
        if depth == 0:
            return self.l_n[index]
        if max_p:
            max_eval = float('-inf')
            for i in range(2):
                eval_val = self.minimax(index * 2 + i, depth - 1, a, b, False)
                max_eval = max(max_eval, eval_val)
                a = max(a, max_eval)
                if b <= a:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for i in range(2):
                eval_val = self.minimax(index * 2 + i, depth - 1, a, b, True)
                min_eval = min(min_eval, eval_val)
                b = min(b, min_eval)
                if b <= a:
                    break
            return min_eval
    def play_round(self, r_num):
        depth = 5
        a = float('-inf')
        b = float('inf')
        if self.s_p == 0:
            max_p = True
        else:
            max_p = False
        result = self.minimax(0, depth, a, b, max_p)
        if result == -1:
            winner = 'Scorpion'
        else:
            winner = 'Sub-Zero'
        self.rounds.append(winner)
        if self.s_p == 0:
            self.s_p = 1
        else:
            self.s_p = 0
    def start_game(self):
        t_r = 3
        c_r = 1
        while c_r <= t_r:
            self.play_round(c_r)
            c_r += 1
        s_wins = 0
        z_wins = 0
        for winner in self.rounds:
            if winner == 'Scorpion':
                s_wins += 1
            elif winner == 'Sub-Zero':
                z_wins += 1
        if s_wins > z_wins:
            g_winner = 'Scorpion'
        else:
            g_winner = 'Sub-Zero'
        print(f"Winner: {g_winner}")
        print(f"Total Rounds: {t_r}")
        r_num = 1
        for winner in self.rounds:
            print(f"Winner Round {r_num}: {winner}")
            r_num += 1

def ab_pruning(depth, mx, a, b, v, idx=0):
    if depth == 0:
        return v[idx]


    if mx:
        max_v = float('-inf')
        for i in range(2):
            val = ab_pruning(depth - 1, False, a, b, v, idx * 2 + i)
            max_v = max(max_v, val)
            a = max(a, val)
            if b <= a:
                break
        return max_v
    else:
        min_v = float('inf')
        for i in range(2):
            val = ab_pruning(depth - 1, True, a, b, v, idx * 2 + i)
            min_v = min(min_v, val)
            b = min(b, val)
            if b <= a:
                break
        return min_v
def pacman_game(c):
    l_n = [3, 6, 2, 3, 7, 1, 2, 0]
    root_val = ab_pruning(3, True, float('-inf'), float('inf'), l_n)
    left_val = max(l_n[0:4]) - c
    right_val = max(l_n[4:8]) - c
    if max(left_val, right_val) > root_val:
        if left_val >= right_val:
            print(f"The new minimax value is {left_val}. Pacman goes left and uses dark magic")
        else:
            print(f"The new minimax value is {right_val}. Pacman goes right and uses dark magic")
    else:
        print(f"The minimax value is {root_val}. Pacman does not use dark magic")
c = int(input())
pacman_game(c)
