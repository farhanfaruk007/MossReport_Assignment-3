def pacman_game(c):

    leaf_s = [3, 6, 2, 3, 7, 1, 2, 0]


    left_scores = leaf_s[:4]
    right_scores = leaf_s[4:]

    l_two_l = [min(left_scores[:2]), min(left_scores[2:4])]


    l_two_r = [min(right_scores[:2]), min(right_scores[2:4])]

    max_l = max(l_two_l)
    max_r = max(l_two_r)

    ndm = max(max_r, max_l)

    dml = max(left_scores[:4]) - c
    dmr = max(right_scores[:4]) - c

    udm = max(dmr, dml)

    if udm > ndm:
        return f"The new minimax value is {udm}. Pacman goes right and uses dark magic."
    else:
        return f"The minimax value is {ndm}. Pacman does not use dark magic."

import random

def ab_prunn(n, d, a, b, maxx_player):
    if d == 0 or n.is_term():
        return n.eva_luate()



    if maxx_player:
        va_lue = float('-inf')
        for childrenn in n.g_child():
            va_lue = max(va_lue, ab_prunn(childrenn, d-1, a, b, False))
            a = max(a, va_lue)
            if b <= a:
                break
        return va_lue


    else:
        va_lue = float('inf')
        for childrenn in n.g_child():
            va_lue = min(va_lue, ab_prunn(childrenn, d-1, a, b, True))
            b = min(b, va_lue)
            if b <= a:
                break
        return va_lue

class Node:
    def __init__(self, name):
        self.name = name
        self.childrenn = []

    def is_term(self):
        return not self.childrenn

    def eva_luate(self):
        return random.choice([-1, 1])

    def g_child(self):
        return []

class MortalKombatGame:
    def __init__(self, first_player):
        self.first_player = first_player
        self.rounds = []

    def minimax(self, depth, is_maximizing_player, alpha, beta):
        if depth == 5:
            return 1 if not is_maximizing_player else -1

        if is_maximizing_player:
            max_eval = -float('inf')
            for _ in range(2):
                eval = self.minimax(depth + 1, False, alpha, beta)
                max_eval = max(max_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for _ in range(2):
                eval = self.minimax(depth + 1, True, alpha, beta)
                min_eval = min(min_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return min_eval

    def play_round(self, first_player):
        result = ab_prunn(Node("Root"), 5, float('-inf'), float('inf'), first_player == 1)
        return 'Sub-Zero' if result == 1 else 'Scorpion'

    def simulate(self):
        current_player = self.first_player
        round_counter = 1

        while round_counter <= 3:
            round_winner = self.play_round(current_player == 1)
            self.rounds.append(f"Winner of Round {round_counter}: {round_winner}")
            current_player = 1 - current_player
            round_counter += 1

        return self.rounds
