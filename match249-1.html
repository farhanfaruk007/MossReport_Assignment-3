class MortalKombat:
    def __init__(self,players):
      self.players = players
      self.BF=2
      self.max_depth=5
      self.round = []

    def AlphaBeta_Pruning(self,tree_depth,alpha,beta,max_no):

       if tree_depth>=self.max_depth:
          return -1 if not max_no else 1



       if max_no:
          max_val = float("-inf")
          for i in range(self.BF):
             value=self.AlphaBeta_Pruning(tree_depth+1,False,alpha,beta)
             max_val= max(max_val,value)
             alpha = max(alpha,value)
             if alpha>=beta:
               break
          return max_val

       else:
            min_val = float("inf")
            for i in range(self.BF):
               value= self.AlphaBeta_Pruning(tree_depth+1,True,alpha,beta)
               min_val= min(min_val,value)
               beta = min(beta,value)
               if alpha>=beta:
                   break

            return min_val

    def round_simulation(self,players):
        winner= self.AlphaBeta_Pruning(0,(players==1),float("-inf"),float("inf"))
        if winner==-1:
            return "Scropion"
        else:
            return "Sub_zero"

    def Gameplay(self):
        currentPlayer= self.players
        while len(self.round)<3:
             select_winner= self.round_simulation(currentPlayer)
             self.round.append(currentPlayer)
             if currentPlayer==0:
                 currentPlayer=1
             else:
                 currentPlayer=0
        gameWinner = max(set(self.round),key=self.round.count)
        return gameWinner, len(self.round), self.round

players = int(input("Enter the starter (0 for Scorpion, 1 for Sub-Zero): "))
