def alpha_beta_pruning(node_values, level, temp, maximizing_player, alpha, beta):


    if level == maximum_depth:

        return node_values[temp]

    if maximizing_player:
        score_best = float('-inf')
        for child in range(branching_factor):
            score_current = alpha_beta_pruning(node_values, level + 1, temp * branching_factor + child, False, alpha, beta)
            score_best = max(score_best, score_current)
            alpha = max(alpha, score_current)

            if beta <= alpha:
                break
        return score_best
    else:
        score_best = float('inf')
        for child in range(branching_factor):
            score_current = alpha_beta_pruning(node_values, level + 1, temp * branching_factor + child, True, alpha, beta)
            score_best = min(score_best, score_current)
            beta = min(beta, score_current)
            if beta <= alpha:
                break
        return score_best

def pacman_strategy(cost_of_magic, temp):
    def explore_tree(depth, value, turn_of_pacman, alpha, beta, maximum_depth):
        if depth == maximum_depth:
            return temp[value]
        if turn_of_pacman:
            best_pacman = float('-inf')
            for x in range(2):
                result = explore_tree(depth + 1, value * 2 + x, False, alpha, beta, maximum_depth)
                best_pacman = max(best_pacman, result)
                alpha = max(alpha, result)
                if alpha >= beta:  # Prune further branches
                    break
            return best_pacman
        else:
            best_of_ghost = float('inf')
            for x in range(2):
                result = explore_tree(depth + 1, value * 2 + x, True, alpha, beta, maximum_depth)
                best_of_ghost = min(best_of_ghost, result)
                beta = min(beta, result)
                if alpha >= beta:
                    break
            return best_of_ghost

    no_magic_value = explore_tree(0, 0, True, float('-inf'), float('inf'), maximum_tree_depth)


    highest_left_branch = max(temp[:4])
    highest_right_branch = max(temp[4:])
