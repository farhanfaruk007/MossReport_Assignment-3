def leaf_nodes_values():
    l=[]
    for i in range(leaf_node):
        n=random.choice([-1,1])
        l.append(n)
    return l

def alpha_beta_pruning(node_values, depth, index, maximizing_player, alpha, beta):


    if depth == mdp :
        return node_values[index]
    if maximizing_player:
        max_eval = float('-inf')
        for i in range(2):
            eval = alpha_beta_pruning(node_values, depth + 1, index * 2 + i, True, alpha, beta)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:


                break
        return max_eval
    else:
        min_eval = float('inf')
        for i in range(2):
            eval = alpha_beta_pruning(node_values, depth + 1, index * 2 + i, False, alpha, beta)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break
        return min_eval
