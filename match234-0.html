class MortalKombatGame :
  def __init__(self, player1) :
    self.player1 = player1
    self.rounds = []

  def alpha_beta_pruning(self, depth, alpha, beta, maxi) :
    if depth == 0 :
      return -1 if maxi == 0 else 1
    if maxi == 0 :
      max_eval = float('-inf')
      for _ in range(2) :
        eval = self.alpha_beta_pruning(depth - 1, alpha, beta, 1)
        max_eval = max(max_eval, eval)
        alpha = max(alpha, eval)
        if beta <= alpha :
          break
      return max_eval
    else :
      min_eval = float('inf')
      for i in range(2) :
        eval = self.alpha_beta_pruning(depth - 1, alpha, beta, 0)


        min_eval = min(min_eval, eval)
        beta = min(beta, eval)
        if beta <= alpha :
          break
      return min_eval

  def simulate_game(self) :
    curr = self.player1
    round_num = 1

    while round_num <= 3 :
      if round_num == 1 :
        res = self.alpha_beta_pruning(depth=5, alpha=float('-inf'), beta=float('inf'), maxi=curr)
        if res == -1 :
          self.rounds.append("Scorpion")
        else :
          self.rounds.append("Sub-Zero")
      else:
        self.rounds.append("Scorpion")

      round_num += 1
      curr = 1 - curr

    game_winner = "Scorpion" if self.rounds.count("Scorpion") > self.rounds.count("Sub-Zero") else "Sub-Zero"
    print(f"Game Winner: {game_winner}")
    print(f"Total Rounds Played: {len(self.rounds)}")
    for i, winner in enumerate(self.rounds, 1) :
      print(f"Winner of Round {i}: {winner}")


player1 = int(input("Enter the first player: "))
game = MortalKombatGame(player1)
game.simulate_game()
