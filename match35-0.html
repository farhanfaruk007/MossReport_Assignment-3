class MortalKombatGame:
    def __init__(self, first_player):
        self.first_player = first_player
        self.total_rounds = 0
        self.round_winners = []

    def minimax(self, depth, is_maximizing, alpha, beta):
        if depth == 5:
            return -1 if is_maximizing else 1

        if is_maximizing:
            best_eval = -1
            for _ in range(2):
                eval = self.minimax(depth + 1, False, alpha, beta)
                best_eval = max(best_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break
            return best_eval

        else:
            min_eval = float('inf')
            for _ in range(2):
                eval = self.minimax(depth + 1, True, alpha, beta)


                min_eval = min(min_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return min_eval

    def simulate_round(self, depth, player):
        result = self.minimax(depth, player == 1, -1, 1)
        self.total_rounds += 1

        round_winner = "Sub-Zero" if result == 1 else "Scorpion"
        self.round_winners.append(f"Round {self.total_rounds}: {round_winner}")

        return round_winner

    def start_game(self):
        current_player = self.first_player
        for _ in range(3):
            self.simulate_round(0, current_player)
            current_player = 1 - current_player
        return self.get_winner()

    def get_winner(self):
        scorpion_wins = self.round_winners.count("Scorpion")
        subzero_wins = self.round_winners.count("Sub-Zero")

        winner = "Scorpion" if scorpion_wins > subzero_wins else "Sub-Zero"
        return winner, self.round_winners, self.total_rounds

def alpha_beta_pruning(node, depth, alpha, beta, is_maximizing):
    if depth == 0 or isinstance(node, int):
        return node

    if is_maximizing:
        value = float('-inf')
        for child in node:
            child_value = alpha_beta_pruning(child, depth - 1, alpha, beta, False)
            value = max(value, child_value)
            alpha = max(alpha, value)
            if beta <= alpha:
                break
        return value
    else:
        value = float('inf')
        for child in node:
            child_value = alpha_beta_pruning(child, depth - 1, alpha, beta, True)


            value = min(value, child_value)
            beta = min(beta, value)
            if beta <= alpha:
                break
        return value

def pacman_strategy(dark_power):
    scenario_tree = [
        [[3, 6], [2, 3]],
        [[7, 1], [2, 0]]
    ]

    root_score = alpha_beta_pruning(scenario_tree, 3, float('-inf'), float('inf'), True)

    left_score = max(max(scenario_tree[0][0]), max(scenario_tree[0][1])) - dark_power
    right_score = max(max(scenario_tree[1][0]), max(scenario_tree[1][1])) - dark_power

    best_score = max(left_score, right_score)

    if best_score > root_score:
        direction = "left" if left_score >= right_score else "right"
        score = left_score if direction == "left" else right_score
        return f"The new minimax value is {score}. Pacman goes {direction} and uses dark power."

    return f"The minimax value is {root_score}. Pacman does not use dark power."
