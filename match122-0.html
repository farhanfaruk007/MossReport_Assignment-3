import random
maxi, mini = float('inf'), float('-inf')


def minimax(maxd, curr, maxp, alpha, beta, choices):
	if maxd == 5:
		return random.choice(choices)
	if maxp:
		temp = mini
		for i in range(0, 2):
			val = minimax(maxd + 1, curr * 2 + i, 0, alpha, beta, choices)
			temp = max(temp, val)
			alpha = max(alpha, temp)


			if beta <= alpha:
				break
		return temp
	else:
		temp = maxi
		for i in range(0, 2):
			val = minimax(maxd + 1, curr * 2 + i, 1, alpha, beta, choices)
			temp = min(temp, val)
			beta = min(beta, temp)
			if beta <= alpha:
				break
		return temp
print('### Part-1 ###')
player = int(input())
total = 0
winner = []
choices = [-1, 1]
while total < 3:
    ans = minimax(0, 0, player, mini, maxi, choices)
    winner.append('Sub-Zero') if ans == 1 else winner.append('Scorpion')
    total += 1
    player = 1 - player
sc = winner.count('Scorpion')
sz = winner.count('Sub-Zero')
print(f"Game Winner: {'Scorpion' if sc > sz else 'Sub-Zero'}")
print("Total Rounds Played:", total)
for i in range(1, 4):
    print(f'Winner of Round {i}: {winner[i - 1]}')

def pacman_game(i):
    lst = [3, 6, 2, 3, 7, 1, 2, 0]
    def mm(x, y, chk):
        if x == 3:
            return lst[y]
        if chk:


            c1 = mm(x + 1, y * 2, False)
            c2 = mm(x + 1, y * 2 + 1, False)
            return max(c1, c2)
        else:
            c1 = mm(x + 1, y * 2, True)
            c2 = mm(x + 1, y * 2 + 1, True)
            return min(c1, c2)
    def hlp(x, y):
        if x == 3:
            return lst[y]
        c1 = hlp(x + 1, y * 2)
        c2 = hlp(x + 1, y * 2 + 1)
        return max(c1, c2)
    v1 = mm(0, 0, True)
    v2 = hlp(0, 0) - i
    if v1 < v2:
        print(f"The new minimax value is {v2}. Pacman goes right and uses dark magic")
    else:
        print(f"The minimax value is {v1}. Pacman does not use dark magic")
