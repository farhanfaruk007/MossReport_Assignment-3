        return player_list[node]

    if starting_player==1:
         max_val = float('-inf')

            else:
                    break

def begin_round(first_player):
    starting_player = first_player
    scorpion_wins = 0
    sub_zero_wins = 0
    round = 0

    winners_by_round = []

    while round<3:
          utility_value = alpha_beta_pruning(0, 5, float('-inf'), float('inf'), starting_player)


          if utility_value == 1:
              winner = "Sub-zero"
          else:
              winner = "Scorpion"
          winners_by_round.append(winner)
          starting_player = 1 - starting_player #alternating the players
          round = round + 1

    for i in winners_by_round:
        if i == "Scorpion":
            scorpion_wins = scorpion_wins + 1
        else:
            sub_zero_wins = sub_zero_wins + 1
    if  sub_zero_wins > scorpion_wins :
        winnner = "Sub-zero"
    else:
        winnner = "Scorpion"
    return winnner, round, winners_by_round

def pacman_game(cost):
    left_side = alpha_beta_pruning(0, 1, float('-inf'), float('inf'), True)
    no_power = alpha_beta_pruning(0, 3, float('-inf'), float('inf'), True)
    right_side = alpha_beta_pruning(2, 1, float('-inf'), float('inf'), True)

    left_power = left_side - cost
    right_power = right_side - cost

    power = max(left_power, right_power)
    if power <= no_power:
        return f'The maximum value is {no_power} Pacman does not use dark magic'
    else:
        if right_power < left_power:
            return f'The maximum value is {left_power} Pacman goes on the left side and uses dark magic'
        else:
            return f'The maximum value is {right_power} Pacman goes on right side and uses dark magic'

cost = int(input("Enter Cost: "))
pacman=pacman_game(cost)
