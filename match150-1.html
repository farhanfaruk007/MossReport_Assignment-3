def pacman_game(c):
    tree = {'root' : {'player_name' : 'Pac-Man' , 'child' : ['left' , 'right' ]},
    'left' : {'player_name' : 'Ghost' , 'child' : ['left_left' , 'left_right' ]},
    'right': {'player_name' : 'Ghost' , 'child' : ['right_left', 'right_right']},
    'left_left': {'player_name': 'Pac-Man' , 'value': 3},
    'left_right': {'player_name' : 'Pac-Man', 'value': 6},
    'right_left': {'player_name' : 'Pac-Man', 'value': 2},
    'right_right': {'player_name' : 'Pac-Man', 'value': 7}
    }



    def pruning_alp_beta(node, alp, bet, player_name):
        if 'value' in node:
            return node['value']

            value = -float('inf')
            for child in node['child']:
                value = max(value, pruning_alp_beta(tree[child], alp, bet, 'Ghost'))
                alp = max(alp, value)
                if bet <= alp:
                    break
            return value
        else:
            value = float('inf')
            for child in node['child']:
                value = min(value, pruning_alp_beta(tree[child], alp, bet, 'Pac-Man'))
                beta = min(bet, value)
                if beta <= alp:
                    break
            return value


    final_val = pruning_alp_beta(tree['root'], -float('inf'), float('inf'), 'Pac-Man')

    right_sub_values = []
