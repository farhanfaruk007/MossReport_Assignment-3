def alpha_beta_pruning(node, depth, alpha, beta, find_powerful_plyer):

    if depth == 5 or type(node) is not list: #new variable initiate for this function
        return node

    # checking plyer information which is correct or not for both condition

    if find_powerful_plyer:

        searching_max_evalution_value = float('-inf')

        for collecting_node_of_child in node:

            select_value = alpha_beta_pruning(collecting_node_of_child, depth + 1, alpha, beta, False)
            searching_max_evalution_value = max(searching_max_evalution_value, select_value)
            alpha = max(alpha, select_value)

            if beta <= alpha:
                break

        return searching_max_evalution_value

    else:

        searching_min_evalution_value2 = float('inf')

        for collecting_node_of_child in node:
            select_value = alpha_beta_pruning(collecting_node_of_child, depth + 1, alpha, beta, True)
            searching_min_evalution_value2 = min(searching_min_evalution_value2, select_value)
            beta = min(beta, select_value)

            if beta <= alpha:
                break

        return searching_min_evalution_value2
