#Task 1

import random

def Mortal_Kombat(player):
  current_player = player
  total_round = 0
  victor = None
  result = []
  max_rounds = 5




  for i in range(max_rounds):
    total_round += 1
    outcome = random.choice([-1,1])


    if outcome == 1:
      round_winner = "Sub-Zero"
    elif outcome == -1:
      round_winner = "Scorpion"
    else:
      round_winner = "Tie"

    result.append(round_winner)


    if result.count('Sub-Zero') > 2:
      victor = "Sub-Zero"
      break
    elif result.count('Scorpion') > 2:
      victor = "Scorpion"
      break
    elif total_round == max_rounds:
      victor = "Tie"
      break

  current_player = 1 - current_player

  return victor, total_round, result

player = int(input("Enter the starting player (0 for Scorpion, 1 for Sub-Zero): "))
if player not in [0, 1]:
  print("Invalid Input. Enter 0 or 1")

victor, total_round, result = Mortal_Kombat(player)

print(f"Game Winner: {victor}")
print(f"Total Rounds Played: {total_round}")

for i in range(0,len(result)):
   print("Winner of Round", i+1, ":", result[i])

def pacman(c):
    left_leaf = [3, 6, 2, 3]
    right_leaf = [7, 1, 2, 0]


    pacman_left = [max(left_leaf[:2]), max(left_leaf[2:])]
    ghost_left = min(pacman_left)

    pacman_right = [max(right_leaf[:2]), max(right_leaf[2:])]


    ghost_right = min(pacman_right)

    if  ghost_left >  ghost_right:
        no_magic = ghost_left
        no_magic_direction = "left"
    else:
         no_magic = ghost_right
         no_magic_direction = "right"


    left_leaf_magic = max(left_leaf) - c
    right_leaf_magic = max(right_leaf) - c

    if left_leaf_magic > right_leaf_magic:
        with_magic = left_leaf_magic
        magic_direction = "left"
    else:
        with_magic = right_leaf_magic
        magic_direction = "right"


    if with_magic > no_magic:
        print(f"The new minimax value is {with_magic}. Pacman goes {magic_direction} and uses dark magic.")
    elif with_magic < no_magic:
        print(f"The new minimax value is {no_magic}. Pacman goes {no_magic_direction} and does not use dark magic.")
    else:
        print(f"The new minimax value is {no_magic}. Pacman moves {no_magic_direction} and does not use dark magic.")

c = int(input())

pacman(c)
