import random

def alpha_beta_pruning(depth, maximizing_player, alpha, beta):

  if depth == 5:


    return random.choice([-1, 1])  # -1 for Scorpion, 1 for Sub-Zero

  if maximizing_player:
    max_eval = float('-inf')
    for i in range(2):
      eval = alpha_beta_pruning(depth + 1, False, alpha, beta)
      max_eval = max(max_eval, eval)
      alpha = max(alpha, eval)
      if beta <= alpha:
        break
    return max_eval
  else:
    min_eval = float('inf')
    for i in range(2):
      eval = alpha_beta_pruning(depth + 1, True, alpha, beta)


      min_eval = min(min_eval, eval)
      beta = min(beta, eval)
      if beta <= alpha:
        break
    return min_eval


def simulate_game(starting_player):

  current_player = starting_player
  rounds_played = 0
  round_results = []

  while rounds_played < 5:  # Play up to 5 rounds
    winner = alpha_beta_pruning(0, current_player == 1, float('-inf'), float('inf'))
    if winner == -1:
        round_winner = "Scorpion"
    else:
        round_winner = "Sub-Zero"

    round_results.append(round_winner)
    if current_player == 0:
      current_player = 1  # Sub-Zero starts next
    else:
      current_player = 0  # Scorpion starts next



    rounds_played += 1

    # Determine winner in 3 rounds
    if round_results.count("Scorpion") == 3:
      return "Scorpion", rounds_played, round_results
    elif round_results.count("Sub-Zero") == 3:
      return "Sub-Zero", rounds_played, round_results

  #Else
  scorpion = round_results.count("Scorpion")
  sub_zero = round_results.count("Sub-Zero")
  if scorpion > sub_zero:
    game_winner = "Scorpion"
  else:
    game_winner = "Sub-Zero"
  return game_winner, rounds_played, round_results


starting_player = int(input("Enter the starting player (0 for Scorpion, 1 for Sub-Zero): "))
result = simulate_game(starting_player)

def pacman_game(c):

  node_values = [3, 6, 2, 3, 7, 1, 2, 0]

  # Without dark magic


  val = node_values
  result = alpha_beta_pruning(val, 0, True, float('-inf'), float('inf'))

  # Dark magic
  l_max = max(6, 3) #left subtree max
  r_max = max(7, 2) #right subtree max
  left_magic = l_max - c
  right_magic = r_max - c

  if left_magic > right_magic and left_magic > result:
    return f"The new minimax value is {left_magic}. Pacman goes left and uses dark magic."
  elif right_magic > left_magic and right_magic > result:
    return f"The new minimax value is {right_magic}. Pacman goes right and uses dark magic."
  else:
    return f"The minimax value is {result}. Pacman does not use dark magic."
