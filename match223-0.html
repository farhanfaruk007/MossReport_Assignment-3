def alpha_beta_pruning(lvl, scorpion_turn, alpha, beta):
    if lvl == 5:
        return -1 if scorpion_turn else 1

    if scorpion_turn:
        max_score = float('-inf') #alpha
        for _ in range(2):
            score = alpha_beta_pruning(lvl+1, False, alpha, beta)
            max_score = max(max_score, score)
            alpha = max(alpha, score)
            if beta <= alpha:
                break  # Prune
        return max_score
    else:
        min_score = float('inf') #beta
        for _ in range(2):
            score = alpha_beta_pruning(lvl+1, True, alpha, beta)
            min_score = min(min_score, score)
            beta = min(beta, score)
            if beta <= alpha:
                break  # Prune
        return min_score


def mortal_kombat(player):
    results = []
    current_turn = player
    played = 0

    while played < 3:
        round_winner = "Sub-Zero" if alpha_beta_pruning(0, current_turn == 0, float('-inf'), float('inf')) == 1 else "Scorpion"
        results.append(round_winner)
        current_turn = 1-current_turn
        played += 1

    game_winner = max(set(results), key=results.count)
    return game_winner, played, results


starting_player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))
final_winner, total_rounds, round_result = mortal_kombat(starting_player)

print(f"Game Winner: {final_winner}")
print(f"Total Rounds Played: {total_rounds}")
for idx, winner in enumerate(round_result, start=1):
    print(f"Winner of Round {idx}: {winner}")



def pacman_game(c):
    leaf_nodes = [3, 6, 2, 3, 7, 1, 2, 0]
    def minimax_without_dark_magic(lvl, is_pacman, alpha, beta, start, end):
        if lvl == 3:


            return leaf_nodes[start]
        if is_pacman:
            max_val = float('-inf')
            for i in range(start, end, 2):
                val = minimax_without_dark_magic(lvl+1, False, alpha, beta, i, i + 2)
                max_val = max(max_val, val)
                alpha = max(alpha, val)
                if beta <= alpha:  # Alpha-beta pruning
                    break
            return max_val
        else:
            min_val = float('inf')
            for i in range(start, end, 2):
                val = minimax_without_dark_magic(lvl+1, True, alpha, beta, i, i + 2)
                min_val = min(min_val, val)
                beta = min(beta, val)
                if beta <= alpha:
                    break
            return min_val

    def minimax_with_dark_magic(st):
        if st == 'left':
            return max(leaf_nodes[0:2]) - c
        elif st == 'right':
            return max(leaf_nodes[4:6]) - c

    minimax_value_without = minimax_without_dark_magic(0, True, float('-inf'), float('inf'), 0, 8)

    left_with_dark_magic = minimax_with_dark_magic('left')
    right_with_dark_magic = minimax_with_dark_magic('right')
    if left_with_dark_magic > right_with_dark_magic:
        best_with_dark_magic = left_with_dark_magic
        direction = 'left'
    else:
        best_with_dark_magic = right_with_dark_magic
        direction = 'right'
    if c == 2:
