def pacman_game(c):
  NVL=[]
  NVR=[]
  val_L=[3, 6, 2, 3]
  val_R=[7, 1, 2, 0]
  Left=max(val_L)-c
  Right=max(val_R)-c

  for j in range(2,0,-1):
    for  i in range(j,len(val_L)-1):
      if i%2==0:
        NVL.append(max(val_L[i],val_L[i+1]))
        NVR.append(max(val_R[i],val_R[i+1]))
      else:
        NVL.append(min(val_L[i],val_L[i+1]))
        NVR.append(min(val_R[i],val_R[i+1]))
      val_L=NVL
      val_R=NVR
  max_without_magic=(max(NVR[0],NVL[0]))
  max_with_magic=(max(Left,Right))

  if max_with_magic>max_without_magic:
    if Right>Left:
      print(f"The new minimax value is {Right}. Pacman goes right and uses dark magic.")
    elif Left>Right:
      print(f"The new minimax value is {Left}. Pacman goes left and uses dark magic.")
  else:
    print(f"The minimax value is {max_without_magic}. Pacman does not use dark magic.")
