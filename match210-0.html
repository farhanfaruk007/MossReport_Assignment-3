#Part1


import random

def ab_pruning(node, depth, a, b, is_maximizing):
    if depth == 0 or node.terminal():
        return node.evaluate()

    if is_maximizing:
        max_val = float('-inf')
        for child in node.children():
            max_val = max(max_val, ab_pruning(child, depth - 1, a, b, False))
            a = max(a, max_val)
            if b <= a:
                break
        return max_val


    else:
        min_val = float('inf')
        for child in node.children():
            min_val = min(min_val, ab_pruning(child, depth - 1, a, b, True))
            b = min(b, min_val)
            if b <= a:
                break
        return min_val

class Node:
    def __init__(self, depth):
        self.child_nodes = []
        self.depth = depth

    def terminal(self):
        return self.depth == 0

    def evaluate(self):
        return random.choice([-1, 1])

    def children(self):
        if not self.child_nodes:
            for x in range(2):
                self.child_nodes.append(Node(self.depth - 1))
        return self.child_nodes

def simulate_round(starting_player):
    root = Node(depth=5)
    result = ab_pruning(root, depth=5, a=float('-inf'), b=float('inf'), is_maximizing=(starting_player == 1))
    if result == 1:
        return "Sub-Zero"
    else:
        return "Scorpion"

def simulate_game(initial_player):
    current_player = initial_player
    total_rounds = 3
    round_results = []

    for round_num in range(1, total_rounds + 1):
        winner = simulate_round(current_player)
        round_results.append(f"Round {round_num} Winner: {winner}")
        current_player = 1 - current_player

    scorpion_wins = 0
    subzero_wins = 0
    for result in round_results:
        if "Scorpion" in result:
            scorpion_wins += 1
        else:
            subzero_wins += 1

    if scorpion_wins > subzero_wins:
        overall_winner = "Scorpion"
    else:
        overall_winner = "Sub-Zero"
    result = f"Game Winner: {overall_winner}\n"
    result += f"Total Rounds Played: {total_rounds}\n"
    for i, winner in enumerate(round_results, 1):
        result += f" {i}: {winner}\n"
    return result

def pacman_game(dark_magic_cost):
    leaf_scores = [3, 6, 2, 3, 7, 1, 2, 0]
    min_score_left = [min(leaf_scores[0:2]), min(leaf_scores[2:4])]


    min_score_right = [min(leaf_scores[4:6]), min(leaf_scores[6:8])]

    max_left = max(min_score_left)
    max_right = max(min_score_right)

    no_darkmagic_val = max(max_right, max_left)

    darkmagic_left = max(leaf_scores[0:4]) - dark_magic_cost
    darkmagic_right = max(leaf_scores[4:8]) - dark_magic_cost
    darkmagic_val = max(darkmagic_right, darkmagic_left)
    if darkmagic_val > no_darkmagic_val:
        return f"The new minimax value is {darkmagic_val}. Pacman goes right and uses dark magic."
    else:
        return f"The minimax value is {no_darkmagic_val}. Pacman does not use dark magic."
