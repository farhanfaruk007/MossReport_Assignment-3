import random


def ab_prn(child, dpth, ap, bt, maxim_contestant, rszltz):

    if dpth == 0 or child in rszltz:
        return rszltz.get(child, 0)


    if maxim_contestant:
        maxx = float('-inf')
        for new_child in new_chld(child):
            eval =  ab_prn(new_child, dpth - 1, ap, bt, False, rszltz)
            maxx = max(maxx, eval)
            ap = max(ap, eval)
            if bt <= ap:
                break
        return maxx

    else:
        minn = float('inf')    #schorpion
        for new_child in new_chld(child):
            eval = ab_prn(new_child, dpth - 1, ap, bt, True, rszltz)
            minn = min(minn, eval)
            bt = min(bt, eval)
            if bt <= ap:
                break
        return minn


def new_chld(child):
    return [child * 2 + 1, child * 2 + 2]


class Mks:
    def __init__(self, rookie):
        self.rookie = rookie
        self.deep_dpth = 5
        self.rszltz = self.crt_outcomezz()
        self.rzlt = []

    def crt_outcomezz(self):
        l_nodez = 2 ** self.deep_dpth

        if self.rookie == 0:
            return {i: -1 for i in range(l_nodez)}
        else:
            return {i: 1 for i in range(l_nodez)}

    def s_roundz(self, active_ply):
        roott = 0

        maxim_contestant = active_ply == 1
        wn = ab_prn(
            roott, self.deep_dpth, float('-inf'), float('inf'), maxim_contestant, self.rszltz
        )
        return "Scorpion" if wn == -1 else "Sub-Zero"


    def s_war(self):
        porbo = 3
        active_ply = self.rookie

        for _ in range(porbo):
            wn = self.s_roundz(active_ply)
            self.rzlt.append(wn)

            active_ply = 1 - active_ply


        scp_w = self.rzlt.count("Scorpion")
        sbz_w = self.rzlt.count("Sub-Zero")




        return ("Scorpion" if scp_w > sbz_w else "Sub-Zero"), porbo, self.rzlt

class pmgm:
    def __init__(self, p):
        self.pricey = p
        self.l_nodes = [3, 6, 2, 3, 7, 1, 2, 0]

    def MNmax(self, dpthz,chld, max_assesment):
        if dpthz == 2:
            return self.l_nodes[chld]

        if max_assesment:
            return max(self.MNmax(dpthz + 1, chld * 2, False),
                       self.MNmax(dpthz + 1, chld * 2 + 1, False))
        else:
            return min(self.MNmax(dpthz + 1, chld * 2, True),
                       self.MNmax(dpthz + 1, chld * 2 + 1, True))

    def usemagix(self):

        zero_magixx = self.MNmax(0, 0, True)


        using_l_magixx = max(self.l_nodes[0],self.l_nodes[1]) - self.pricey
        using_r_magixx = max(self.l_nodes[4], self.l_nodes[5]) - self.pricey


        if using_r_magixx > using_l_magixx and using_r_magixx >zero_magixx:
            return using_r_magixx, "Pacman goes right and uses dark magic"
        elif using_l_magixx >zero_magixx:
            return using_l_magixx, "Pacman goes left and uses dark magic"
        else:
            return zero_magixx, "Pacman does not use dark magic"
