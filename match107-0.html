def alpha_beta_pruning(player, depth, tree_index, maximizer, tree, alpha, beta):
    if depth ==0:
        return tree[tree_index]

    if maximizer:
        v= -1000

        for i in range(0,2):
            value= alpha_beta_pruning(player, depth-1, tree_index*2 +i, False, tree, alpha, beta)


            v= max(v, value)
            alpha= max(alpha, v)

            if alpha>= beta:
                break

        return v

    else:
        v=1000

        for j in range(0,2):
            value= alpha_beta_pruning(player, depth-1, tree_index*2 +j, True, tree, alpha, beta)
            v= min(v, value)
            beta= min(beta, v)

            if alpha>= beta:
                break

        return v
