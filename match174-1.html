#task-1

import random

def alp_bit_pruning(Node1, dept, Alph, Bta, max_playr):
    if dept == 0 or Node1.is_terminal():
        return Node1.evaluate()

    if max_playr:
        max_eval = float('-inf')
        for child in Node1.get_children():
            eval = alp_bit_pruning(child, dept - 1, Alph, Bta, False)

            max_eval = max(max_eval, eval)
            Alph = max(Alph, eval)
            if Bta <= Alph:
                break
        return max_eval

    else:
        min_eval = float('inf')
        for child in Node1.get_children():

            eval = alp_bit_pruning(child, dept - 1, Alph, Bta, True)
            min_eval = min(min_eval, eval)
            Bta = min(Bta, eval)
            if Bta <= Alph:
                break
        return min_eval

class tre_nod_makeing:
    def __init__(self,identifier):
        self.identifier = identifier
        self.children = []

    def evaluate(self):
        return random.choice([-1, 1])

    def is_terminal(self):
        return len(self.children) == 0



    def get_children(self):
        return self.children

def game_playing(initial_Player):
    root = tre_nod_makeing("root")
    CUREn_player = initial_Player
    Round_Plyed = 0


    round_results = []

    subZero_wins = 0

    while Round_Plyed  < 3:
        result = alp_bit_pruning(root, 5, float('-inf'), float('inf'), CUREn_player == 1)
        if result == 1:
            Winer = "Sub-Zero"
            subZero_wins += 1
        else:
            Winer = "Scorpion"

        round_results.append(Winer)
        CUREn_player = 1 - CUREn_player
        Round_Plyed += 1


    Game_winer = "Sub-Zero" if subZero_wins > 1 else "Scorpion"

    print(f"Game Winner: {Game_winer}")
    print(f"Total Rounds Played: {Round_Plyed}")

    for i, roundWinner in enumerate(round_results, start=1):
        print(f"Winner of Round {i}: {roundWinner}")

starting_player = int(input("Enter zero or one: "))

game_playing(starting_player)

class PaCManGame:
    def __init__(self, Magic_pris):
        self.input = [3, 6, 2, 3, 7, 1, 2, 0]

        self.magic_price = Magic_pris

    def Mini_max_and_alpha_beta(self, dept, index, if_is_max, alpha, beta, use_magic):
        if dept == 3:

            return self.input[index] - self.magic_price if use_magic else self.input[index]



        if if_is_max:
            best =  float('-inf')

            for i in range(2):

                value = self.Mini_max_and_alpha_beta(dept + 1, index * 2 + i, False, alpha, beta, use_magic)
                best = max(best, value)

                alpha = max(alpha, best)
                if beta <= alpha:

                    break
            return best
        else:


            best = float('inf')
            for i in range(2):

                value = self.Mini_max_and_alpha_beta(dept + 1, index * 2 + i, True, alpha, beta, use_magic)
                best = min(best, value)
                beta = min(beta, best)

                if beta <= alpha:
                    break
            return best

    def game_run_kora(self):

        no_magic = self.Mini_max_and_alpha_beta(0, 0, True, float('-inf'), float('inf'), False)
        with_magic = self.Mini_max_and_alpha_beta(0, 0, True, float('-inf'), float('inf'), True)

        if with_magic > no_magic:
            print(f"The optimal minimax outcome is {with_magic}. Pacman uses dark magic.")
        else:
            print(f"The optimal minimax outcome is {no_magic}. Pacman does not use dark magic.")
