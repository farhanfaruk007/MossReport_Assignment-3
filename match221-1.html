import random
class MortalKombatGame:
  def __init__(self, first_player):
    self.first_player = first_player
    self.round_winners = []
    self.level = 5

  def alpha_beta_pruning(self, level, maximum, alpha, beta):
    if level == 0:
      return random.choice([-1, 1])

    if maximum:
      max_alpha = float('-inf')
      for child in range(2):
        store = self.alpha_beta_pruning(level - 1, False, alpha, beta)
        max_alpha = max(max_alpha, store)
        alpha = max(alpha, store)
        if beta <= alpha:
          break
      return max_alpha

    else:
      min_beta = float('inf')
      for child in range(2):
        store = self.alpha_beta_pruning(level - 1, True, alpha, beta)
        min_beta = min(min_beta, store)
        beta = min(beta, store)
        if beta <= alpha:
          break
      return min_beta

  def playing_round(self, starting_player):
    result = self.alpha_beta_pruning(self.level, starting_player == 1, float('-inf'), float('inf'))
    if result == -1:
      return "Scorpion"
    else:
      return "Sub-Zero"

  def simulate_game(self):
    current_player = self.first_player
    round_count = 0

    while len(self.round_winners) < 3:
      round_count += 1
      winner = self.playing_round(current_player)
      self.round_winners.append(winner)
      current_player = 1 - current_player

    scorpion_wins = self.round_winners.count("Scorpion")
    sub_zero_wins = self.round_winners.count("Sub-Zero")

    game_winner = "Scorpion"
    if scorpion_wins > sub_zero_wins:
      game_winner = "Scorpion"
    elif sub_zero_wins > scorpion_wins:
      game_winner = "Sub-Zero"

    print(f"Game Winner: {game_winner}")
    print(f"Total Rounds Played: {round_count}")
    for i in range(len(self.round_winners)):
      for j in range(len(self.round_winners)):
        if i == j:
          print(f"Winner of Round {i + 1}: {self.round_winners[j]}")

first_player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))
game = MortalKombatGame(first_player)
game.simulate_game()

class PacmanGame:
  def __init__(self, dark_magic_cost):
    self.dark_magic_cost = dark_magic_cost
    self.leaf_scores = [3, 6, 2, 3, 7, 1, 2, 0]



  def alpha_beta_pruning(self, level, maximum, alpha, beta, leaf_idx):
    if level == 3:
      return self.leaf_scores[leaf_idx]

    if maximum:
      max_alpha = float('-inf')
      for child in range(2):
        store = self.alpha_beta_pruning(level + 1, False, alpha, beta, leaf_idx * 2 + child)
        max_alpha = max(max_alpha, store)
        alpha = max(alpha, store)
        if beta <= alpha:
          break
      return max_alpha
    else:
      min_beta = float('inf')
      for child in range(2):
        store = self.alpha_beta_pruning(level + 1, True, alpha, beta, leaf_idx * 2 + child)


        min_beta = min(min_beta, store)
        beta = min(beta, store)
        if beta <= alpha:
          break
      return min_beta

  def calculate_minimax_with_magic(self):
    left_subtree_max = max(self.leaf_scores[:4])
    right_subtree_max = max(self.leaf_scores[4:])
    left_magic_score = left_subtree_max - self.dark_magic_cost
    right_magic_score = right_subtree_max - self.dark_magic_cost
    magic_score = []
    if left_magic_score >= right_magic_score:
      magic_score.append(left_magic_score)
      magic_score.append("left")
      magic_score.append(True)
    else:
      magic_score.append(right_magic_score)
      magic_score.append("right")
      magic_score.append(True)
    return magic_score

  def simulate_game(self):
    minimax_without_magic = self.alpha_beta_pruning(0, True, float('-inf'), float('inf'), 0)
    magic_value, direction, used_magic = self.calculate_minimax_with_magic()



    if magic_value > minimax_without_magic:
      print(f"The new minimax value is {magic_value}. Pacman goes {direction} and uses dark magic.")
    else:
      print(f"The minimax value is {minimax_without_magic}. Pacman does not use dark magic.")
