#Part1
class Mortal_Kombat:
  def __init__(self,p1):
    self.player_1=p1
    self.cur_r=1
    self.r=[]




  def determining(self,index1):
    if self.cur_r==1:
      if self.player_1==0:
        return 1
      else:
        return -1
    elif self.cur_r==2:
      return -1
    elif self.cur_r==3:
      return -1


  def alphabetapruning(self,depthness,index1,player_max,alpha,beta):
    if depthness==5:
      return self.determining(index1)

    if player_max:
      evaluating_max=float('-inf')
      for i in range(2):
        e=self.alphabetapruning(depthness+1,index1*2+i,False,alpha,beta)
        evaluating_max=max(evaluating_max,e)
        alpha=max(alpha,e)
        if beta<=alpha:
          break
      return evaluating_max

    else:
      evaluating_min=float('inf')
      for i in range(2):
        e=self.alphabetapruning(depthness+1,index1*2+i,True,alpha,beta)
        evaluating_min=min(evaluating_min,e)
        beta=min(beta,e)
        if beta<=alpha:
          break
      return evaluating_min

  def determining_players(self):
    player_max=self.player_1=0


    win_1=self.alphabetapruning(0,0,player_max,float('-inf'),float('inf'))
    if win_1==-1:
      self.r.append('Scorpion')
    else:
      self.r.append('Sub-Zero')
    self.player_1=1-self.player_1
    self.cur_r+=1

  def game_result(self):
    while self.cur_r<=3:
      self.determining_players()
    finalwinner= max(set(self.r),key=self.r.count)


    return finalwinner,self.r
