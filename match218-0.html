class game:
    def __init__(self, x, p, attacker, value=None):
        self.x = x
        self.p = p
        self.attacker = attacker
        self.value = value
        self.b = []
    def branch(self, branch):
        self.b.append(branch)


def tree(c, max):
    if c.x == max:
        c.value = 1 if c.p == 1 else -1
        return
    next_p = 1 - c.p
    for _ in range(2):
        new = game(c.x + 1, next_p, not c.attacker)
        c.branch(new)
        tree(new, max)


def opt(node, alpha, beta, maximize):
    if node.value is not None:
        return node.value
    if maximize:
        temp_max = float('-inf')
        for branch in node.b:
            temp_max = max(temp_max, opt(branch, alpha, beta, False))
            alpha = max(alpha, temp_max)
            if alpha >= beta:
                break
        return temp_max
    else:
        temp_min = float('inf')
        for branch in node.b:
            temp_min = min(temp_min, opt(branch, alpha, beta, True))
            beta = min(beta, temp_min)
            if beta <= alpha:
                break
        return temp_min


def battle(fp):
    rounds = 5
    root_node = game(0, fp, True)
    tree(root_node, rounds)
    active = fp
    outcomes = []
    match = 0
    while match < 3:
        match += 1
        round = opt(root_node, float('-inf'), float('inf'), root_node.attacker)
        rw = "Sub-Zero" if round == 1 else "Scorpion"
        outcomes.append(rw)
        active= 1 - active
        root_node = game(0, active, True)
        tree(root_node, rounds)
    winner = max(set(outcomes), key=outcomes.count)
    return winner, match, outcomes



fp = int(input())
champ, rounds, round_results = battle(fp)
print(f"Game Winner: {champ}")
print(f"Total Rounds Played: {rounds}")
for i, v in enumerate(round_results, 1):
    print(f"Winner of Round {i}: {v}")

def pacman_game():
    leaf = [3, 6, 2, 3, 7, 1, 2, 0]
    c = int(input(""))

    l2l = min(leaf[0:2])
    l2r = min(leaf[2:4])
    ls = max(l2l, l2r)

    l2l = min(leaf[4:6])
    l2r = min(leaf[6:8])
    rs = max(l2l, l2r)

    wdm = max(ls, rs)

    lwm = max(leaf[0:4]) - c
    rwm = max(leaf[4:8]) - c

    if lwm > wdm and lwm >= rwm:
        print(f"The new minimax value is {lwm}. Pacman goes left and uses dark magic.")
    elif rwm > wdm:
        print(f"The new minimax value is {rwm}. Pacman goes right and uses dark magic.")
    else:
        print(f"The minimax value is {wdm}. Pacman does not use dark magic.")

pacman_game()
