def roundsimulator(depth, maxplayer, alpha, beta):

    if depth == 0: #basecase
        if maxplayer:
            return -1
        else:
            return 1

    if maxplayer:  # Subzero
        maxeval = -99999
        for i in range(2):  # Branching 2
            eval = roundsimulator(depth - 1, False, alpha, beta)
            if eval > maxeval:
                maxeval = eval
            if eval > alpha:
                alpha = eval
            if beta <= alpha:
                break
        return maxeval

    else:  # Scorpion
        mineval = 99999
        for i in range(2):
            eval = roundsimulator(depth - 1, True, alpha, beta)
            if eval < mineval:
                mineval = eval
            if eval < beta:
                beta = eval
            if beta <= alpha:
                break
        return mineval

def minimaxAB(level, pacmanTurn, values, A, B):
    if level == 3: #basecase
        return values.pop(0)

    if pacmanTurn:
        maxV = -99999999
        for i in range(2):  # 2 branch
            rslt = minimaxAB(level + 1, False, values, A, B)
            maxV = max(maxV, rslt)
            A = max(A, rslt)
            if B <= A:  #pruning
                break
        return maxV
    else:
        minV = 9999999
        for j in range(2):
            rslt = minimaxAB(level + 1, True, values, A, B)
            minV = min(minV, rslt)
            B = min(B, rslt)
            if B <= A:  # prun
                break
        return minV

def minimaxDM(values, c): #dark magic
    leftV = values[1] - c  # magic left
    rightV = values[4] - c  # magic right
    return leftV, rightV
