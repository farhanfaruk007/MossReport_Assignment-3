# Part-1 [Mortal-Kombat]

import random

def minimax_with_pruning(depth,is_subzero_turn,alpha,beta):
    if depth==0:
        return random.choice([-1, 1])

    if is_subzero_turn:
        max_value=float('-inf')
        for i in range(2):
            value=minimax_with_pruning(depth - 1,False,alpha,beta)
            max_value=max(max_value,value)
            alpha=max(alpha,value)
            if beta<=alpha:
                break
        return max_value
    else:
        min_value=float('inf')
        for i in range(2):
            value=minimax_with_pruning(depth - 1,True,alpha,beta)
            min_value=min(min_value,value)
            beta=min(beta,value)
            if beta<=alpha:
                break
        return min_value

def play_mortal_kombat(start_player):
    total_rounds=3
    current_turn=start_player
    round_results={}
    scores={"Scorpion": 0, "Sub-Zero": 0}
    count=0
    for round_num in range(1,total_rounds+1):
        winner=minimax_with_pruning(5,current_turn==1,float('-inf'),float('inf'))
        if winner==-1:
            count+=1
            round_results[count]=("Scorpion")
            scores["Scorpion"]+=1
        else:
            count+=1
            round_results[count]=("Sub-Zero")
            scores["Sub-Zero"]+=1

        current_turn=1-current_turn

    if scores['Scorpion']>scores['Sub-Zero']:
      overall_winner='Scorpion'
    else:
      overall_winner='Sub-Zero'


    print(f"Game Winner: {overall_winner}")
    print(f"Total Rounds Played: {total_rounds}")

    for i, j in round_results.items():
        print(f"Winner of Round {i}: {j}")


first_player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))
play_mortal_kombat(first_player)

def pacman_game(c):
    leaf_nodes=[3,6,2,3,7,1,2,0]

    def alpha_beta(depth,is_maximizing,alpha,beta,start,end):
        if depth==3:
            if is_maximizing:
              return max(leaf_nodes[start:end])
            else:


              return min(leaf_nodes[start:end])


        if is_maximizing:
            max_eval=float('-inf')
            for i in range(start,end,2):
                eval=alpha_beta(depth+1,False,alpha,beta,i,i+2)
                max_eval=max(max_eval,eval)
                alpha=max(alpha,eval)
                if beta<=alpha:
                    break
            return max_eval
        else:
            min_eval=float('inf')
            for i in range(start,end,2):
                eval=alpha_beta(depth+1,True,alpha,beta,i,i+2)
                min_eval=min(min_eval,eval)
                beta=min(beta,eval)
                if beta<=alpha:
                    break
            return min_eval

    # Without dark magic
    root_value=alpha_beta(0,True,float('-inf'),float('inf'),0,len(leaf_nodes))

    # With dark magic
    left_with_magic=max(leaf_nodes[0:4])-c
    right_with_magic=max(leaf_nodes[4:8])-c
    magic_value=max(left_with_magic,right_with_magic)

    if magic_value>root_value:
        if left_with_magic>right_with_magic:
            print(f"The new minimax value is {magic_value}. Pacman goes left and uses dark magic.")
        else:
            print(f"The new minimax value is {magic_value}. Pacman goes right and uses dark magic.")

        print(f"The minimax value is {root_value}. Pacman does not use dark magic.")
