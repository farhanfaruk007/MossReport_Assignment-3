def alpha_beta_pruning(index, depth, first_player, leaf_values, alpha1, beta1):
  if depth==5:
    return leaf_values[index]

  if first_player == False:
    fitnessValue = 99999999999999999
    for element in range(2):
      result = alpha_beta_pruning(index*2+element,depth+1,True,leaf_values,alpha1,beta1)
      fitnessValue = min(result,fitnessValue)
      beta1 = min(beta1,fitnessValue)
      if beta1<=alpha1:
        break
    return fitnessValue
  else:
    fitnessValue = -99999999999999999

    for elem in range(2):
      result = alpha_beta_pruning(index*2+elem,depth+1,False,leaf_values,alpha1,beta1)
      fitnessValue = max(result,fitnessValue)
      alpha1 = max(alpha1, fitnessValue)
      if beta1<=alpha1:
        break
    return fitnessValue

def pacman_game(c):
    alpha1 = -99999999999999999
    beta1 = 99999999999999999
    leaf_values = [3, 6, 2, 3, 7, 1, 2, 0]
    alphaBetaOutput = alpha_beta_pruning2(0, 0, True, leaf_values,alpha1, beta1)
    rightVal = leaf_values[len(leaf_values)//2:]
    leftVal = leaf_values[:len(leaf_values)//2]
    max_right = max(rightVal)-c
    max_left = max(leftVal)-c
    maximum = max(max_right,max_left)

    if maximum>alphaBetaOutput:
        if max_right>max_left:
            print(f"The new minimax value is {max_right}. Pacman goes right and uses dark magic")
        else:
            print(f"The new minimax value is {max_left}. Pacman goes left and uses dark magic")
    else:
        print(f"The minimax value is {alphaBetaOutput}. Pacman does not use dark magic.")
