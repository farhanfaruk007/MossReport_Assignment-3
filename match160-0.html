def mortalkombat(a, b, maximize_player, alpha, beta):
    c=abs(a)
    if b == 0 or c == 1:
        return a

    if maximize_player:
        value = -float('infinity')
        for i in  [-1, 1]:  # Simulating branching factor of 2
            number = mortalkombat(i, b - 1, False, alpha, beta)
            value = max(value, number)
            alpha = max(alpha, number)
            if beta <= alpha:
                break
        return value
    else:
        value_one = float('infinity')
        for m in [-1, 1]:
            number = mortalkombat(m, b - 1, True, alpha, beta)
            value_one = min(value_one, number)
            beta = min(beta, number)
            if beta <= alpha:
                break
        return value_one

def game_simulation(variable):
    winner_name = []
    total_number_of_rounds = 0

    for round_numbers in range(1, 4):
        if round_numbers % 2 == 1:
            player_one = variable
        else:
           player_one = 1 - variable

        # The root node in our game tree
        final_result = mortalkombat(0, 5,player_one == 1, -float('infinity'), float('inf'))
        total_number_of_rounds = total_number_of_rounds+ 1

        if  final_result == 1:
            winner_name.append("Sub-Zero")
        else:
            winner_name.append("Scorpion")

    # Determining the overall game winner
    sub_zero_wins_no = winner_name.count("Sub-Zero")
    scorpion_wins_no = winner_name.count("Scorpion")

    if sub_zero_wins_no > scorpion_wins_no:
        game_winner = "Sub-Zero"
    else:
        game_winner = "Scorpion"

    return game_winner, total_number_of_rounds, winner_name
