import random

def min_max(lvl, idx, max_player, points, alpha, beta, brnch, rounds):

    if lvl == brnch: #Maximum depth
        return points[idx], rounds
    if max_player:
        max_v = float('-inf')
        winner = []
        for branch_idx in range(2):
            val, win_round = min_max(lvl + 1, idx * 2 + branch_idx, False, points, alpha, beta, brnch, rounds)


            if val > max_v:
                max_v = val
                winner = win_round

            alpha = max(alpha, max_v)

            if beta <= alpha:
                break

        if lvl < 3:
            winner.append("Sub-Zero")

        return max_v, winner

    else:  # Scorpion's turn
        min_v = float('inf')
        winner = []

        for branch_idx in range(2):
            val, win_round = min_max(lvl + 1, idx * 2 + branch_idx, True, points, alpha, beta, brnch, rounds)
            if val < min_v:
                min_v = val
                winner = win_round

            beta = min(beta, min_v)

            if beta <= alpha:
                break

        if lvl < 3:
            winner.append("Scorpion")

        return min_v, winner

def minmax(lvl,idx,maxplayer, alpha,beta):
  if lvl == 3:  #Depth = 3
    return outcome[idx]

  if maxplayer:
    max_v = float('-inf')

    for i in range(2):

      val = minmax(lvl + 1, idx * 2 + i, False,alpha,beta)

      max_v = max(max_v, val)

      alpha = max(alpha, max_v)

      if beta <= alpha:
        break

    return max_v



  else:
    min_v = float('inf')

    for i in range(2):
      val = minmax(lvl + 1, idx*2+i,True,alpha,beta)

      min_v = min(min_v,val)

      beta = min(beta,min_v)

def minmax_dark_magic(c):


  l = max(minmax(2, 0, True, -float('inf'), float('inf')),
                     minmax(2, 1, True, -float('inf'), float('inf')))
  left_magic = l - c


  r = max(minmax(2, 2, True, -float('inf'), float('inf')),
                      minmax(2, 3, True, -float('inf'), float('inf')))


  right_magic = r - c

  return max(left_magic,right_magic)



def pacman(c):

  minmax_val = minmax(0,0,True,-float('inf'),float('inf'))
