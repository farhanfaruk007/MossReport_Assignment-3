import random
def minimax(maxd, curr, choice, maxp, alpha, beta):
	if maxd == 5:
		return random.choice(choice)
	if maxp:
		temp = beta
		for i in range(0, 2):
			val = minimax(maxd + 1, curr * 2 + i, choice, 0, alpha, beta)
			temp = max(temp, val)
			alpha = max(alpha, temp)
			if beta <= alpha:
				break
		return temp
	else:
		temp = alpha
		for i in range(0, 2):
			val = minimax(maxd + 1, curr * 2 + i, choice, 1, alpha, beta)
			temp = min(temp, val)
			beta = min(beta, temp)
			if beta <= alpha:
				break
		return temp

player = int(input("enter player:"))
winner = []
choice=[1,-1]
scorpion=0
subzero=0
for i in range(3):
    alpha, beta = float('inf'), float('-inf')
    ans = minimax(0, 0, choice, player, alpha, beta)
    if ans == 1:
      winner.append('Sub-Zero')
      subzero+=1
    else:
      winner.append('Scorpion')
      scorpion+=1
    player = 1 - player
W_P=max(scorpion,subzero)
if W_P:
  print('Game winner: Sub-Zero')
else:
  print('Game winner: Scorpion')
print('Total Rounds Played: 3')
for i in range(len(winner)):
  print(f'Winner of Round {i+1}: {winner[i]}')

def pacman_game(initial_val):
    list1= [3, 6, 2, 3, 7, 1, 2, 0]
    def minimax(q, p, checker):
        if q == 3:
            return list1[p]
        if checker==True:
            child1 = minimax(q + 1, p*2, False)
            child2 = minimax(q+ 1, p*2 + 1, False)
            return max(child1, child2)
        else:
            child1 = minimax(q + 1, p*2, True)
            child2 = minimax(q + 1, p*2 + 1, True)
            return min(child1, child2)
    def helper(q, m):
        if q == 3:
            return list1[m]
        child1= helper(q + 1, m * 2)
        child2 = helper(q + 1, m * 2 + 1)
        return max(child1, child2)
    val1 = minimax(0, 0, True)
    val2= helper(0, 0) - initial_val
    if val1<val2:
        print(f"The new minimax value is {val2}. Pacman goes right and uses dark magic")
    else:
        print(f"The minimax value is {val1}. Pacman does not use dark magic")
