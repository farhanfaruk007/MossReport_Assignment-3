import math
import random

def reverse_alpha_beta_pruning(depth, max_depth, index, alpha, beta, max_player, utilities):

    if depth == 0:


        return utilities[index]

    if max_player:
        max_value = float('-inf')
        for i in range(2):
            value = reverse_alpha_beta_pruning(depth - 1, max_depth, index * 2 + i, alpha, beta, False, utilities)
            max_value = max(max_value, value)
            alpha = max(alpha, value)
            if beta <= alpha:
                break


        return max_value
    else:
        min_value = float('inf')
        for i in range(2):
            value = reverse_alpha_beta_pruning(depth - 1, max_depth, index * 2 + i, alpha, beta, True, utilities)
            min_value = min(min_value, value)
            beta = min(beta, value)
            if beta <= alpha:
                break
        return min_value

def simulate_reverse_game(start_player):
    max_depth = 5


    utilities = [random.choice([-1, 1]) for _ in range(2**max_depth)]

    results = []
    current_player = start_player

    for round_num in range(1, 4):
        result = reverse_alpha_beta_pruning(max_depth, max_depth, 0, float('-inf'), float('inf'), current_player, utilities)
        winner = "Sub-Zero" if result == 1 else "Scorpion"
        results.append(winner)
        current_player = 1 - current_player


    scorpion_win_count = results.count("Scorpion")


    sub_zero_win_count = results.count("Sub-Zero")
    game_winner = "Scorpion" if scorpion_win_count > sub_zero_win_count else "Sub-Zero"

    print(f"Game Winner: {game_winner}")
    print(f"Total Rounds Played: {len(results)}")
    for i, winner in enumerate(results, 1):
        print(f"Winner of Round {i}: {winner}")


start_player = int(input("Enter First Player (0 for Scorpion, 1 for Sub-Zero): "))
if start_player!=0 and start_player!=1:
    print("Invalid input. Please enter 0 or 1.")
else:
    simulate_reverse_game(start_player)

def pacman_game(c):

    leaf_values = [3, 6, 2, 3, 7, 1, 2, 0]

    def minimax(depth, maximizing_player, alpha, beta, node_idx):
        if depth == 0:
            return leaf_values[node_idx]



        if maximizing_player:
            max_eval = float('-inf')
            for i in range(2):
                eval = minimax(depth - 1, False, alpha, beta, node_idx * 2 + i)
                max_eval = max(max_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha:
                    break


            return max_eval
        else:
            min_eval = float('inf')
            for i in range(2):
                eval = minimax(depth - 1, True, alpha, beta, node_idx * 2 + i)
                min_eval = min(min_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha:
                    break
            return min_eval

    without_dark_magic = minimax(3, True, float('-inf'), float('inf'), 0)
    def with_dark_magic():
        left_subtree_value = max(leaf_values[0:4]) - c
        right_subtree_value = max(leaf_values[4:8]) - c
        if left_subtree_value > right_subtree_value:
            return left_subtree_value, "left"
        else:
            return right_subtree_value, "right"


    dark_magic_value, direction = with_dark_magic()
    if dark_magic_value > without_dark_magic:
        return f"The new minimax value is {dark_magic_value}. Pacman goes {direction} and uses dark magic."
    else:
        return f"The minimax value is {without_dark_magic}. Pacman does not use dark magic."


c = int(input("Enter cost:"))
print(pacman_game(c))
