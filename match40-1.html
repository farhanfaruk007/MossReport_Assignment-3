import random

def final(lvl, outcome):
    return lvl == 0 or outcome is not None

def game_dimension_tree(lvl):

    if lvl == 0:
        return {'lvl': lvl, 'outcome': random.choice([-1, 1]), 'branches': []}
    node = {"lvl": lvl, "outcome": None, 'branches': []}
    for i in range(2):
        branch = game_dimension_tree(lvl - 1)
        node['branches'].append(branch)
    return node

def alpha_beta_prun_func(node, alpha, beta, maximizing_player):


    if final(node["lvl"], node["outcome"]):
        return node["outcome"]
    if maximizing_player:
        alpha_value = float("-inf")
        for branch in node["branches"]:
            score = alpha_beta_prun_func(branch, alpha, beta, False)
            alpha_value = max(alpha_value, score)
            alpha = max(alpha, score)
            if beta <= alpha:
                break
        return alpha_value

    else:
        beta_value = float("inf")
        for branch in node["branches"]:
            score = alpha_beta_prun_func(branch, alpha, beta, True)
            beta_value = min(beta_value, score)
            beta = min(beta, score)
            if beta <= alpha:
                break
        return beta_value

def Match(fst_player):
    scorpion_score = 0
    sub_zero_score = 0
    game_results = []

    for rounds in range(1, 4):
        root = game_dimension_tree(5)
        match_outcome = alpha_beta_prun_func(root, float('-inf'), float('inf'), fst_player == 1)
        if match_outcome == 1:
            match_winner = "Sub-Zero"
            sub_zero_score += 1
        else:
            match_winner = "Scorpion"
            scorpion_score += 1
        game_results.append(f"Winner of Round {rounds}: {match_winner}")


        fst_player = 1 - fst_player

    if sub_zero_score > scorpion_score:
        tournament_winner = "Sub-Zero"
    else:
        tournament_winner = "Scorpion"

    print('Tournament Winner:', tournament_winner)
    print('Total Rounds Played:', 3)
    for result in game_results:
        print(result)


try:
    initial_move = int(input("Select initial player (0 for Scorpion, 1 for Sub-Zero): "))
    if initial_move not in [0, 1]:
        raise ValueError
    Match(initial_move)
except ValueError:

def final(level, tree_leaf):

    return level == 3 or len(tree_leaf) == 1

def minimax_with_magic(level, is_pacman, tree_leaf):
    if final(level, tree_leaf):
        return tree_leaf[0]

    mid = len(tree_leaf) // 2

    left_part = minimax_with_magic(level + 1, is_pacman, tree_leaf[:mid])
    right_part = minimax_with_magic(level + 1, is_pacman, tree_leaf[mid:])

    return max(left_part, right_part)



def minimax_without_magic(level, is_pacman, tree_leaf, alpha, beta):
    if final(level, tree_leaf):
        return tree_leaf[0]

    mid = len(tree_leaf) // 2

    if is_pacman:
        left_part = minimax_without_magic(level + 1, not is_pacman, tree_leaf[:mid], alpha, beta)
        right_part = minimax_without_magic(level + 1, not is_pacman, tree_leaf[mid:], alpha, beta)
        max_score = max(left_part, right_part)
        alpha = max(alpha, max_score)

        if beta <= alpha:
          return max_score

        return max_score

    else:

        left_part = minimax_without_magic(level + 1, not is_pacman, tree_leaf[:mid], alpha, beta)
        right_part = minimax_without_magic(level + 1, not is_pacman, tree_leaf[mid:], alpha, beta)
        min_score = min(left_part, right_part)
        beta = min(beta, min_score)

        if beta <= alpha:
            return min_score

        return min_score

def pacman_game(cost):

    tree = [3, 6, 2, 3, 7, 1, 2, 0]

    val_with_magic = minimax_with_magic(0, True, tree) - cost
    val_without_magic = minimax_without_magic(0, True, tree, float('-inf'), float('inf'))

    if val_with_magic > val_without_magic:
        print(f"The new minimax value is {val_with_magic}. Pacman uses dark magic.")
    else:
        print(f"The minimax value is {val_without_magic}. Pacman does not use dark magic.")

pacman_game(2)
pacman_game(5)
