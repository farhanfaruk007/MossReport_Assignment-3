def AlphaBetaPurning(depth, maxplayer, alpha, beta, cPlayer):
    if depth == 0:

        return random.choice([-1, 1])

    if maxplayer:
      max_cal = float('-inf')

      for i in range(2):
        eval =AlphaBetaPurning(depth-1,False,alpha,beta,cPlayer)
        max_cal=max(max_cal,eval)
        alpha= max(alpha,eval)

        if beta<=alpha:
          break



      return max_cal

    else:
      min_evaluate=float('inf')
      for i in range(2):
        eval =AlphaBetaPurning(depth-1,True,alpha,beta,cPlayer)
        min_evaluate=min(min_evaluate,eval)
        beta=min(beta,eval)
        if beta<=alpha:
          break
      return min_evaluate

  if sco_wins>sub_wins:
    game_winner="Scorpion"
  else:
    game_winner="Sub-Zero"

  print(f"Game:Winner: {game_winner}")
  print(f"Total Rounds Played: {rounds}")

  for i,winner in enumerate(r_winners,start=1):
    print(f"Winner of Round {i}: {winner}")
