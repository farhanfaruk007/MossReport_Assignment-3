import random

class BattleState:
    def __init__(self, dpth, result=None):
        self.dpth = dpth
        self.result = result
        self.children = []

    def is_terminal(self):
        return self.dpth == 0 or self.result is not None

    def add_child(self, child):
        self.children.append(child)

def build_game_tree(dpth):
    if dpth == 0:
        outcome = random.choice([-1, 1])  # -1 for Scorpion win, 1 for Sub-Zero win
        return BattleState(dpth, outcome)

    node = BattleState(dpth)
    for _ in range(2):  # Binary tree
        child = build_game_tree(dpth - 1)
        node.add_child(child)
    return node

def alpha_beta_pruning(node, alpha, beta, is_maximizing_player):
    if node.is_terminal():
        return node.result

    if is_maximizing_player:
        max_eval = float('-inf')
        for child in node.children:
            eval = alpha_beta_pruning(child, alpha, beta, False)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:
                break  # Beta cutoff
        return max_eval


    else:
        min_eval = float('inf')
        for child in node.children:
            eval = alpha_beta_pruning(child, alpha, beta, True)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break  # Alpha cutoff
        return min_eval

def simulate_mortal_kombat(first_player):
    scorpion_wins = 0
    sub_zero_wins = 0
    round_results = []

    for round_num in range(1, 4):
        print(f"\n--- Round {round_num} ---")
        print(f"Starting Player: {'Scorpion' if first_player == 0 else 'Sub-Zero'}")

        root_node = build_game_tree(5)
        outcome = alpha_beta_pruning(root_node, float('-inf'), float('inf'), first_player == 1)

        if outcome == 1:
            round_winner = "Sub-Zero"
            sub_zero_wins += 1
        else:
            round_winner = "Scorpion"
            scorpion_wins += 1

        round_results.append(f"Winner of Round {round_num}: {round_winner}")


        first_player = 1 - first_player  # Alternate the starting player

    overall_winner = "Sub-Zero" if sub_zero_wins > scorpion_wins else "Scorpion"
    print(f"\nGame Winner: {overall_winner}")

class GameNode:
    def __init__(self, value=None, children=None):
        self.value = value
        self.children = children or []

    def is_terminal(self):
        return self.value is not None


def build_tree(leaf_values):
    leaves = [GameNode(value=val) for val in leaf_values]

    #tree bottom-up
    level_2 = [GameNode(children=[leaves[i], leaves[i + 1]]) for i in range(0, len(leaves), 2)]
    level_1 = [GameNode(children=[level_2[i], level_2[i + 1]]) for i in range(0, len(level_2), 2)]
    root = GameNode(children=level_1)
    return root


def alpha_beta_minimax(node, alpha, beta, is_maximizing_player):
    if node.is_terminal():
        return node.value

    if is_maximizing_player:
        max_eval = float('-inf')
        for child in node.children:
            eval = alpha_beta_minimax(child, alpha, beta, False)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:
                break  # Beta cutoff
        return max_eval
    else:
        min_eval = float('inf')
        for child in node.children:
            eval = alpha_beta_minimax(child, alpha, beta, True)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break  # Alpha cutoff
        return min_eval


def pacman_game(leaf_values, magic_cost):
    root = build_tree(leaf_values)

    root_value_without_magic = alpha_beta_minimax(root, float('-inf'), float('inf'), True)

    left_subtree_max = max(leaf_values[0:4])
    right_subtree_max = max(leaf_values[4:8])

    left_with_magic = left_subtree_max - magic_cost
    right_with_magic = right_subtree_max - magic_cost

    # Decide best strategy
    if right_with_magic > left_with_magic:
        best_value_with_magic = right_with_magic
        direction_with_magic = "right"
    else:
        best_value_with_magic = left_with_magic
        direction_with_magic = "left"

    print(f"The minimax value is {root_value_without_magic}. Pacman does not use dark magic.")
    if best_value_with_magic > root_value_without_magic:
        print(f"The new minimax value is {best_value_with_magic}. Pacman goes {direction_with_magic} and uses dark magic.")
        print(f"Using dark magic is more advantageous for Pacman as it provides a higher value of {best_value_with_magic}.")
    else:
        print("Using dark magic is not advantageous for Pacman as the value without magic is better.")


magic_cost = int(input("Enter the cost of using dark magic (integer): "))
leaf_values = [3, 6, 2, 3, 7, 1, 2, 0]

pacman_game(leaf_values, magic_cost)
