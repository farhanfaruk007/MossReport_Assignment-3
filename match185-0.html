#task1

import random


class Node:
    def __init__(self, depth, player, score=None):
        self.depth = depth


        self.player = player
        self.score = score
        self.m = []
    def is_terminal(self):

        return self.depth == 0 or self.score is not None

    def add_child(self, child):
        self.m.append(child)


def create_tree(depth, player):
    if depth == 0:
        score = random.choice([-1, 1])
        return Node(depth, player, score)
    node = Node(depth, player)
    for _ in range(2):
        child = create_tree(depth - 1, 1 - player)


        node.add_child(child)
    return node


def alpha_beta(node, alpha, beta, is_maximizing):
    if node.is_terminal():
        return node.score
    if is_maximizing:
        max_score = float('-inf')
        for child in node.m:
            score = alpha_beta(child, alpha, beta, False)
            max_score = max(max_score, score)
            alpha = max(alpha, score)
            if beta <= alpha:
                break
        return max_score


    else:
        min_score = float('inf')
        for child in node.m:
            score = alpha_beta(child, alpha, beta, True)
            min_score = min(min_score, score)
            beta = min(beta, score)
            if beta <= alpha:
                break
        return min_score


def mortal_kombat(start_player, rounds=3):
    results = []
    for _ in range(rounds):
        tree = create_tree(5, start_player)
        winner_score = alpha_beta(tree, float('-inf'), float('inf'), start_player == 1)
        winner = "Sub-Zero" if winner_score == 1 else "Scorpion"
        results.append(winner)
        start_player = 1 - start_player

    scorpion_wins = results.count("Scorpion")
    subzero_wins = results.count("Sub-Zero")


    game_winner = "Scorpion" if scorpion_wins > subzero_wins else "Sub-Zero"
    return game_winner, results


start_player = 0
winner, round_winners = mortal_kombat(start_player)


print(f"Game Winner: {winner}")

round_num = 1

    round_num += 1

def pacman_game(cost):

    results = [3, 6, 2, 3, 7, 1, 2, 0]


    def minimax(depth, idx, is_pacman, alpha, beta):
        if depth == 3:
            return results[idx]

        if is_pacman:
            best = float('-inf')
            for i in range(2):
                score = minimax(depth + 1, idx * 2 + i, False, alpha, beta)
                best = max(best, score)
                alpha = max(alpha, score)
                if beta <= alpha:
                    break
            return best
        else:
            best = float('inf')
            for i in range(2):
                score = minimax(depth + 1, idx * 2 + i, True, alpha, beta)
                best = min(best, score)
                beta = min(beta, score)
                if beta <= alpha:
                    break
            return best

    # Calculate the best score without using dark magic
    no_score = minimax(0, 0, True, float('-inf'), float('inf'))




    left_best = max(results[0:4])
    right_best = max(results[4:8])
    left_with_magic = left_best - cost
    right_with_magic = right_best - cost


    if right_with_magic > left_with_magic:
        direction = "right"
        best_with_magic = right_with_magic
    else:
        direction = "left"
        best_with_magic = left_with_magic


    if best_with_magic > no_score:
        return f"Pacman should go {direction} and use dark magic. The new score is {best_with_magic}."
    else:
        return f"Pacman should not use dark magic. The score without magic is {no_score}."
