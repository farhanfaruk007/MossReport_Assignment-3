import random

def minMax(dep, n_ind, max_p, alp, bet, val, dep_H):
    if dep == dep_H:
        return val[n_ind]
    if max_p:
        maxVal = float('-inf')
        for k in range(2):
            res = minMax(dep + 1, n_ind * 2 + k, False, alp, bet, val, dep_H)
            maxVal = max(maxVal, res)
            alp = max(alp, res)
            if bet <= alp:
                break
        return maxVal
    else:
        minVal = float('inf')
        for l in range(2):
            res = minMax(dep + 1, n_ind * 2 + l, True, alp, bet, val, dep_H)


            minVal = min(minVal, res)
            bet = min(bet, res)
            if bet <= alp:
                break
        return minVal


def round_check(pl_st, depMax=5):
    l_er_val = [random.choice([-1, 1]) for _ in range(2 ** depMax)]
    if pl_st == 0:
        return minMax(0, 0, False, float('-inf'), float('inf'), l_er_val, depMax)
    else:
        return minMax(0, 0, True, float('-inf'), float('inf'), l_er_val, depMax)


def simulate_game(pl_st, rounT=3):
    rWin = []
    st = pl_st
    for numR in range(1, rounT + 1):
        resR = round_check(st)
        rWin.append(resR)
        st = 1 - st
    scor_w = rWin.count(-1)
    sub_w = rWin.count(1)
    winner = "Scorpion" if scor_w > sub_w else "Sub-Zero"
    name = {1: "Sub-Zero", -1: "Scorpion"}
    resR= [name[w] for w in rWin]
    print(f"Game Winner: {winner}")
    print(f"Total Rounds Played: {rounT}")
    for i, winner in enumerate(resR, start=1):
        print(f"Winner of Round {i}: {winner}")

simulate_game(pl_st=0)



def pacman_game(c):
    lf_l = [3, 6, 2, 3]
    lf_r = [7, 1, 2, 0]
    lf_ghst_mn = min(3, 6), min(2, 3)

    lt_mg = max(lf_ghst_mn)
    rt_mg = max(rt_ghst_mn)
    lftwthmgc = max(lf_l) - c
    rtwthmgc = max(lf_r) - c
    if lftwthmgc > lt_mg and lftwthmgc >= rtwthmgc:
