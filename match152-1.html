def play(current_player):
    return 1 if current_player == 1 else -1

def game(first_player):
    current_player = first_player
    round_results = []

    for round_number in range(1, 4):
        result = play(current_player)
        round_winner = "Sub-Zero" if result == 1 else "Scorpion"


        round_results.append(f"Winner of Round {round_number}: {round_winner}")
        current_player = 1 - current_player

    scorpion_wins = sum(1 for result in round_results if "Scorpion" in result)
    sub_zero_wins = sum(1 for result in round_results if "Sub-Zero" in result)

    overall_winner = "Scorpion" if scorpion_wins > sub_zero_wins else "Sub-Zero"
    return overall_winner, len(round_results), round_results

def minimaxAB(level, pacmanTurn, values, A, B):
    if level == 3:
        return values.pop(0)

    if pacmanTurn:
        maxV = -float('inf')
        for i in range(2):
            result = minimaxAB(level + 1, False, values, A, B)
            maxV = max(maxV, result)
            A = max(A, result)
            if B <= A:
                break
        return maxV


    else:
        minV = float('inf')
        for j in range(2):
            result = minimaxAB(level + 1, True, values, A, B)
            minV = min(minV, result)


            B = min(B, result)
            if B <= A:
                break
        return minV

def minimaxDM(values, c):
    leftV = values[1] - c
    rightV = values[4] - c
    return leftV, rightV

def pacman_game(c):
    values = [3, 6, 2, 3, 7, 1, 2, 0]

    prunV = values[:]
    minimaxVNM = minimaxAB(0, True, prunV, -float('inf'), float('inf'))

    leftWM, rightWM = minimaxDM(values, c)

    if leftWM > minimaxVNM or rightWM > minimaxVNM:
        if rightWM > leftWM:
            print(f"The new minimax value is {rightWM}. Pac-Man goes right and uses dark magic.")
        else:
            print(f"The new minimax value is {leftWM}. Pac-Man goes left and uses dark magic.")
    else:
        print(f"The minimax value is {minimaxVNM}. Pac-Man does not use dark magic.")

while True:
    try:
        c = int(input("Enter the cost of dark magic: "))
        break
    except ValueError:
        print("Invalid input. Please enter a valid number for the cost.")
pacman_game(c)
