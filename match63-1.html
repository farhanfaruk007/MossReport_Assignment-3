def alphaBetaPruning(node, alpha, beta, current_depth, max_player):
    if current_depth == 0 and node.leaf_node() == True:
        return node.value

    if max_player == False:
        min_value = float('inf')


        for i in node.children_list:
            value = alphaBetaPruning(i, alpha, beta, current_depth-1, True)
            min_value = min(min_value, value)
            beta = min(beta, value)
            if alpha >= beta:
                break
        return min_value

    elif max_player==True:
        max_value = float('-inf')


        for i in node.children_list:
            value = alphaBetaPruning(i, alpha, beta, current_depth-1, False)
            max_value = max(max_value, value)
            alpha = max(alpha, value)
            if alpha >= beta:
                break
        return max_value
