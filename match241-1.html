#task—1
import random

class structure:
    def __init__(self, depth, plyr):
        self.depth=depth
        self.plyr=plyr
        self.chldrn=[]
        self.score=None


def c_tree(dpt, plyr):
    nde=structure(dpt, plyr)

    if dpt==0:
        if random.random() < 0.5:
            nde.score = -1
        else:
            nde.score = 1
    else:
        for a in range(2):
            nde.chldrn.append(c_tree(dpt - 1, 1 - plyr))
    return nde

def alpha_beta(nde, a, b, mx_playr):
    if nde.depth == 0:


        return nde.score

    if mx_playr:
        val = float('-inf')
        for cld in nde.chldrn:
            val = max(val, alpha_beta(cld, a, b, False))
            a = max(a, val)
            if a>= b:
                break
        return val
    else:
        val = float('inf')
        for cld in nde.chldrn:
            val = min(val, alpha_beta(cld, a, b, True))
            b = min(b, val)
            if a>= b:
                break
        return val

def play_game(first_player):
    rounds_played = 0
    scorpion_wins = 0
    sub_zero_wins = 0
    round_winners = []

    for _ in range(3):
        rounds_played += 1
        root_node = c_tree(5, first_player)
        result = alpha_beta(root_node, float('-inf'), float('inf'), first_player == 1)

        if result> 0:
            sub_zero_wins += 1
            round_winners.append("Sub-Zero")
        else:
            scorpion_wins += 1
            round_winners.append("Scorpion")

        first_player = 1 - first_player

    if scorpion_wins > sub_zero_wins:
        overall_winner = "Scorpion"
    else:
        overall_winner = "Sub-Zero"

    return overall_winner, rounds_played, round_winners

def minimax(nde, depth, mx_playr):
    if isinstance(nde, int):
        return nde

    if depth== 0:
        if mx_playr:
            return max(nde)
        else:
            return min(nde)


    if mx_playr:
        return max(minimax(child, depth - 1, False) for child in nde)
    else:
        return min(minimax(child, depth - 1, True) for child in nde)
