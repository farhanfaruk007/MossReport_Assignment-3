import random
class MortalKombat:
    def __init__(self, first_attack, round_number):
        self.first_attack = first_attack
        self.total_round = round_number
        self.branching_factor = 2
        self.max_depth = 5
        self.count_subzero = 0
        self.count_scorpion = 0
        self.winner = []

    def Start(self):
        current_turn = self.first_attack

        for i in range(0, self.total_round):
            winner_of_round = self.pruning(current_turn, 0, float('inf'), -float('inf'))


            if self.count_scorpion == 2 or self.count_subzero == 2:
                break


            if winner_of_round == 1:


                self.winner.append('Sub-Zero')
                self.count_subzero += 1

            else:
                self.winner.append('Scorpion')
                self.count_scorpion += 1



            current_turn = 1 - current_turn




        if self.count_scorpion < self.count_subzero:
            final_winner ='Sub-Zero'
        else:
            final_winner = 'Scorpion'

        round_winner=""
        for i in range (len(self.winner)):
          round_winner+=f"\nWinner of this Round {i+1}: {self.winner[i]}"

        return final_winner,len(self.winner),round_winner


    def pruning(self, players, current_depth, a, b):
        if current_depth == self.max_depth:
            return random.choice([-1, 1])
        if players == 1:
            node = -float('inf')


            for i in range(self.branching_factor):
                node = max(node, self.pruning(1 - players, current_depth + 1, a, b))
                a = max(a, node)
                if b <= a:
                    break
            return node
        elif players == 0:


            node = float('inf')
            for i in range(self.branching_factor):
                node = min(node, self.pruning(1 - players, current_depth + 1, a, b))
                b = min(b, node)
                if b <= a:
                    break
            return node

def pruning(depths, score, player, a, b,values):
    if depths == 3:
        return values[score]
    if player == 1:
        node = float('inf')
        for i in range(2):
            node = min(node, pruning(depths + 1,  i + score * 2, 0, a, b, values))
            b = min(b, node)
            if b <= a:
                break
        return node
    elif player == 0:
        node = float('-inf')
        for i in range(2):
            node = max(node, pruning(depths + 1,  i + score * 2, 1, a, b, values))
            a = max(a, node)
            if b <= a:
                break
        return node
