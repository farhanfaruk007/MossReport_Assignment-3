from math import inf

def buildGameTree(depth):
    count, totalNodes = 0, 0

    while count < depth:
        totalNodes += 2**count
        count += 1

    nodesList = [0] * (1 + totalNodes)

    leafNodeStart = 1
    for i in range(1, depth - 1):
        leafNodeStart = 2 * leafNodeStart + 1

    leafNodeStart += 1

    level = 0


    count = 1
    nodeStart = 1
    while level < depth - 1:

        for i in range(nodeStart, 2**level + nodeStart):
            count = i
            nodesList[i] = [nodesList[i], level, -inf, inf]

        nodeStart = count + 1
        level += 1

    return nodesList


def maxPlayer(nodesList, index, alpha, beta):
    if type(nodesList[index]) == int:
        return nodesList[index]
    
    value = -inf


    for i in range(2 * index, 2 * index + 1 + 1):
        newValue = minPlayer(nodesList, i, alpha, beta)
        if newValue > value:
            value = newValue
            nodesList[index][0] = newValue
        if newValue >= beta:
            return value
        if newValue > alpha:
            nodesList[index][2] = newValue
    
    return nodesList[index][0]


def minPlayer(nodesList, index, alpha, beta):
    if type(nodesList[index]) == int:
        return nodesList[index]

    value = inf


    for i in range(2 * index, 2 * index + 1 + 1):
        newValue = maxPlayer(nodesList, i, alpha, beta)
        if newValue < value:
            value = newValue
            nodesList[index][0] = newValue
        if newValue <= alpha:
            return value
        if newValue < beta:
            nodesList[index][3] = newValue

    return nodesList[index][0]    


def maxPlayerWithCost(nodesList, index, alpha, beta):
    if type(nodesList[index]) == int:
        return nodesList[index]
    
    value = -inf


    for i in range(2 * index, 2 * index + 1 + 1):
        newValue = maxPlayerWithCost(nodesList, i, alpha, beta)
        if newValue > value:
            value = newValue
            nodesList[index][0] = newValue
        if newValue >= beta:
            return value
        if newValue > alpha:
            nodesList[index][2] = newValue
    
    return nodesList[index][0]


def pacmanGame(nodesList, index, cost):

def alphaBetaSearch(nodesList, index):
    value = maxPlayer(nodesList, index, nodesList[index][2], nodesList[index][3])
    return value
