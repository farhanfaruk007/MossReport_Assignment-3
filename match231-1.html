#Task-01

import random

def mortal_combat(p):
    leaf_nodes = []
    for i in range(32):
        leaf_nodes.append(random.choice([-1, 1]))

    def alphabeta_pruning(node, depth, alpha, beta, maximizingPlayer):
        if depth == 5:
            return leaf_nodes[node]

        if maximizingPlayer:


            max_v = float('-inf')
            for i in range(2):
                val = alphabeta_pruning(2 * node + i, depth + 1, alpha, beta, False)
                max_v = max(max_v, val)
                alpha = max(alpha, val)
                if beta <= alpha:
                    break
            return max_v
        else:
            min_v = float('inf')
            for i in range(2):
                val = alphabeta_pruning(2 * node + i, depth + 1, alpha, beta, True)
                min_v = min(min_v, val)
                beta = min(beta, val)
                if beta <= alpha:
                    break
            return min_v

    base = alphabeta_pruning(0, 0, float('-inf'), float('inf'), p == 1)

    return base, leaf_nodes

initial_p = 0
rounds = 3

# looping for 3 rounds
game_results = []
for round_num in range(rounds):
    if round_num % 2 == 0:
        current_p = initial_p
    else:
        current_p = 1 - initial_p

    base, leaf_nodes = mortal_combat(current_p)

    if base == 1:
        round_winner ="Sub-Zero"
    else:
        round_winner = "Scorpion"
    game_results.append((round_num + 1, round_winner))

#game winner counting
subzero_wins = 0
for round_num, winner in game_results:
    if winner != "Scorpion":
        subzero_wins += 1
scorpion_wins = rounds - subzero_wins

if scorpion_wins < subzero_wins:
    winner = "Sub-Zero"
else:
    winner = "Scorpion"




print(f"Game Winner: {winner}")
print(f"Total Rounds Played: {rounds}")
for round_num, winner in game_results:
    print(f"Winner of Round {round_num}: {winner}")

def pacman_game(c):
    leaf_nodes = [3, 6, 2, 3, 7, 1, 2, 0]

    def minimax_without_magic(node, depth, pacman_turn):
        if depth == 3:
            return leaf_nodes[node]

        if pacman_turn:
            max_v = float('-inf')
            for i in range(2):
                val = minimax_without_magic(2 * node + i, depth + 1, False)
                max_v = max(max_v, val)
            return max_v
        else:
            min_v = float('inf')
            for i in range(2):
                val = minimax_without_magic(2 * node + i, depth + 1, True)
                min_v = min(min_v, val)
            return min_v

    def minimax_with_magic(node, depth, pacman_turn):
        if depth == 3:
            return leaf_nodes[node]

        if pacman_turn:
            max_v = float('-inf')
            for i in range(2):
                val = minimax_with_magic(2 * node + i, depth + 1, False)
                max_v = max(max_v, val)
            return max_v
        else:
            max_v = float('-inf')
            for i in range(2):
                val = minimax_with_magic(2 * node + i, depth + 1, True)
                max_v = max(max_v, val)
            max_v -= c
            return max_v


    value_no_magic = minimax_without_magic(0, 0, True)
    value_with_magic = minimax_with_magic(0, 0, True)

    if value_with_magic > value_no_magic:
        choice = value_with_magic
        msg = f"The new minimax with magic value is {value_with_magic}. Pacman uses dark magic"
    else:
        choice = value_no_magic
        msg = f"The minimax without magic value is {value_no_magic}. Pacman does not use dark magic"
    print(msg)


    def alpha_beta(node, depth, alpha, beta, is_maximizing_player):
        if depth == 3:
            return leaf_nodes[node]

        if is_maximizing_player:
            max_v = float('-inf')
            for i in range(2):
                val = alpha_beta(2 * node + i, depth + 1, alpha, beta, False)
                max_v = max(max_v, val)
                alpha = max(alpha, max_v)
                if beta <= alpha:
                    break
            return max_v
        else:
            min_v = float('inf')
            for i in range(2):
                val = alpha_beta(2 * node + i, depth + 1, alpha, beta, True)
                min_v = min(min_v, val)
                beta = min(beta, min_v)
                if beta <= alpha:
                    break
            return min_v

    ab_prun = alpha_beta(0, 0, float('-inf'), float('inf'), True)

    if ab_prun >= choice:
        print(f"Here {ab_prun}>={choice}, Dark Magic is not advantageous for pac-man")
    else:
        print(f"{ab_prun}<{choice}, Dark Magic is advantageous for pac-man")

pacman_game(2)
