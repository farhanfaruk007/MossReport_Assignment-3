from math import inf

leaf_values = {i: -1 if i % 2 != 0 else 1 for i in range(32)}

def evaluate_moves(pos, depth, alpha, beta, maximizing):
    if depth == 0:
        return leaf_values[pos]

    if maximizing:
        max_score = -inf
        for branch in range(2):
            score = evaluate_moves(pos * 2 + branch, depth - 1, alpha, beta, False)
            max_score = max(max_score, score)
            alpha = max(alpha, max_score)
            if beta <= alpha:
                break
        return max_score
    else:
        min_score = inf
        for branch in range(2):
            score = evaluate_moves(pos * 2 + branch, depth - 1, alpha, beta, True)
            min_score = min(min_score, score)
            beta = min(beta, min_score)
            if beta <= alpha:
                break
        return min_score

def simulate_battle(first_player, total_rounds=3):
    current_turn = first_player
    scores = {"Scorpion": 0, "Sub-Zero": 0}
    round_winners = []

    for _ in range(total_rounds):
        result = evaluate_moves(0, 5, -inf, inf, current_turn == 1)
        round_winner = "Sub-Zero" if result == 1 else "Scorpion"
        scores[round_winner] += 1
        round_winners.append(round_winner)
        current_turn = 1 - current_turn  # Alternate player

    overall_winner = max(scores, key=scores.get)
    return round_winners, overall_winner, total_rounds

# Input
player_start = int(input("Who fights first? Enter 0 (Scorpion) or 1 (Sub-Zero): "))
round_results, game_winner, rounds_played = simulate_battle(player_start)

# Output
print(f"Game Winner: {game_winner}")
print(f"Total Rounds Played: {rounds_played}")
for i, winner in enumerate(round_results, 1):
    print(f"Winner of Round {i}: {winner}")

class PacmanGame:
    def __init__(self, magic_cost):
        self.leaf_values = [3, 6, 2, 3, 7, 1, 2, 0]
        self.magic_cost = magic_cost
        self.final_no_magic_value = None
        self.final_with_magic_value = None



    def alpha_beta_pruning(self, depth, index, is_pacman_turn, alpha, beta):

        if depth == 3:
            return self.leaf_values[index]

        if is_pacman_turn:
            max_value = float('-inf')
            for i in range(2):
                value = self.alpha_beta_pruning(depth + 1, index * 2 + i, False, alpha, beta)
                max_value = max(max_value, value)
                alpha = max(alpha, max_value)
                if beta <= alpha:
                    break
            return max_value
        else:
            min_value = float('inf')
            for i in range(2):
                value = self.alpha_beta_pruning(depth + 1, index * 2 + i, True, alpha, beta)


                min_value = min(min_value, value)
                beta = min(beta, min_value)
                if beta <= alpha:
                    break
            return min_value

    def calculate_with_magic(self):
        left_subtree_max = max(self.leaf_values[:4])
        right_subtree_max = max(self.leaf_values[4:])

        left_with_magic = left_subtree_max - self.magic_cost
        right_with_magic = right_subtree_max - self.magic_cost

        self.final_with_magic_value = max(left_with_magic, right_with_magic)
        if left_with_magic >= right_with_magic:
            return "left", left_with_magic
        else:
            return "right", right_with_magic

    def play_game(self):
        self.final_no_magic_value = self.alpha_beta_pruning(0, 0, True, float('-inf'), float('inf'))
        direction, value_with_magic = self.calculate_with_magic()


        if self.final_with_magic_value > self.final_no_magic_value:
            print(f"The new minimax value is {self.final_with_magic_value}. Pacman goes {direction} and uses dark magic.")
        else:
            print(f"The minimax value is {self.final_no_magic_value}. Pacman does not use dark magic.")
