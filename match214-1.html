def evaluate_node(level, index, is_max_turn, scores, min_val, max_val):
    if level == 3:
        return scores[index]

    if is_max_turn:
        best_score = float('-inf')
        for step in range(2):
            value = evaluate_node(level + 1, index * 2 + step, False, scores, min_val, max_val)
            best_score = max(best_score, value)
            min_val = max(min_val, value)
            if max_val <= min_val:
                break
        return best_score
    else:
        best_score = float('inf')
        for step in range(2):
            value = evaluate_node(level + 1, index * 2 + step, True, scores, min_val, max_val)
            best_score = min(best_score, value)
            max_val = min(max_val, value)
            if max_val <= min_val:
                break
        return best_score


def khela_hobe_khela(starting_fighter):
    outcomes = [-1, 1, -1, 1, -1, 1, -1, 1]


    round_results = []
    total_rounds = 3
    current_turn = starting_fighter

    for match in range(total_rounds):
        match_outcome = evaluate_node(0, 0, current_turn == 1, outcomes, float('-inf'), float('inf'))
        winner = "Sub-Zero" if match_outcome == 1 else "Scorpion"
        round_results.append(winner)
        current_turn = 1 if current_turn == 0 else 0
    return round_results

def strategic_evaluation(index, level, pacman_khali_khai, alpha, beta, scores):

    if level == 3:
        return scores[index]

    if pacman_khali_khai:
        best_score = float('-inf')
        for move in range(2):


            value = strategic_evaluation(index * 2 + move, level + 1, False, alpha, beta, scores)
            best_score = max(best_score, value)
            alpha = max(alpha, best_score)
            if alpha >= beta:
                break
        return best_score
    else:
        best_score = float('inf')
        for move in range(2):
            value = strategic_evaluation(index * 2 + move, level + 1, True, alpha, beta, scores)
            best_score = min(best_score, value)
            beta = min(beta, best_score)
            if alpha >= beta:
                break
        return best_score


def pacman_game(c):



    outcomes = [3, 6, 2, 3, 7, 1, 2, 0]


    normal_play_value = strategic_evaluation(0, 0, True, float('-inf'), float('inf'), outcomes)


    left_path_value = max(outcomes[0], outcomes[1]) - c
    right_path_value = max(outcomes[4], outcomes[5]) - c


    if left_path_value > right_path_value:
        optimal_magic_value = left_path_value
        chosen_direction = "left"
    else:
        optimal_magic_value = right_path_value
        chosen_direction = "right"


    if optimal_magic_value > normal_play_value:
        print(f"The new minimax value is {optimal_magic_value}. Pacman goes {chosen_direction} and uses dark magic.")
    else:
        print(f"The minimax value is {normal_play_value}. Pacman does not use dark magic.")
