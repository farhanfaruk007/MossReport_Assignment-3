import random


#Tree creation:

def battleGround(depth,maximizingPlayer,a,b):
    if depth == 5:  #max Depth which is the end
        return random.choice([-1, 1])

    if maximizingPlayer == True:  # Sub0 turn



        max_val = -999999
        for i in range(2):  #child 2 ta
            value = battleGround(depth+1,False,a,b) #recursive call
            max_val=max(max_val, value)
            a = max(a,value)

            if a>=b:
                break #prune


        return max_val



    else:  # Scorpio turn
        min_value = 999999
        for j in range(2):

            value = battleGround(depth + 1, True, a, b)
            min_value = min(min_value, value)
            b = min(b, value)

            if a>=b:
                break #prune

        return min_value


def battle(strtPlayer):
    rounds = []
    currPlayer =strtPlayer
    for round_num in range(1,4):

        result=battleGround(0,currPlayer == 1,float('-inf'),float('inf'))
        if result==-1:
            rounds.append("Scorpion")
        else:
            rounds.append("Sub-Zero")
        currPlayer= 1-currPlayer #plawer swap

# Determine the overall winner



    scorpion_wins= rounds.count("Scorpion")
    sub_zero_wins= rounds.count("Sub-Zero")

    if scorpion_wins>sub_zero_wins:
        game_winner="Scorpion"
    else:
        game_winner="Sub-Zero"

    return game_winner,rounds



strtPlayer = int(input("Start Player :"))

def pacman_game(c):
    outcome = [3, 6, 2, 3, 7, 1, 2, 0] #given
#a-b pruning
    def minimax(depth,idx,maximizingPlayer,a,b):
        if depth ==3:
            return outcome[idx]
        if maximizingPlayer ==True:  # suppose pacman always turns firs
            maxVal = -999999
            for i in range(2): #2ta child hobe

                val=minimax(depth+1,idx*2+i,False,a,b)

                maxVal =max(maxVal,val)
                a =max(a,val)

                if a>=b:#Prune hobe
                    break
            return maxVal


        else:  #Ghost turn:
            min_val = +99999

            for i in range(2):  #2child

                val = minimax(depth+1,idx*2+i,True,a,b)
                min_val = min(min_val, val)

                b = min(b, val)
                if a>=b:  # Prune hobe
                    break
            return min_val
    noDarkRootRes= minimax(0, 0, True, float('-inf'), float('inf')) #always 3
# Left subtree:
    maxLval = max(outcome[0:4])  #first 4 ta mile left subTree ; max val 6

    LDarkMag =maxLval-c
# Right subtree:
    maxRval= max(outcome[4:8]) #last 4 ta mile right subTree ; max val = 7
    RDarkMag =maxRval-c
# calculate which is better:
    if LDarkMag >RDarkMag and LDarkMag> noDarkRootRes:
        print(f"The new minimax value is {LDarkMag}. Pacman goes left and uses dark magic.")
    elif RDarkMag>LDarkMag and RDarkMag>noDarkRootRes:
        print(f"The new minimax value is {RDarkMag}. Pacman goes right and uses dark magic.")
    else:
        print(f"The minimax value is {noDarkRootRes}. Pacman does not use dark magic.")
