import random
#task 1
# 0 = scorpio
# 1 = sub-zero
# winner = -1 for scorpio and 1 for subzero
# branching factor = 2
# max depth = 5

def mortal_kombat(player):
  current_player = player
  winner_list = []
  max_tree_depth = 5
  branching_factor = 2


  #creating game tree first
  def generate_tree(depth, branching_factor):
    if depth == 0:
      return random.choice([-1, 1])
    return [generate_tree(depth - 1, branching_factor) for _ in range(branching_factor)]

  # alpha beta pruning algorithm


  def alpha_beta_pruning(node, depth, maximizing_player, alpha, beta):
    if depth == 0:
      return node

    if maximizing_player:
      best_eval = float('-inf')
      for child in node:
        eval = alpha_beta_pruning(child, depth - 1, False, alpha, beta)
        best_eval = max(best_eval, eval)


        alpha = max(alpha, eval)
        if alpha >= beta:
          break
      return best_eval
    else:
      best_eval = float('inf')
      for child in node:
        eval = alpha_beta_pruning(child, depth - 1, True, alpha, beta)
        best_eval = min(best_eval, eval)
        beta = min(beta, eval)
        if alpha >= beta:
          break
      return best_eval


  def game_simulation(current_player):
      tree = generate_tree(max_tree_depth, branching_factor)
      result = alpha_beta_pruning(tree, max_tree_depth,current_player, float('-inf'), float('inf'))
      if result == -1:
        winner = 0
      else:
        winner = 1
      return winner

  player_list = []
  numOfRounds =  3 #taking fixed number of rounds as per the output of the question
  for i in range(numOfRounds):
    win = game_simulation(current_player)
    winner_list.append(win)
    if win == 0:
      player_list.append("Scorpio")
    else:
      player_list.append("Sub-Zero")
    current_player = 1-current_player

  if winner_list.count(0)>winner_list.count(1):
    winningPlayer = "Scorpio"
  else:
    winningPlayer = "Sub-Zero"

  print(f"Game Winner: {winningPlayer} \nTotal Rounds Played: {numOfRounds}")
  for i in range(numOfRounds):
    print(f"Winner of Round {i+1}: {player_list[i]}")


player = int(input("Enter the starting player (0 for Scorpion, 1 for Sub-Zero): "))
mortal_kombat(player)
