import random

class Node:
    def __init__(self, depth, starting_player):
        self.left = None
        self.right = None
        self.parent = None
        self.depth = depth
        self.alpha = float('-inf')
        self.beta = float('inf')

        if self.depth == 5:
            self.value = random.choice([-1, 1])
        else:
            if starting_player == 0:
                if self.depth % 2 == 0:
                    self.value = float('inf')
                    self.mode = 'minimize'
                else:
                    self.value = float('-inf')
                    self.mode = 'maximize'
            else:
                if self.depth % 2 == 1:
                    self.value = float('inf')
                    self.mode = 'minimize'
                else:
                    self.value = float('-inf')
                    self.mode = 'maximize'

def buildGameTree(start_player):
    root = Node(0, start_player)
    addChildNodes(root, 0, start_player)
    return root

def addChildNodes(current, level, starter):
    if level == 5:
        return
    else:


        current.left = Node(level + 1, starter)
        current.left.parent = current
        addChildNodes(current.left, level + 1, starter)

        current.right = Node(level + 1, starter)
        current.right.parent = current
        addChildNodes(current.right, level + 1, starter)

def alphaBeta(node):
    if node.left is None and node.right is None:


        return node.value

    if node.mode == 'maximize':
        for child in [node.left, node.right]:
            child_value = alphaBeta(child)
            node.value = max(node.value, child_value)
            node.alpha = max(node.alpha, child_value)
            if node.alpha >= node.beta:
                break
    else:
        for child in [node.left, node.right]:
            child_value = alphaBeta(child)
            node.value = min(node.value, child_value)
            node.beta = min(node.beta, child_value)
            if node.alpha >= node.beta:
                break

    return node.value

class TreeNode:
    def __init__(self, level, magic, leaf_values):
        self.left = None
        self.right = None
        self.parent = None
        self.level = level
        self.alpha = float('-inf')
        self.beta = float('inf')

        if self.level == 3:
            self.value = leaf_values.pop(0)
        else:


            if magic == 0 and self.level == 1:
                self.value = float('inf')  # Ghost minimize krbe
                self.type = 'min'
            else:
                self.value = float('-inf')  # Pacman maximize krbe
                self.type = 'max'


def buildTree(magic, leaf_values):
    root = TreeNode(0, magic, leaf_values[:])
    generateNodes(root, 0, magic, leaf_values[:])


    return root


def generateNodes(current, depth, magic, leaf_values):
    if depth == 3:
        return
    else:
        current.left = TreeNode(depth + 1, magic, leaf_values)
        current.left.parent = current
        generateNodes(current.left, depth + 1, magic, leaf_values)

        current.right = TreeNode(depth + 1, magic, leaf_values)
        current.right.parent = current
        generateNodes(current.right, depth + 1, magic, leaf_values)


def alphaBetaEvaluation(node):
    if not node.left and not node.right:


        return node.value

    if node.type == 'max':
        for child in [node.left, node.right]:
            eval_value = alphaBetaEvaluation(child)
            node.value = max(node.value, eval_value)
            node.alpha = max(node.alpha, eval_value)
            if node.alpha >= node.beta:
                break
    else:
        for child in [node.left, node.right]:
            eval_value = alphaBetaEvaluation(child)
            node.value = min(node.value, eval_value)
            node.beta = min(node.beta, eval_value)
            if node.alpha >= node.beta:
                break

    return node.value


def evaluateGame(magic, leaf_values):
    game_tree_root = buildTree(magic, leaf_values)
    return alphaBetaEvaluation(game_tree_root)
