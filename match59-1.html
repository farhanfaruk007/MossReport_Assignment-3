import math




def alpha_beta_game_algo(node,depth_ofTree,alpha,beta,max_player_flag):
    if depth_ofTree==0:
        return node.value
    if max_player_flag:
        maximum_values = -math.inf
        for i in node.child:
            value = alpha_beta_game_algo(i, depth_ofTree - 1, alpha, beta, False)
            maximum_values = max(maximum_values, value)
            alpha = max(alpha, value)
            if alpha>= beta:
                break
        return maximum_values
    else:
        minimum_values = math.inf
        for j in node.child:
            value = alpha_beta_game_algo(j, depth_ofTree - 1, alpha, beta, True)
            minimum_values = min(minimum_values, value)
            beta = min(beta, value)
            if alpha>= beta:
                break
        return minimum_values



class NodeClass:
    def __init__(self,value_of_node,child_node=[]):
        self.value=value_of_node
        self.child=child_node

def pacman_game(c):
    given_values=[3, 6, 2, 3, 7, 1, 2, 0]

    leafs_store=[]
    for value in given_values:
        leafs_store.append(NodeClass(value))

    level_2=[]
    for i in range(0, len(leafs_store), 2):
        left_child = leafs_store[i]
        right_child = leafs_store[i + 1]
        level_2.append(NodeClass(None, [left_child, right_child]))

    level_1 = []
    for i in range(0, len(level_2), 2):
        left_child = level_2[i]
        right_child = level_2[i + 1]
        level_1.append(NodeClass(None, [left_child, right_child]))
    tree_root= NodeClass(None,level_1)

    magic_with_game =alpha_beta_game_algo(tree_root,3,-math.inf,math.inf,True)

    #brings out the subtree max
    LeftsubMax= max(given_values[:4])-c
    RightsubMax=max(given_values[4:])-c

    #decide pacman moves
    if LeftsubMax>magic_with_game  and LeftsubMax-magic_with_game >=RightsubMax-magic_with_game  :

        print(f"The new minimax value is {LeftsubMax}. Pacman goes left and uses dark magic")
    elif RightsubMax>magic_with_game  and LeftsubMax-magic_with_game <=RightsubMax-magic_with_game :
        print(f"The new minimax value is {RightsubMax}. Pacman goes right and uses dark magic")
    else:
        print(f"The minimax value is {magic_with_game }. Pacman does not use dark magic")
