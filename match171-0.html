import random

class Node:
    def __init__(self, value):
        self.data = value
        self.left = None
        self.right = None

    def game_tree(self,d):
        if d == 5:
            value = random.choice([1, -1])
            self.data = value


        else:
            self.left = Node(None)
            self.right = Node(None)
            self.left.game_tree(d + 1)
            self.right.game_tree(d + 1)

    def alpha_beta_pruning(self, alpha, beta, d):
      if self.left == None and self.right == None:
          return self.data

      if d%2==0:
          maxvalue = float('-inf')
          if self.left:
              value = self.left.alpha_beta_pruning(alpha, beta, d+1)
              if maxvalue < value:
                maxvalue = value
              if alpha < value:
                alpha  = value
          if beta <= alpha:
              return maxvalue
          if self.right:
              value = self.left.alpha_beta_pruning(alpha, beta, d+1)
              if maxvalue < value:
                maxvalue = value
              if alpha < value:
                alpha  = value
          return maxvalue
      else:
          minvalue = float('inf')
          if self.left:
              value = self.left.alpha_beta_pruning(alpha, beta, d+1)
              if minvalue > value:
                maxvalue = value
              if beta > value:
                beta  = value

          if beta <= alpha:
              return minvalue
          if self.right:
              value = self.left.alpha_beta_pruning(alpha, beta, d+1)
              if minvalue > value:
                maxvalue = value
              if beta > value:
                beta  = value

          return minvalue
