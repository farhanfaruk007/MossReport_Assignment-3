import random

def alpha_beta(depth, maximizing, alpha, beta, turn):
    if depth == 0:
        return random.choice([-1, 1])

    if maximizing:


        max_score = float('-inf')
        for _ in range(2):
            score = alpha_beta(depth - 1, False, alpha, beta, 1 - turn)
            max_score = max(max_score, score)
            alpha = max(alpha, score)
            if beta <= alpha:
                break
        return max_score

    else:
        min_score = float('inf')
        for _ in range(2):
            score = alpha_beta(depth - 1, True, alpha, beta, 1 - turn)
            min_score = min(min_score, score)
            beta = min(beta, score)
            if beta <= alpha:
                break
        return min_score


def game_simulation(start):
    turn = start
    match_results = []
    for round_index in range(3):
        outcome = alpha_beta(5, turn == 1, float('-inf'), float('inf'), turn)
        if outcome == -1:
            match_results.append("Scorpion")
        else:
            match_results.append("Sub-Zero")
        turn = 1 - turn

    scorpion_victories = match_results.count("Scorpion")
    subzero_victories = match_results.count("Sub-Zero")


    overall_winner = "Scorpion" if scorpion_victories > subzero_victories else "Sub-Zero"


    print(f"Game Winner: {overall_winner}")
    print(f"Total Rounds Played: {len(match_results)}")
    for index, winner in enumerate(match_results, start=1):
        print(f"Winner of Round {index}: {winner}")


starting_player = int(input("Enter 0 for Scorpion, Enter 1 for Sub-Zero: "))

if starting_player == 0:
    print("Scorpion starts!")
elif starting_player == 1:
    print("Sub-Zero starts!")

print("")


game_simulation(starting_player)

import random

outcome_values = [3, 6, 2, 3, 7, 1, 2, 0]

def decision_tree(depth, node_index, is_maximizing, lower_bound, upper_bound):
    if depth == 3:
        return outcome_values[node_index]

    if is_maximizing:
        best_value = float('-inf')

        for branch in range(2):
            evaluation = decision_tree(
                depth + 1, node_index * 2 + branch, False, lower_bound, upper_bound
            )
            best_value = max(best_value, evaluation)
            lower_bound = max(lower_bound, best_value)
            if upper_bound <= lower_bound:
                break
        return best_value

    else:
        worst_value = float('inf')

        for branch in range(2):
            evaluation = decision_tree(
                depth + 1, node_index * 2 + branch, True, lower_bound, upper_bound
            )
            worst_value = min(worst_value, evaluation)
            upper_bound = min(upper_bound, worst_value)
            if upper_bound <= lower_bound:
                break
        return worst_value


def evaluate_with_special_cost(extra_cost):
    branches = [0, 1, 2, 3]

    left_branch_scores = [
        decision_tree(2, i, True, -float('inf'), float('inf')) for i in branches[:2]
    ]
    right_branch_scores = [
        decision_tree(2, i, True, -float('inf'), float('inf')) for i in branches[2:]
    ]

    left_adjusted = max(left_branch_scores) - extra_cost


    right_adjusted = max(right_branch_scores) - extra_cost

    return max(left_adjusted, right_adjusted)

def ghost_game(special_cost):
    standard_value = decision_tree(0, 0, True, -float('inf'), float('inf'))
    adjusted_value = evaluate_with_special_cost(special_cost)

    if adjusted_value > standard_value:
        print(f"The new decision value is {adjusted_value}. Ghost goes right and uses the special ability.")
    else:
        print(f"The new decision value is {standard_value}. Ghost does not use the special ability.")


special_cost = int(input("Enter the cost of using the special ability: "))
ghost_game(special_cost)
