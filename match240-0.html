def decide_winner(current_player):

    player_results = {1: 1, 0: -1}

    return player_results.get(current_player, -1)


def battle(first_turn):

    active_player = first_turn

    round_outcomes = []

    round_count = 1

    while round_count <= 3:

        outcome = decide_winner(active_player)
        round_champion = "Sub-Zero" if outcome == 1 else "Scorpion"

        round_outcomes.append(f"Winner of Round {round_count}: {round_champion}")
        active_player = 1 - active_player
        round_count += 1



    scorpion_victories = sum(1 for outcome in round_outcomes if "Scorpion" in outcome)
    sub_zero_victories = sum(1 for outcome in round_outcomes if "Sub-Zero" in outcome)
    overall_champion = "Scorpion" if scorpion_victories > sub_zero_victories else "Sub-Zero"
    return overall_champion, len(round_outcomes), round_outcomes

def minimax_prune(depth, is_pacman, tree_values, alpha, beta):


    if depth == 3:
        return tree_values.pop(0)

    elif is_pacman:
        max_value = -float('inf')
        i = 0
        while i < 2:
            result = minimax_prune(depth + 1, False, tree_values, alpha, beta)
            max_value = max(max_value, result)
            alpha = max(alpha, result)
            if beta <= alpha:
                break
            i += 1
        return max_value

    else:
        min_value = float('inf')
        j = 0


        while j < 2:
            result = minimax_prune(depth + 1, True, tree_values, alpha, beta)
            min_value = min(min_value, result)
            beta = min(beta, result)
            if beta <= alpha:
                break
            j += 1


        return min_value


def calculate_dark_magic(tree_values, cost):

    left_result = tree_values[1] - cost

    right_result = tree_values[4] - cost
    return left_result, right_result

def pacman_decision(cost):

    game_tree = [3, 6, 2, 3, 7, 1, 2, 0]

    pruned_tree = game_tree[:]

    minimax_value = minimax_prune(0, True, pruned_tree, -float('inf'), float('inf'))

    left_magic, right_magic = calculate_dark_magic(game_tree, cost)

    if left_magic > minimax_value or right_magic > minimax_value:
        direction = "right" if right_magic > left_magic else "left"
        new_value = max(left_magic, right_magic)
        print(f"The new minimax value is {new_value}. Pac-Man goes {direction} and uses dark magic.")
    else:
        print(f"The minimax value is {minimax_value}. Pac-Man does not use dark magic.")

        magic_cost = int(input("Enter the cost of dark magic: "))
    except ValueError:

pacman_decision(magic_cost)
