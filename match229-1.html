import random
def generate_leaf(depth):
    total_leaf = 2 ** depth
    nodes = []
    for i in range(total_leaf):
        nodes.append(random.choice([-1, 1]))
    return nodes



def AB_pruning(depth, alpha, beta, maximizing, leaf_nodes, index=0):
    if depth == 0:
        return leaf_nodes[index]
    if maximizing:
        max_value = float('-inf')
        for i in range(2):
            value = AB_pruning(depth - 1, alpha, beta, False, leaf_nodes, index * 2 + i)
            max_value = max(max_value, value)
            alpha = max(alpha, value)
            if beta <= alpha:
                break
        return max_value
    else:
        min_value = float('inf')
        for i in range(2):
            value = AB_pruning(depth - 1, alpha, beta, True, leaf_nodes, index * 2 + i)
            min_value = min(min_value, value)
            beta = min(beta, value)
            if beta <= alpha:
                break
        return min_value

def game(user):
    depth = 5
    leaf_nodes = generate_leaf(depth)
    winners = []
    total_rounds = 0
    player = user

    while len(leaf_nodes) > 1:
        total_rounds += 1
        is_maximizing = player == 1

        winner_value = AB_pruning(depth, float('-inf'), float('inf'), is_maximizing, leaf_nodes)
        if winner_value == 1:
          round_winner = "Sub-Zero"
        else:
          round_winner = "Scorpion"
        winners.append(round_winner)

        new_leaf_nodes = []
        for i in range(0, len(leaf_nodes) - 1, 2):
            if is_maximizing:
                new_leaf_nodes.append(max(leaf_nodes[i], leaf_nodes[i + 1]))
            else:
                new_leaf_nodes.append(min(leaf_nodes[i], leaf_nodes[i + 1]))

        leaf_nodes = new_leaf_nodes
        depth -= 1
        player = 1 - player

    if leaf_nodes[0] == 1:
      final_winner = "Sub-Zero"
    else:
      final_winner = "Scorpion"

    print("Game Winner:", final_winner)
    print("Total Rounds Played:", total_rounds)
    for i in range(len(winners)):
        print("Winner of Round", i + 1, ":", winners[i])

user=int(input("Enter 0 or 1 : "))
game(user)
