def player(current):
    if current == 1:
        return 1
    else:
        return -1

def game(first):
    round_p = 0
    current = first
    win_list = []


    for number_r in range(1, 4):
        result = player(current)
        round_p += 1

        if result == 1:
            win_list.append(f"Winner of Round {number_r}: Sub-Zero")
        else:


            win_list.append(f"Winner of Round {number_r}: Scorpion")

        current = 1 - current

    win_scorp = sum(1 for winner in win_list if "Scorpion" in winner)
    win_sub = sum(1 for winner in win_list if "Sub-Zero" in winner)

    if win_scorp > win_sub:
        won_p = "Scorpion"
    else:
        won_p = "Sub-Zero"

    return won_p, round_p, win_list

def min_max(lvl, turn_p, val, A, B):



    if turn_p:
        val_max = float('-inf')
        for i in range(2):
            product = min_max(lvl + 1, False, val, A, B)
            val_max = max(val_max, product)
            A = max(A, product)
            if B <= A:
                break
        return val_max

    if lvl == 3:
        return val.pop(0)

    else:
        val_min = float('inf')
        for j in range(2):
            product = min_max(lvl + 1, True, val, A, B)


            val_min = min(val_min, product)
            B = min(B, product)
            if B <= A:
                break
        return val_min

def min_max_DM(val, c):

    leftV = val[1] - c
    rightV = val[4] - c
    return leftV, rightV


def pacman_game(c):

    val = [3, 6, 2, 3, 7, 1, 2, 0]
    val_prun = val[:]
    min_max_f = min_max(0, True, val_prun, -float('inf'), float('inf'))

    m_left, m_right = min_max_DM(val, c)

    if m_left > min_max_f or m_right > min_max_f:
        if m_right > m_left:
