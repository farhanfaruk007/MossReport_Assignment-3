import random



class MortalGame:
    def __init__(self, fir_plyr):
        self.fir_plyr = fir_plyr
        self.plars = ["Scorpion", "Sub-Zero"]
        self.tot_rou = 3
        self.rouns = []


    def al_be_pru(self, deth, maxia_plyr):
        if deth == 0:
            ashbe = self.opinion()
            return ashbe



        chek = []
        for mi in range(2):
            wer = self.al_be_pru(deth - 1, not maxia_plyr)
            chek.append(wer)

        if maxia_plyr:
            return max(chek)
        else:
            return min(chek)
    def opinion(self):
        mer = random.choice([-1, 1])
        return mer



    def gamekhele(self):
        for _ in range(self.tot_rou):
            champ_h = self.al_be_pru(5, self.fir_plyr == 0)
            self.rouns.append("Scorpion" if champ_h == -1 else "Sub-Zero")
        yer = "Scorpion" if self.rouns.count("Scorpion") > self.rouns.count("Sub-Zero") else "Sub-Zero"
        return yer

    vale_lef = [3, 6, 2, 3, 7, 1, 2, 0]
    ashbe = minimmaxi(0, 0, True, vale_lef, float('-inf'), float('inf'))

    dark_magicleft = vale_lef[1] - c
    dark_magicright = vale_lef[4] - c
    bhalo_dark_magic = max(dark_magicleft, dark_magicright)

    dark_magic = max(bhalo_dark_magic, max(vale_lef[:4]) - c, max(vale_lef[4:]) - c)
    neh = f"The new minimax value is {dark_magic if dark_magic > ashbe else ashbe}. Pacman {'goes right and uses' if dark_magic > ashbe else 'does not use'} dark magic"
    return neh

def minimmaxi(deth, inex, i_maxi, vale, alph, bet):
    if deth == 3:
        return vale[inex]

    bhalo = float('-inf') if i_maxi else float('inf')
    for mi in range(2):
        vale = minimmaxi(deth + 1, inex * 2 + mi, not i_maxi, vale, alph, bet)
        bhalo = max(bhalo, vale) if i_maxi else min(bhalo, vale)
        alph, bet = max(alph, bhalo), min(bet, bhalo)
        if bet <= alph:
            break
    return bhalo
