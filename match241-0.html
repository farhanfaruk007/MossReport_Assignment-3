# Task 2
import random

class GameNode:
    def __init__(self, depth, max_depth, player, value=None):
        self.children = []
        self.depth = depth
        self.max_depth = max_depth
        self.player = player
        self.value = value

    def create_children(self):
        if self.depth < self.max_depth:
            self.children.extend([GameNode(self.depth + 1, self.max_depth, 1 - self.player)
                                  for _ in range(2)])

            for child in self.children:
                child.create_children()


        elif self.value is None:
            if self.depth % 2 == 0:
                weight = [0.8, 0.2] if self.player == 0 else [0.2, 0.8]
            else:
                weight = [0.6, 0.4] if self.player == 1 else [0.4, 0.6]

            self.value = random.choices([-1, 1], weight)[0]

def alpha_beta_pruning(node, alpha, beta, is_maximizing_player):
    if len(node.children) == 0 or node.depth == node.max_depth:


        return node.value

    if is_maximizing_player:
        max_eval = float('-inf')
        for child in node.children:
            max_eval = max(max_eval, alpha_beta_pruning(child, alpha, beta, False))
            alpha = max(alpha, max_eval)
            if alpha >= beta:
                break
        return max_eval

    else:
        min_eval = float('inf')
        for child in node.children:
            min_eval = min(min_eval, alpha_beta_pruning(child, alpha, beta, True))
            beta = min(beta, min_eval)
            if beta <= alpha:
                break
        return min_eval

def simulate_battle(starting_player):
    max_depth = 5
    root_node = GameNode(0, max_depth, starting_player)

    root_node.create_children()

    final_score = alpha_beta_pruning(root_node, float('-inf'), float('inf'), starting_player == 0)

    if final_score > 0:
        victor = 'Sub-Zero'
    else:
        victor = 'Scorpion'

    return victor, final_score


start_player = int(input("Enter starting player (0 for Scorpion, 1 for Sub-Zero): "))

battle_winner, battle_score = simulate_battle(start_player)

round_results = []


for _ in range(3):
    while True:
        round_winner, _ = simulate_battle(1 - start_player)
        if not round_results or round_winner != round_results[-1]:
            round_results.append(round_winner)
            break


print('Game Winner: {}'.format(battle_winner))
print('Total Rounds Played: {}'.format(len(round_results)))

for index, round_winner in enumerate(round_results, start=1):
    print('Winner of Round {}: {}'.format(index, round_winner))

def minimax(node, depth, alpha, beta, is_maximizing_player):
    if depth == 0 or not node:
        return node
    if is_maximizing_player:
        max_value = -float("inf")

        for child in node:
            value = minimax(child, depth - 1, alpha, beta, False)
            max_value = max(max_value, value)
            alpha = max(alpha, max_value)
            if beta <= alpha:
                break
        return max_value
    else:
        min_value = float("inf")
        for child in node:
            value = minimax(child, depth - 1, alpha, beta, True)
            min_value = min(min_value, value)
            beta = min(beta, min_value)
            if beta <= alpha:
                break
        return min_value



def pacman_game(dark_magic_cost):

    game_tree = [[3, 6], [2, 3], [7, 1], [2, 0]]
    minimax_value = minimax(game_tree, 2, -float("inf"), float("inf"), True)

    left_subtree_value = max(game_tree[0])
    right_subtree_value = max(game_tree[2])


    left_with_magic = left_subtree_value - dark_magic_cost
    right_with_magic = right_subtree_value - dark_magic_cost


    if dark_magic_cost < minimax_value:
        if left_with_magic > right_with_magic and left_with_magic > minimax_value:
            return f"The new minimax value is {left_with_magic}. Pacman goes left and uses dark magic"


        elif right_with_magic > left_with_magic and right_with_magic > minimax_value:
            return f"The new minimax value is {right_with_magic}. Pacman goes right and uses dark magic"
        else:
            return f"The minimax value is {minimax_value}. Pacman does not use dark magic"
    else:
        return f"The minimax value is {minimax_value}. Pacman does not use dark magic"


magic_cost =int(input("Enter the cost of using dark magic: "))
game_result = pacman_game(magic_cost)
