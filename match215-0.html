# Part-01:

class MortalKombat:
    def __init__(self, starting_player):
        self.starting_player = starting_player
        self.total_round = 0
        self.round_winners = []

    def alpha_beta_pruning(self, depth, current_player, alpha, beta):
        if depth == 5:
            return -1 if current_player else 1



        if current_player:
            max_val = float('-inf')
            for _ in range(2):
                val = self.alpha_beta_pruning(depth + 1, False, alpha, beta)
                max_val = max(max_val, val)
                alpha = max(alpha, val)
                if alpha >= beta:
                  break
            return max_val
        else:
            min_val = float('inf')
            for _ in range(2):
                val = self.alpha_beta_pruning(depth + 1, True, alpha, beta)
                min_val = min(min_val, val)
                beta = min(beta, val)
                if alpha >= beta:
                  break
            return min_val

    def play(self):
        current_player = self.starting_player
        total_round = []
        while self.total_round < 5:
            result = self.alpha_beta_pruning(0, current_player == 0, float('-inf'), float('inf'))
            self.round_winners.append("Scorpion" if result == -1 else "Sub-Zero")
            self.total_round += 1
            current_player = 1 - current_player

        winner = max(set(self.round_winners), key=self.round_winners.count)
        return winner, self.total_round, self.round_winners


starting_player = 0
game = MortalKombat(starting_player)
winner, total_round, round_winners = game.play()
print(f"Game Winner: {winner}")
print(f"Total Rounds Played: {total_round}")
for i, rw in enumerate(round_winners, 1):
    print(f"Winner of Round {i}: {rw}")
