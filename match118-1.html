def alpha_beta_pruning(depth, index, player, alpha, beta, value):
    if depth == 3:
        return value[index]
    if player:
        maxeval = float('-inf')
        for i in range(0, 2):
            n_depth = depth + 1
            n_node = index * 2 + i
            eval = alpha_beta_pruning(n_depth, n_node, False, alpha, beta, value)
            maxeval = max(maxeval, eval)
            alpha = max(alpha, eval)
            if alpha >= beta:
                break
        return maxeval
    else:
        mineval = float('inf')
        for i in range(0, 2):
            n_depth = depth + 1
            n_node = index * 2 + i
            eval = alpha_beta_pruning(n_depth, n_node, True, alpha, beta, value)
            mineval = min(mineval, eval)
            beta = min(beta, eval)
            if alpha >= beta:
                break
        return mineval

def leaf_values():
    return [-1, 1, 1, -1, 1, -1, -1, 1]



def simulate_game(start):
    values = leaf_values()
    rounds = 3
    round_winner = []
    current_player = start
    for round_num in range(rounds):
        player = current_player == 1
        winner = alpha_beta_pruning(0, 0, player, float('-inf'), float('inf'), values)
        if winner == -1:
            round_winner.append('Scorpion')
            current_player = 1
        else:
            round_winner.append('Sub-Zero')
            current_player = 0

    scorpion = round_winner.count('Scorpion')
    sub_zero = round_winner.count('Sub-Zero')
    if scorpion > sub_zero:
        game_winner = 'Scorpion'
    elif scorpion < sub_zero:
        game_winner = 'Sub-Zero'
    else:
        game_winner = 'Draw'
    print(f'Game Winner: {game_winner}')
    print(f'Total Rounds Played: {rounds}')
    for i, winner in enumerate(round_winner):
        print(f'Winner of Round {i+1}: {winner}')

start = int(input())
simulate_game(start)

def pacman_game(c):

    leaf_nodes = [3, 6, 2, 3, 7, 1, 2, 0]


    left_subtree = max(min(leaf_nodes[0], leaf_nodes[1]), min(leaf_nodes[2], leaf_nodes[3]))
    right_subtree = max(min(leaf_nodes[4], leaf_nodes[5]), min(leaf_nodes[6], leaf_nodes[7]))
    minimax_value_without_magic = max(left_subtree, right_subtree)


    left_subtree_with_magic = max(leaf_nodes[0], leaf_nodes[1]) - c
    right_subtree_with_magic = max(leaf_nodes[4], leaf_nodes[5]) - c
    minimax_value_with_magic = max(left_subtree_with_magic, right_subtree_with_magic)


    if minimax_value_with_magic > minimax_value_without_magic:
        if minimax_value_with_magic == right_subtree_with_magic:
            print(f"The new minimax value is {minimax_value_with_magic}. Pacman goes right and uses dark magic.")
        else:
            print(f"The new minimax value is {minimax_value_with_magic}. Pacman goes left and uses dark magic.")
    else:
        print(f"The minimax value is {minimax_value_without_magic}. Pacman does not use dark magic.")


c = int(input("Input (c): "))
pacman_game(c)
