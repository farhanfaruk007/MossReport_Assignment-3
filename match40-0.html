#PART1



import random


def is_final(level,outcome):
    return level== 0 or outcome is not None
def game_tree(level):
    if level==0:    #level 0 means the game is at the last stage
        return {'level': level, 'outcome': random.choice([-1, 1]), 'branches': []}

    leaf={'level': level, 'outcome': None, 'branches': []}      #from the begining state outcome is NONE and the BRANCHES are empty
    for i in range(2):
        branch= game_tree(level-1)
        leaf['branches'].append(branch)       #storing new branch in the branches list
    return leaf
def alpha_beta_func(leaf,alpha,beta,is_maximizer):   #is_maximizer determines if the level is maximising or minimizing. Accepts TRUE/FALSE


    if is_final(leaf['level'], leaf['outcome']):
        return leaf['outcome']  # Return the outcome if the state is final

    if is_maximizer:         #if the level is maximizing
        alpha_var = float('-inf')
        for branch in leaf['branches']:
            score = alpha_beta_func(branch,alpha,beta,False)         #the next level will be minimizer
            alpha_var = max(alpha_var, score)
            alpha = max(alpha,score)   #comparing alpha with the leaf score is choosing the maximum value
            if beta<=alpha:   #breaking the loop for pruning
                break
        return alpha_var

    else:   #if the level is minimizing
        beta_var = float('inf')
        for branch in leaf['branches']:
            score = alpha_beta_func(branch,alpha,beta,True)
            beta_var = min(beta_var, score)
            alpha = min(alpha, score)  # comparing beta with the leaf score is choosing the minimizing value
            if beta <= alpha:  # breaking the loop for pruning
                break
        return beta_var
def tournament(initial_player):
    scorpion_score = 0
    sub_zero_score = 0
    game_round_results = []         #storing the results of the three rounds played


    for rounds in range(1,4):  #3 rounds will be played
        root = game_tree(5)
        match_result = alpha_beta_func(root,float('-inf'),float('inf'),initial_player==1)
        if match_result==1:
            match_winner= 'Sub-Zero'
        else:
            match_winner= 'Scorpion'

        game_round_results.append(f"Winner of Round {rounds}: {match_winner}")
        if match_result==1:
            sub_zero_score +=1
        else:
            scorpion_score+=1



        initial_player = 1 - initial_player     #alternate player starting each round


        if sub_zero_score> scorpion_score :
            tournament_winner = 'Sub-Zero'
        else:
            tournament_winner = 'Scorpion'

    print('Tournament Winner:',tournament_winner)
    print('Total Rounds Player:',3)
    for result in game_round_results:
        print(result)


initial_turn = int(input("Select initial player (0 for Scorpion, 1 for Sub-Zero): "))
if initial_turn!=0 and initial_turn!=1:
    print("Invalid input. Please enter 0 for Scorpion or 1 for Sub-Zero.")

tournament(initial_turn)

def is_final(level, tree):
    return level == 3 or len(tree) == 1     #gives True value only if level is at the last or the lenth of the tree list is 1.

def minimax_with_magic(level, is_pacman, tree):     #if pacman use magic
    if is_final(level, tree):
        return tree[0]

    mid = len(tree) // 2            #4

    left_part = minimax_with_magic(level + 1, is_pacman, tree[:mid])       #next turn is pacman's using magic and left part is [3, 6, 2, 3]
    right_part = minimax_with_magic(level + 1, is_pacman, tree[mid:])      #right part is [7, 1, 2, 0]

    return max(left_part, right_part)



def minimax_without_magic(level, is_pacman, tree, alpha, beta):  # (0,True,tree,-inf,inf)
    if is_final(level, tree):
        return tree[0]

    mid = len(tree) // 2            #4
    if is_pacman:
        left_part = minimax_without_magic(level + 1, not is_pacman, tree[:mid], alpha, beta)     #[3, 6, 2, 3] and so on
        right_part = minimax_without_magic(level + 1, not is_pacman, tree[mid:], alpha, beta)    #[7, 1, 2, 0] and so on
        max_score = max(left_part, right_part)
        alpha = max(alpha, max_score)

        if beta <= alpha:  #for pruning
          return max_score

        return max_score

    else:
        left_part = minimax_without_magic(level + 1, not is_pacman, tree[:mid], alpha, beta)      #[3, 6, 2, 3] and so on
        right_part = minimax_without_magic(level + 1, not is_pacman, tree[mid:], alpha, beta)     #[7, 1, 2, 0] and so on
        min_score = min(left_part, right_part)
        beta = min(beta, min_score)

        if beta <= alpha:
            return min_score

        return min_score
def pacman_game(cost):
    tree = [3, 6, 2, 3, 7, 1, 2, 0]

    value_with_magic = minimax_with_magic(0, True, tree) - cost
    value_without_magic = minimax_without_magic(0, True, tree, float('-inf'), float('inf'))

    if value_with_magic > value_without_magic:         #better using magic
        print(f"The new minimax value is {value_with_magic}. Pacman uses dark magic.")
    else:   #better without magic
        print(f"The minimax value is {value_without_magic}. Pacman does not use dark magic.")

pacman_game(2)
pacman_game(5)
