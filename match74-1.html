    if Dep == 3:  # Leaf nodes

        return Vals[Nod_i]

    if MXplayer:

        Minval = float('-inf')

        for i in range(2):

            Minval = max(Minval, ival)
            alpha = max(alpha, ival)

            if beta <= alpha:
                break
        return Minval
    else:
