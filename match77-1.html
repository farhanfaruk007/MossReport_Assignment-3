    if level == 5:
        return val[node_idx]

    if max_player == True:
        best_case = - math.inf

        for i in range(2):
            val_new = alpha_beta(level + 1, ((node_idx * 2)+i), False, val, alpha, beta)
            best_case = max(best_case, val_new)
            # print(best_case)
            alpha = max(alpha, best_case)
            # print(alpha)
            # print(beta)
            if beta <= alpha:
                break

        return best_case

    else:
        best_case = math.inf
        for i in range(2):


            new_val = alpha_beta(level + 1, ((node_idx * 2)+i), True, val, alpha, beta)
            best_case = min(best_case, new_val)
            beta = min(beta, best_case)
            # print(best_case)
            # print(beta)
            if beta <= alpha:
                break

        return best_case

import math

def minimax(level, nod_idx, max_player, values, alpha, beta):
    if level == 3:
        return values[nod_idx]

    if max_player == True:

        best_value = -math.inf

        for i in range(2):
            new_val = minimax((level+1), ((nod_idx * 2)+i), False, values, alpha, beta)
            best_value = max(best_value, new_val)
            # print(best_value)
            alpha = max(alpha, best_value)
            # print(alpha)

            if beta <= alpha:
                break

        return best_value

    else:

        best_value = math.inf

        for i in range(2):
            new_val = minimax((level+1), ((nod_idx * 2)+i), True, values, alpha, beta)
            best_value = min(best_value, new_val)
            beta = min(beta, best_value)
            # print(beta)
            if beta <= alpha:
                break

        return best_value




def pacman_game(c):
    values = [3, 6, 2, 3, 7, 1, 2, 0]

    root_value = minimax(0, 0, True, values, (-(math.inf)), (math.inf))

    left_val = 6 - c
    right_val = 7 - c



    if left_val > root_value or right_val > root_value:
        if left_val >= right_val:
            print(f"The new minimax value is {left_val}. Pacman goes left and uses dark magic")
        else:
            print(f"The new minimax value is {right_val}. Pacman goes right and uses dark magic")
    else:
        print(f"The minimax new_val is {root_value}. Pacman does not use dark magic.")
